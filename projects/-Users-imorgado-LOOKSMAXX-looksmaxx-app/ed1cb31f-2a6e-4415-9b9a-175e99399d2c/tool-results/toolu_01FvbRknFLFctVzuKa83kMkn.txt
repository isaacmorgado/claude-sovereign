     1→/**
     2→ * Insights Engine
     3→ *
     4→ * Processes insight definitions from insights.json and classifies them
     5→ * as strengths or areas of improvement based on user's metric scores.
     6→ *
     7→ * Logic:
     8→ * - For each insight definition, calculate average score of affected metrics
     9→ * - Weakness: if avg score < threshold → classify with severity label
    10→ * - Strength: if avg score > threshold → classify with grade label
    11→ *
    12→ * Z-Score Based Severity Logic (NEW):
    13→ * - Ideal (Green): User is inside the ideal range
    14→ * - Good (Blue): Within 1 Standard Deviation of the mean (|z| < 1)
    15→ * - Moderate (Yellow): Between 1 and 2 Standard Deviations (1 ≤ |z| < 2)
    16→ * - Severe (Red): More than 2 Standard Deviations away (|z| ≥ 2)
    17→ *
    18→ * Severity Labels (Weaknesses):
    19→ *   0-3 = "Severe"
    20→ *   3-5 = "Moderate"
    21→ *   5-7 = "Minor"
    22→ *
    23→ * Grade Labels (Strengths):
    24→ *   8-9   = "Good"
    25→ *   9-9.5 = "Excellent"
    26→ *   9.5-10 = "Ideal"
    27→ */

  1402→/**
  1403→ * Get the appropriate metric configuration for a given gender and ethnicity
  1404→ * Falls back to base config if no override exists
  1405→ */
  1406→export function getMetricConfig(
  1407→  metricId: string,
  1408→  gender: Gender = 'male',
  1409→  ethnicity: Ethnicity = 'white'
  1410→): MasterMetricConfig | undefined {
  1411→  const key = `${gender}_${ethnicity}`;
  1412→  const override = ETHNICITY_OVERRIDES[key]?.[metricId];
  1413→  const baseConfig = MASTER_SCORING_DB[metricId];
  1414→
  1415→  if (!baseConfig) return undefined;
  1416→
  1417→  // Merge override with base config
  1418→  if (override) {
  1419→    return {
  1420→      ...baseConfig,
  1421→      ...override,
  1422→      flaws: {
  1423→        ...baseConfig.flaws,
  1424→        ...override.flaws
  1425→      }
  1426→    } as MasterMetricConfig;
  1427→  }
  1428→
  1429→  return baseConfig;
  1430→}
  1431→
  1432→// ============================================
  1433→// Z-SCORE SEVERITY FUNCTIONS
  1434→// ============================================
  1435→
  1436→/**
  1437→ * Calculate Z-score for a given value
  1438→ * Z = (value - mean) / std_dev
  1439→ */
  1440→export function calculateZScore(value: number, mean: number, stdDev: number): number {
  1441→  if (stdDev === 0) return 0;
  1442→  return Math.abs((value - mean) / stdDev);
  1443→}
  1444→
  1445→/**
  1446→ * Determine severity level based on Z-score and ideal range
  1447→ * - Ideal: Value is within ideal range
  1448→ * - Good: |z| < 1 (within 1 standard deviation)
  1449→ * - Moderate: 1 ≤ |z| < 2 (between 1-2 standard deviations)
  1450→ * - Severe: |z| ≥ 2 (more than 2 standard deviations away)
  1451→ */
  1452→export function getSeverityFromZScore(
  1453→  value: number,
  1454→  config: MasterMetricConfig
  1455→): { severity: SeverityLevel; zScore: number; isInIdeal: boolean } {
  1456→  const { ideal, mean, std_dev } = config;
  1457→  const isInIdeal = value >= ideal[0] && value <= ideal[1];
  1458→  const zScore = calculateZScore(value, mean, std_dev);
  1459→
  1460→  let severity: SeverityLevel;
  1461→  if (isInIdeal) {
  1462→    severity = 'ideal';
  1463→  } else if (zScore < 1) {
  1464→    severity = 'good';
  1465→  } else if (zScore < 2) {
  1466→    severity = 'moderate';
  1467→  } else {
  1468→    severity = 'severe';
  1469→  }
  1470→
  1471→  return { severity, zScore, isInIdeal };
  1472→}
  1473→
  1474→/**
  1475→ * Get severity for a metric value with ethnicity/gender awareness
  1476→ */
  1477→export function getSeverityForMetric(
  1478→  metricId: string,
  1479→  value: number,
  1480→  gender: Gender = 'male',
  1481→  ethnicity: Ethnicity = 'white'
  1482→): { severity: SeverityLevel; zScore: number; isInIdeal: boolean; config: MasterMetricConfig } | null {
  1483→  const config = getMetricConfig(metricId, gender, ethnicity);
  1484→  if (!config) return null;
  1485→
  1486→  const result = getSeverityFromZScore(value, config);
  1487→  return { ...result, config };
  1488→}

  2338→/**
  2339→ * Get weakness severity based on average score
  2340→ * 0-3 = "Severe", 3-5 = "Moderate", 5-7 = "Minor"
  2341→ */
  2342→function getWeaknessSeverity(avgScore: number): WeaknessSeverity {
  2343→  if (avgScore <= 3) return 'severe';
  2344→  if (avgScore <= 5) return 'moderate';
  2345→  return 'minor';
  2346→}
  2347→
  2348→/**
  2349→ * Get strength grade based on average score
  2350→ * 8-9 = "Good", 9-9.5 = "Excellent", 9.5-10 = "Ideal"
  2351→ */
  2352→function getStrengthGrade(avgScore: number): StrengthGrade {
  2353→  if (avgScore >= 9.5) return 'ideal';
  2354→  if (avgScore >= 9) return 'excellent';
  2355→  return 'good';
  2356→}

  2358→/**
  2359→ * Find metric by display name from measurements array
  2360→ */
  2361→function findMetricByName(
  2362→  metricName: string,
  2363→  measurements: FaceIQScoreResult[]
  2364→): FaceIQScoreResult | undefined {
  2365→  // First try direct ID mapping
  2366→  const metricId = METRIC_NAME_TO_ID[metricName];
  2367→  if (metricId) {
  2368→    return measurements.find(m => m.metricId === metricId);
  2369→  }
  2370→
  2371→  // Fallback: try fuzzy matching on name
  2372→  const normalizedName = metricName.toLowerCase().replace(/[^a-z0-9]/g, '');
  2373→  return measurements.find(m => {
  2374→    const normalizedMetricName = m.name.toLowerCase().replace(/[^a-z0-9]/g, '');
  2375→    return normalizedMetricName.includes(normalizedName) ||
  2376→           normalizedName.includes(normalizedMetricName);
  2377→  });
  2378→}
  2379→
  2380→/**
  2381→ * Check if a metric should be considered a "false positive" weakness.
  2382→ * This handles directional/dimorphic metrics where low scores don't mean weakness.
  2383→ *
  2384→ * Example: Canthal Tilt of 3° gets a low score (not in 6-8° ideal range)
  2385→ * but is still a POSITIVE trait, not a weakness.
  2386→ */
  2387→function isFalsePositiveWeakness(metric: MatchedMetric): boolean {
  2388→  const config = FACEIQ_METRICS[metric.metricId];
  2389→  if (!config) return false;
  2390→
  2391→  // Check if the value is actually acceptable based on polarity
  2392→  const { acceptable } = isValueAcceptable(metric.value, config);
  2393→  return acceptable;
  2394→}

  2396→/**
  2397→ * Process all insight definitions against user's measurements
  2398→ */
  2399→export function classifyInsights(
  2400→  measurements: FaceIQScoreResult[],
  2401→  insightDefinitions: InsightDefinition[] = INSIGHTS_DEFINITIONS
  2402→): InsightClassificationResult {
  2403→  const strengths: ClassifiedStrength[] = [];
  2404→  const weaknesses: ClassifiedWeakness[] = [];
  2405→
  2406→  for (const insight of insightDefinitions) {
  2407→    const { metrics, threshold } = insight.severity_logic;
  2408→
  2409→    // Find matching measurements for this insight
  2410→    const matchedMetrics: MatchedMetric[] = [];
  2411→    for (const metricName of metrics) {
  2412→      const measurement = findMetricByName(metricName, measurements);
  2413→      if (measurement) {
  2414→        matchedMetrics.push({
  2415→          metricId: measurement.metricId,
  2416→          metricName: measurement.name,
  2417→          score: measurement.score,
  2418→          value: measurement.value,
  2419→          idealMin: measurement.idealMin,
  2420→          idealMax: measurement.idealMax,
  2421→          unit: measurement.unit,
  2422→          category: measurement.category,
  2423→        });
  2424→      }
  2425→    }
  2426→
  2427→    // Skip if no metrics matched (need at least 1)
  2428→    if (matchedMetrics.length === 0) continue;
  2429→
  2430→    // Calculate average score
  2431→    const avgScore = matchedMetrics.reduce((sum, m) => sum + m.score, 0) / matchedMetrics.length;
  2432→
  2433→    // Classify based on type and threshold
  2434→    if (insight.type === 'weakness') {
  2435→      // Weakness: avg < threshold → add to areas of improvement
  2436→      if (avgScore < threshold) {
  2437→        // ========== FALSE POSITIVE CHECK ==========
  2438→        // Before adding to weaknesses, check if ANY of the matched metrics
  2439→        // are actually in an "acceptable" zone based on their polarity.
  2440→        // If so, this is a false positive and should be skipped or reclassified.
  2441→        const falsePositiveMetrics = matchedMetrics.filter(isFalsePositiveWeakness);
  2442→
  2443→        // If ALL matched metrics are false positives, skip this weakness entirely
  2444→        if (falsePositiveMetrics.length === matchedMetrics.length) {
  2445→          // This is a false positive weakness - the values are actually acceptable
  2446→          // Consider adding to strengths if scores are decent
  2447→          if (avgScore >= 6) {
  2448→            strengths.push({
  2449→              insightId: `${insight.id}_acceptable`,
  2450→              title: insight.title.replace('Negative', 'Neutral').replace('Weak', 'Moderate'),
  2451→              description: `While not in the peak ideal range, the measured values are within acceptable limits.`,
  2452→              grade: 'good',
  2453→              gradeLabel: 'Acceptable',
  2454→              avgScore,
  2455→              matchedMetrics,
  2456→            });
  2457→          }
  2458→          continue; // Skip adding as weakness
  2459→        }
  2460→
  2461→        // If SOME metrics are false positives, filter them out
  2462→        const trueWeaknessMetrics = matchedMetrics.filter(m => !isFalsePositiveWeakness(m));
  2463→        if (trueWeaknessMetrics.length === 0) continue;
  2464→
  2465→        // Recalculate average with only true weakness metrics
  2466→        const trueAvgScore = trueWeaknessMetrics.reduce((sum, m) => sum + m.score, 0) / trueWeaknessMetrics.length;
  2467→
  2468→        // Only add as weakness if true average is still below threshold
  2469→        if (trueAvgScore < threshold) {
  2470→          const severity = getWeaknessSeverity(trueAvgScore);
  2471→          const labels = insight.content.severity_labels;
  2472→          const severityLabel = labels?.[severity] || severity;
  2473→
  2474→          weaknesses.push({
  2475→            insightId: insight.id,
  2476→            title: insight.title,
  2477→            description: insight.content.description,
  2478→            severity,
  2479→            severityLabel,
  2480→            avgScore: trueAvgScore,
  2481→            matchedMetrics: trueWeaknessMetrics,
  2482→          });
  2483→        }
  2484→      }
  2485→    } else {
  2486→      // Strength: avg > threshold → add to key strengths
  2487→      if (avgScore > threshold) {
  2488→        const grade = getStrengthGrade(avgScore);
  2489→        const labels = insight.content.grade_labels;
  2490→        const gradeLabel = labels?.[grade] || grade;
  2491→
  2492→        strengths.push({
  2493→          insightId: insight.id,
  2494→          title: insight.title,
  2495→          description: insight.content.description,
  2496→          grade,
  2497→          gradeLabel,
  2498→          avgScore,
  2499→          matchedMetrics,
  2500→        });
  2501→      }
  2502→    }
  2503→  }
  2504→
  2505→  // Sort by score (weaknesses by lowest score first, strengths by highest first)
  2506→  weaknesses.sort((a, b) => a.avgScore - b.avgScore);
  2507→  strengths.sort((a, b) => b.avgScore - a.avgScore);
  2508→
  2509→  return { strengths, weaknesses };
  2510→}
