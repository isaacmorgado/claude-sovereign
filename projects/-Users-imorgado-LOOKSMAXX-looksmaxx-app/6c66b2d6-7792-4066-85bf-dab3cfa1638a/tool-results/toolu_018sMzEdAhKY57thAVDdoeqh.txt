   714→function generateIllustration(metricId: string, landmarks: LandmarkPoint[]): Ratio['illustration'] {
   715→  const landmarkMap: Record<string, LandmarkPoint> = {};
   716→  landmarks.forEach(l => { landmarkMap[l.id] = l; });
   717→
   718→  const config = ILLUSTRATION_CONFIGS[metricId];
   719→  const points: Record<string, { type: 'landmark'; landmarkId: string }> = {};
   720→  const lines: Record<string, { from: string; to: string; color: string; label?: string; labelPosition?: 'start' | 'middle' | 'end' }> = {};
   721→
   722→  if (config) {
   723→    config.points.forEach((pointId, i) => {
   724→      if (landmarkMap[pointId]) {
   725→        points[`point_${i}`] = { type: 'landmark', landmarkId: pointId };
   726→      }
   727→    });
   728→    config.lines.forEach((line, i) => {
   729→      lines[`line_${i}`] = {
   730→        from: line.from,
   731→        to: line.to,
   732→        color: line.color || '#67e8f9',
   733→        label: line.label,
   734→        labelPosition: line.labelPosition,
   735→      };
   736→    });
   737→  }
   738→
   739→  return { points, lines };
   740→}

   867→function generateStrengthsFromAnalysis(analysis: HarmonyAnalysis): Strength[] {
   868→  const strengths: Strength[] = [];
   869→  const usedMetricIds = new Set<string>();
   870→
   871→  // First pass: Create grouped strengths from related high-scoring metrics
   872→  Object.entries(STRENGTH_GROUPINGS).forEach(([groupId, groupConfig]) => {
   873→    // Find all high-scoring measurements that belong to this group
   874→    const groupMeasurements = analysis.measurements.filter(m =>
   875→      groupConfig.metrics.includes(m.metricId) &&
   876→      m.score >= 7.5 &&
   877→      !usedMetricIds.has(m.metricId)
   878→    );
   879→
   880→    // Only create a grouped strength if we have 2+ metrics with good scores
   881→    if (groupMeasurements.length >= 2) {
   882→      const avgScore = groupMeasurements.reduce((sum, m) => sum + m.score, 0) / groupMeasurements.length;
   883→      const bestQuality = avgScore >= 9 ? 'excellent' : avgScore >= 8 ? 'good' : 'below_average';
   884→
   885→      // Mark these metrics as used
   886→      groupMeasurements.forEach(m => usedMetricIds.add(m.metricId));
   887→
   888→      // Create grouped strength with multiple contributing ratios
   889→      strengths.push({
   890→        id: `strength_group_${groupId}`,
   891→        strengthName: groupConfig.name,
   892→        summary: groupConfig.description,
   893→        avgScore,
   894→        qualityLevel: bestQuality,
   895→        categoryName: groupConfig.category,
   896→        responsibleRatios: groupMeasurements.map(m => ({
   897→          ratioName: m.name,
   898→          ratioId: m.metricId,
   899→          score: m.score,
   900→          value: m.value,
   901→          idealMin: m.idealMin,
   902→          idealMax: m.idealMax,
   903→          unit: m.unit,
   904→          category: m.category,
   905→        })),
   906→      });
   907→    }
   908→  });
   909→
   910→  // Second pass: Add remaining individual strengths that weren't grouped
   911→  analysis.strengths.forEach((s, i) => {
   912→    if (usedMetricIds.has(s.metricId)) return; // Skip if already used in a group
   913→
   914→    const matchingMeasurement = analysis.measurements.find(m => m.metricId === s.metricId);
   915→
   916→    strengths.push({
   917→      id: `strength_${i}`,
   918→      strengthName: s.metricName,
   919→      summary: s.reasoning,
   920→      avgScore: matchingMeasurement?.score || 8,
   921→      qualityLevel: s.qualityTier,
   922→      categoryName: s.category,
   923→      responsibleRatios: [{
   924→        ratioName: s.metricName,
   925→        ratioId: s.metricId,
   926→        score: matchingMeasurement?.score || 8,
   927→        value: s.value,
   928→        idealMin: matchingMeasurement?.idealMin || 0,
   929→        idealMax: matchingMeasurement?.idealMax || 1,
   930→        unit: matchingMeasurement?.unit || 'ratio',
   931→        category: s.category,
   932→      }],
   933→    });
   934→  });
   935→
   936→  // Sort by average score (highest first) then by number of contributing ratios
   937→  return strengths.sort((a, b) => {
   938→    if (b.responsibleRatios.length !== a.responsibleRatios.length) {
   939→      return b.responsibleRatios.length - a.responsibleRatios.length;
   940→    }
   941→    return b.avgScore - a.avgScore;
   942→  });
   943→}
