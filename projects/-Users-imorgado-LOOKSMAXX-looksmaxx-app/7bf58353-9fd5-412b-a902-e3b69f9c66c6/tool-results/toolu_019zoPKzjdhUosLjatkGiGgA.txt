    46→@router.get("/categories", response_model=List[CategoryResponse])
    47→async def list_categories(db: AsyncSession = Depends(get_db)):
    48→    """List all active forum categories with sub-forums and post counts.
    49→
    50→    Optimized to use only 2 queries instead of N+1 pattern.
    51→    """
    52→    # Query 1: Get all categories with sub-forums
    53→    query = (
    54→        select(ForumIssueCategory)
    55→        .options(selectinload(ForumIssueCategory.sub_forums))
    56→        .where(ForumIssueCategory.is_active == True)
    57→        .order_by(ForumIssueCategory.display_order)
    58→    )
    59→    result = await db.execute(query)
    60→    categories = result.scalars().all()
    61→
    62→    # Query 2: Get ALL sub-forum post counts in ONE query (batched)
    63→    sf_counts_query = (
    64→        select(
    65→            ForumPost.sub_forum_id,
    66→            func.count(ForumPost.id).label('post_count')
    67→        )
    68→        .where(
    69→            ForumPost.is_deleted == False,
    70→            ForumPost.is_approved == True
    71→        )
    72→        .group_by(ForumPost.sub_forum_id)
    73→    )
    74→    sf_counts_result = await db.execute(sf_counts_query)
    75→    sf_counts = {row[0]: row[1] for row in sf_counts_result}
    76→
    77→    # Build response using pre-fetched counts (no additional queries)
    78→    response = []
    79→    for cat in categories:
    80→        cat_post_count = 0
    81→        sub_forums = []
    82→
    83→        for sf in sorted(cat.sub_forums, key=lambda x: x.display_order):
    84→            if sf.is_active:
    85→                sf_post_count = sf_counts.get(sf.id, 0)
    86→                cat_post_count += sf_post_count
    87→
    88→                sub_forums.append(SubForumResponse(
    89→                    id=sf.id,
    90→                    name=sf.name,
    91→                    slug=sf.slug,
    92→                    description=sf.description,
    93→                    icon=sf.icon,
    94→                    display_order=sf.display_order,
    95→                    post_count=sf_post_count
    96→                ))
    97→
    98→        response.append(CategoryResponse(
    99→            id=cat.id,
   100→            name=cat.name,
   101→            slug=cat.slug,
   102→            description=cat.description,
   103→            icon=cat.icon,
   104→            display_order=cat.display_order,
   105→            post_count=cat_post_count,
   106→            sub_forums=sub_forums
   107→        ))
   108→
   109→    return response

   112→@router.get("/categories/{slug}", response_model=CategoryResponse)
   113→async def get_category(
   114→    slug: str,
   115→    db: AsyncSession = Depends(get_db)
   116→):
   117→    """Get a single category by slug with its sub-forums.
   118→
   119→    Optimized to use only 2 queries instead of N+1 pattern.
   120→    """
   121→    # Query 1: Get category with sub-forums
   122→    query = (
   123→        select(ForumIssueCategory)
   124→        .options(selectinload(ForumIssueCategory.sub_forums))
   125→        .where(ForumIssueCategory.slug == slug, ForumIssueCategory.is_active == True)
   126→    )
   127→    result = await db.execute(query)
   128→    cat = result.scalar_one_or_none()
   129→
   130→    if not cat:
   131→        raise HTTPException(status_code=404, detail="Category not found")
   132→
   133→    # Get sub-forum IDs for this category
   134→    sf_ids = [sf.id for sf in cat.sub_forums if sf.is_active]
   135→
   136→    # Query 2: Get post counts for all sub-forums in this category (batched)
   137→    sf_counts = {}
   138→    if sf_ids:
   139→        sf_counts_query = (
   140→            select(
   141→                ForumPost.sub_forum_id,
   142→                func.count(ForumPost.id).label('post_count')
   143→            )
   144→            .where(
   145→                ForumPost.sub_forum_id.in_(sf_ids),
   146→                ForumPost.is_deleted == False,
   147→                ForumPost.is_approved == True
   148→            )
   149→            .group_by(ForumPost.sub_forum_id)
   150→        )
   151→        sf_counts_result = await db.execute(sf_counts_query)
   152→        sf_counts = {row[0]: row[1] for row in sf_counts_result}
   153→
   154→    # Build sub-forums list using pre-fetched counts
   155→    cat_post_count = 0
   156→    sub_forums = []
   157→    for sf in sorted(cat.sub_forums, key=lambda x: x.display_order):
   158→        if sf.is_active:
   159→            sf_post_count = sf_counts.get(sf.id, 0)
   160→            cat_post_count += sf_post_count
   161→
   162→            sub_forums.append(SubForumResponse(
   163→                id=sf.id,
   164→                name=sf.name,
   165→                slug=sf.slug,
   166→                description=sf.description,
   167→                icon=sf.icon,
   168→                display_order=sf.display_order,
   169→                post_count=sf_post_count
   170→            ))
   171→
   172→    return CategoryResponse(
   173→        id=cat.id,
   174→        name=cat.name,
   175→        slug=cat.slug,
   176→        description=cat.description,
   177→        icon=cat.icon,
   178→        display_order=cat.display_order,
   179→        post_count=cat_post_count,
   180→        sub_forums=sub_forums
   181→    )

   186→@router.get("/search", response_model=ForumSearchResponse)
   187→async def search_forum(
   188→    q: str = Query(..., min_length=2, max_length=100, description="Search query"),
   189→    type: str = Query("all", description="Search type: posts, comments, or all"),
   190→    limit: int = Query(20, ge=1, le=50),
   191→    offset: int = Query(0, ge=0),
   192→    db: AsyncSession = Depends(get_db)
   193→):
   194→    """Search forum posts and comments using ILIKE pattern matching.
   195→
   196→    - Search by title, content for posts
   197→    - Search by content for comments
   198→    - Results sorted by vote_count DESC, created_at DESC
   199→    """
   200→    search_pattern = f"%{q}%"
   201→    results = []
   202→    total_count = 0
   203→
   204→    if type in ("posts", "all"):
   205→        # Search posts by title and content
   206→        post_query = (
   207→            select(ForumPost)
   208→            .options(
   209→                selectinload(ForumPost.author),
   210→                selectinload(ForumPost.sub_forum).selectinload(ForumSubForum.issue_category)
   211→            )
   212→            .where(
   213→                ForumPost.is_deleted == False,
   214→                ForumPost.is_approved == True,
   215→                or_(
   216→                    ForumPost.title.ilike(search_pattern),
   217→                    ForumPost.content.ilike(search_pattern)
   218→                )
   219→            )
   220→            .order_by(ForumPost.vote_count.desc(), ForumPost.created_at.desc())
   221→        )
   222→
   223→        # Count posts
   224→        post_count_query = (
   225→            select(func.count(ForumPost.id))
   226→            .where(
   227→                ForumPost.is_deleted == False,
   228→                ForumPost.is_approved == True,
   229→                or_(
   230→                    ForumPost.title.ilike(search_pattern),
   231→                    ForumPost.content.ilike(search_pattern)
   232→                )
   233→            )
   234→        )
   235→        post_count_result = await db.execute(post_count_query)
   236→        post_count = post_count_result.scalar() or 0
   237→
   238→        if type == "posts":
   239→            total_count = post_count
   240→            post_query = post_query.offset(offset).limit(limit)
   241→        else:
   242→            total_count += post_count
   243→            # For "all", we get all posts first then merge with comments
   244→            post_query = post_query.limit(limit * 2)  # Get more to merge
   245→
   246→        post_result = await db.execute(post_query)
   247→        posts = post_result.scalars().all()
   248→
   249→        for post in posts:
   250→            content_preview = post.content[:200] + "..." if len(post.content) > 200 else post.content
   251→            results.append(ForumSearchResult(
   252→                id=post.id,
   253→                type=SearchResultType.POST,
   254→                title=post.title,
   255→                content_preview=content_preview,
   256→                post_id=None,
   257→                post_title=None,
   258→                author=PostAuthor(id=post.author.id, username=post.author.username),
   259→                vote_count=post.vote_count,
   260→                category_slug=post.sub_forum.issue_category.slug if post.sub_forum and post.sub_forum.issue_category else "",
   261→                sub_forum_slug=post.sub_forum.slug if post.sub_forum else "",
   262→                created_at=post.created_at
   263→            ))
   264→
   265→    if type in ("comments", "all"):
   266→        # Search comments by content
   267→        comment_query = (
   268→            select(ForumComment)
   269→            .options(
   270→                selectinload(ForumComment.author),
   271→                selectinload(ForumComment.post).selectinload(ForumPost.sub_forum).selectinload(ForumSubForum.issue_category)
   272→            )
   273→            .where(
   274→                ForumComment.is_deleted == False,
   275→                ForumComment.content.ilike(search_pattern)
   276→            )
   277→            .order_by(ForumComment.vote_count.desc(), ForumComment.created_at.desc())
   278→        )
   279→
   280→        # Count comments
   281→        comment_count_query = (
   282→            select(func.count(ForumComment.id))
   283→            .where(
   284→                ForumComment.is_deleted == False,
   285→                ForumComment.content.ilike(search_pattern)
   286→            )
   287→        )
   288→        comment_count_result = await db.execute(comment_count_query)
   289→        comment_count = comment_count_result.scalar() or 0
   290→
   291→        if type == "comments":
   292→            total_count = comment_count
   293→            comment_query = comment_query.offset(offset).limit(limit)
   294→        else:
   295→            total_count += comment_count
   296→            comment_query = comment_query.limit(limit * 2)
   297→
   298→        comment_result = await db.execute(comment_query)
   299→        comments = comment_result.scalars().all()
   300→
   301→        for comment in comments:
   302→            content_preview = comment.content[:200] + "..." if len(comment.content) > 200 else comment.content
   303→            post = comment.post
   304→            if post and post.sub_forum:
   305→                category_slug = post.sub_forum.issue_category.slug if post.sub_forum.issue_category else ""
   306→                sub_forum_slug = post.sub_forum.slug
   307→            else:
   308→                category_slug = ""
   309→                sub_forum_slug = ""
   310→
   311→            results.append(ForumSearchResult(
   312→                id=comment.id,
   313→                type=SearchResultType.COMMENT,
   314→                title=None,
   315→                content_preview=content_preview,
   316→                post_id=comment.post_id,
   317→                post_title=post.title if post else None,
   318→                author=PostAuthor(id=comment.author.id, username=comment.author.username),
   319→                vote_count=comment.vote_count,
   320→                category_slug=category_slug,
   321→                sub_forum_slug=sub_forum_slug,
   322→                created_at=comment.created_at
   323→            ))
   324→
   325→    # For "all" type, sort merged results by vote_count and created_at, then paginate
   326→    if type == "all":
   327→        results.sort(key=lambda x: (-x.vote_count, x.created_at), reverse=False)
   328→        results.sort(key=lambda x: x.vote_count, reverse=True)
   329→        results = results[offset:offset + limit]
   330→
   331→    return ForumSearchResponse(
   332→        results=results,
   333→        total_count=total_count,
   334→        query=q,
   335→        search_type=type,
   336→        has_more=offset + limit < total_count
   337→    )

   340→@router.get("/recommended", response_model=List[RecommendedForumResponse])
   341→async def get_recommended_forums(
   342→    flaws: str = Query(..., description="Comma-separated list of flaw IDs"),
   343→    db: AsyncSession = Depends(get_db)
   344→):
   345→    """Get recommended forums based on detected flaws."""
   346→    flaw_list = [f.strip() for f in flaws.split(",") if f.strip()]
   347→
   348→    if not flaw_list:
   349→        return []
   350→
   351→    # Find mappings for these flaws
   352→    query = (
   353→        select(FlawToForumMapping)
   354→        .options(selectinload(FlawToForumMapping.issue_category))
   355→        .where(FlawToForumMapping.flaw_id.in_(flaw_list))
   356→        .order_by(FlawToForumMapping.priority)
   357→    )
   358→
   359→    result = await db.execute(query)
   360→    mappings = result.scalars().all()
   361→
   362→    # Group by category
   363→    category_flaws = {}
   364→    for mapping in mappings:
   365→        cat_id = mapping.issue_category_id
   366→        if cat_id not in category_flaws:
   367→            category_flaws[cat_id] = {
   368→                "category": mapping.issue_category,
   369→                "flaws": [],
   370→                "priority": mapping.priority
   371→            }
   372→        category_flaws[cat_id]["flaws"].append(mapping.flaw_id)
   373→        # Use lowest priority (1 = primary)
   374→        if mapping.priority < category_flaws[cat_id]["priority"]:
   375→            category_flaws[cat_id]["priority"] = mapping.priority
   376→
   377→    # Build response sorted by priority
   378→    response = []
   379→    for cat_data in sorted(category_flaws.values(), key=lambda x: x["priority"]):
   380→        cat = cat_data["category"]
   381→        response.append(RecommendedForumResponse(
   382→            category=CategoryListResponse(
   383→                id=cat.id,
   384→                name=cat.name,
   385→                slug=cat.slug,
   386→                description=cat.description,
   387→                icon=cat.icon,
   388→                display_order=cat.display_order,
   389→                post_count=0
   390→            ),
   391→            matched_flaws=cat_data["flaws"],
   392→            priority=cat_data["priority"]
   393→        ))
   394→
   395→    return response

   508→@router.get("/sub-forums/{sub_forum_id}/posts", response_model=PostListResponse)
   509→async def list_sub_forum_posts(
   510→    sub_forum_id: UUID,
   511→    sort: SortOrder = Query(SortOrder.HOT),
   512→    limit: int = Query(PAGE_SIZE, ge=1, le=50),
   513→    offset: int = Query(0, ge=0),
   514→    current_user: Optional[User] = Depends(get_current_user_optional),
   515→    db: AsyncSession = Depends(get_db),
   516→):
   517→    """List posts in a sub-forum with sorting and pagination."""
   518→    # Verify sub-forum exists and get category
   519→    sf_result = await db.execute(
   520→        select(ForumSubForum)
   521→        .options(selectinload(ForumSubForum.issue_category))
   522→        .where(ForumSubForum.id == sub_forum_id)
   523→    )
   524→    sub_forum = sf_result.scalar_one_or_none()
   525→    if not sub_forum:
   526→        raise HTTPException(status_code=404, detail="Sub-forum not found")
   527→
   528→    category_slug = sub_forum.issue_category.slug
   529→
   530→    # Base query
   531→    query = (
   532→        select(ForumPost)
   533→        .options(selectinload(ForumPost.author))
   534→        .where(
   535→            ForumPost.sub_forum_id == sub_forum_id,
   536→            ForumPost.is_deleted == False,
   537→            ForumPost.is_approved == True
   538→        )
   539→    )
   540→
   541→    # Sorting - pinned always first
   542→    if sort == SortOrder.NEW:
   543→        query = query.order_by(ForumPost.is_pinned.desc(), ForumPost.created_at.desc())
   544→    elif sort == SortOrder.TOP:
   545→        query = query.order_by(ForumPost.is_pinned.desc(), ForumPost.vote_count.desc())
   546→    else:  # HOT - combination of votes and recency
   547→        query = query.order_by(ForumPost.is_pinned.desc(), ForumPost.vote_count.desc(), ForumPost.created_at.desc())
   548→
   549→    # Get total count
   550→    count_query = (
   551→        select(func.count(ForumPost.id))
   552→        .where(
   553→            ForumPost.sub_forum_id == sub_forum_id,
   554→            ForumPost.is_deleted == False,
   555→            ForumPost.is_approved == True
   556→        )
   557→    )
   558→    total_result = await db.execute(count_query)
   559→    total_count = total_result.scalar() or 0
   560→
   561→    # Paginate
   562→    query = query.offset(offset).limit(limit)
   563→    result = await db.execute(query)
   564→    posts = result.scalars().all()
   565→
   566→    # Get user votes if authenticated
   567→    user_votes = {}
   568→    if current_user:
   569→        post_ids = [p.id for p in posts]
   570→        if post_ids:
   571→            votes_query = select(ForumVote).where(
   572→                ForumVote.user_id == current_user.id,
   573→                ForumVote.target_type == TargetType.POST,

   825→@router.post("/posts/{post_id}/vote", response_model=VoteResponse)
   826→@limiter.limit(ForumRateLimits.VOTES)
   827→async def vote_on_post(
   828→    request: Request,
   829→    post_id: UUID,
   830→    data: VoteRequest,
   831→    current_user: User = Depends(get_current_user),
   832→    db: AsyncSession = Depends(get_db),
   833→):
   834→    """Vote on a post (upvote/downvote). Voting again removes vote, changing vote type switches it.
   835→
   836→    Rate limit: 30 votes per minute per user.
   837→    """
   838→    # Verify post exists
   839→    post_result = await db.execute(
   840→        select(ForumPost).where(ForumPost.id == post_id, ForumPost.is_deleted == False)
   841→    )
   842→    post = post_result.scalar_one_or_none()
   843→    if not post:
   844→        raise HTTPException(status_code=404, detail="Post not found")
   845→
   846→    # Prevent self-voting
   847→    if post.author_id == current_user.id:
   848→        raise HTTPException(status_code=400, detail="You cannot vote on your own post")
   849→
   850→    # Check existing vote
   851→    vote_result = await db.execute(
   852→        select(ForumVote).where(
   853→            ForumVote.user_id == current_user.id,
   854→            ForumVote.target_type == TargetType.POST,
   855→            ForumVote.target_id == post_id
   856→        )
   857→    )
   858→    existing_vote = vote_result.scalar_one_or_none()
   859→
   860→    new_user_vote = None
   861→    if existing_vote:
   862→        # Robustly get value (handle if DB returns string or Enum)
   863→        current_val = existing_vote.vote_type.value if hasattr(existing_vote.vote_type, 'value') else existing_vote.vote_type
   864→        # Request is from Pydantic Schema Enum
   865→        request_val = data.vote_type.value
   866→
   867→        if current_val == request_val:
   868→            # Same vote - remove it
   869→            await db.delete(existing_vote)
   870→            post.vote_count += -1 if request_val == "up" else 1
   871→        else:
   872→            # Different vote - switch it
   873→            # Ensure we assign the Model Enum (or string if that's what it wants, but Model Enum is safer if Typed)
   874→            existing_vote.vote_type = VoteType(request_val)
   875→            # Swing by 2 (remove old, add new)
   876→            post.vote_count += 2 if request_val == "up" else -2
   877→            new_user_vote = data.vote_type
   878→    else:
   879→        # New vote
   880→        vote = ForumVote(
   881→            user_id=current_user.id,
   882→            target_type=TargetType.POST,
   883→            target_id=post_id,
   884→            # Explicitly cast to Model Enum
   885→            vote_type=VoteType(data.vote_type.value)
   886→        )
   887→        db.add(vote)
   888→        post.vote_count += 1 if data.vote_type.value == "up" else -1
   889→        new_user_vote = data.vote_type
   890→
   891→    await db.commit()
   892→
   893→    return VoteResponse(success=True, new_vote_count=post.vote_count, user_vote=new_user_vote)

  1622→def _build_comment_tree(comments: List[ForumComment], user_votes: dict) -> List[CommentResponse]:
  1623→    """Build a tree structure from flat comment list."""
  1624→    comment_map = {}
  1625→    roots = []
  1626→
  1627→    # First pass: create all response objects
  1628→    for c in comments:
  1629→        comment_map[c.id] = _comment_to_response(c, user_votes.get(c.id))
  1630→
  1631→    # Second pass: build tree
  1632→    for c in comments:
  1633→        response = comment_map[c.id]
  1634→        if c.parent_id and c.parent_id in comment_map:
  1635→            comment_map[c.parent_id].replies.append(response)
  1636→        else:
  1637→            roots.append(response)
  1638→
  1639→    return roots
