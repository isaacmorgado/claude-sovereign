     1→'use client';
     2→
     3→import { useState, useCallback, useEffect, useRef } from 'react';
     4→import { useRouter } from 'next/navigation';
     5→import { Search, X, Filter, ChevronDown } from 'lucide-react';
     6→import { motion, AnimatePresence } from 'framer-motion';
     7→
     8→interface SearchBarProps {
     9→  initialQuery?: string;
    10→  initialType?: 'posts' | 'comments' | 'all';
    11→  onSearch?: (query: string, type: 'posts' | 'comments' | 'all') => void;
    12→  showFilters?: boolean;
    13→  placeholder?: string;
    14→  className?: string;
    15→  autoFocus?: boolean;
    16→}
    17→
    18→export function SearchBar({
    19→  initialQuery = '',
    20→  initialType = 'all',
    21→  onSearch,
    22→  showFilters = true,
    23→  placeholder = 'Search posts and comments...',
    24→  className = '',
    25→  autoFocus = false,
    26→}: SearchBarProps) {
    27→  const router = useRouter();
    28→  const [query, setQuery] = useState(initialQuery);
    29→  const [searchType, setSearchType] = useState<'posts' | 'comments' | 'all'>(initialType);
    30→  const [showFilterDropdown, setShowFilterDropdown] = useState(false);
    31→  const inputRef = useRef<HTMLInputElement>(null);
    32→  const dropdownRef = useRef<HTMLDivElement>(null);
    33→  const debounceTimerRef = useRef<NodeJS.Timeout | null>(null);
    34→
    35→  // Close dropdown when clicking outside
    36→  useEffect(() => {
    37→    function handleClickOutside(event: MouseEvent) {
    38→      if (dropdownRef.current && !dropdownRef.current.contains(event.target as Node)) {
    39→        setShowFilterDropdown(false);
    40→      }
    41→    }
    42→    document.addEventListener('mousedown', handleClickOutside);
    43→    return () => document.removeEventListener('mousedown', handleClickOutside);
    44→  }, []);
    45→
    46→  // Handle search with debounce
    47→  const handleSearch = useCallback(() => {
    48→    if (query.trim().length < 2) return;
    49→
    50→    if (onSearch) {
    51→      onSearch(query.trim(), searchType);
    52→    } else {
    53→      // Navigate to search results page
    54→      const params = new URLSearchParams();
    55→      params.set('q', query.trim());
    56→      if (searchType !== 'all') params.set('type', searchType);
    57→      router.push(`/forum/search?${params.toString()}`);
    58→    }
    59→  }, [query, searchType, onSearch, router]);
    60→
    61→  // Debounced search (300ms)
    62→  const handleInputChange = useCallback((value: string) => {
    63→    setQuery(value);
    64→
    65→    if (debounceTimerRef.current) {
    66→      clearTimeout(debounceTimerRef.current);
    67→    }
    68→
    69→    // Only auto-search if we have an onSearch callback (inline search)
    70→    if (onSearch && value.trim().length >= 2) {
    71→      debounceTimerRef.current = setTimeout(() => {
    72→        onSearch(value.trim(), searchType);
    73→      }, 300);
    74→    }
    75→  }, [onSearch, searchType]);
    76→
    77→  // Handle Enter key
    78→  const handleKeyDown = (e: React.KeyboardEvent) => {
    79→    if (e.key === 'Enter') {
    80→      e.preventDefault();
    81→      if (debounceTimerRef.current) {
    82→        clearTimeout(debounceTimerRef.current);
    83→      }
    84→      handleSearch();
    85→    }
    86→  };
    87→
    88→  // Clear search
    89→  const handleClear = () => {
    90→    setQuery('');
    91→    inputRef.current?.focus();
    92→  };
    93→
    94→  const filterOptions = [
    95→    { value: 'all', label: 'All' },
    96→    { value: 'posts', label: 'Posts' },
    97→    { value: 'comments', label: 'Comments' },
    98→  ] as const;
    99→
   100→  const selectedFilter = filterOptions.find(f => f.value === searchType);
   101→
   102→  return (
   103→    <div className={`relative flex items-center gap-2 ${className}`}>
   104→      {/* Search Input */}
   105→      <div className="relative flex-1">
   106→        <div className="absolute left-4 top-1/2 -translate-y-1/2 pointer-events-none">
   107→          <Search size={16} className="text-neutral-500" />
   108→        </div>
   109→        <input
   110→          ref={inputRef}
   111→          type="text"
   112→          value={query}
   113→          onChange={(e) => handleInputChange(e.target.value)}
   114→          onKeyDown={handleKeyDown}
   115→          placeholder={placeholder}
   116→          autoFocus={autoFocus}
   117→          className="w-full h-11 pl-11 pr-10 rounded-xl bg-neutral-900/50 border border-white/10
   118→                     text-sm text-white placeholder:text-neutral-600
   119→                     focus:outline-none focus:border-cyan-500/30 focus:ring-1 focus:ring-cyan-500/20
   120→                     transition-all"
   121→        />
   122→        {query && (
   123→          <button
   124→            onClick={handleClear}
   125→            className="absolute right-3 top-1/2 -translate-y-1/2 p-1 rounded-md
   126→                       text-neutral-500 hover:text-white hover:bg-white/10 transition-all"
   127→          >
   128→            <X size={14} />
   129→          </button>
   130→        )}
   131→      </div>
   132→
   133→      {/* Filter Dropdown */}
   134→      {showFilters && (
   135→        <div className="relative" ref={dropdownRef}>
   136→          <button
   137→            onClick={() => setShowFilterDropdown(!showFilterDropdown)}
   138→            className="h-11 px-4 rounded-xl bg-neutral-900/50 border border-white/10
   139→                       flex items-center gap-2 text-[10px] font-black uppercase tracking-widest
   140→                       text-neutral-400 hover:text-white hover:border-white/20 transition-all"
   141→          >
   142→            <Filter size={14} />
   143→            <span className="hidden sm:inline">{selectedFilter?.label}</span>
   144→            <ChevronDown size={12} className={`transition-transform ${showFilterDropdown ? 'rotate-180' : ''}`} />
   145→          </button>
   146→
   147→          <AnimatePresence>
   148→            {showFilterDropdown && (
   149→              <motion.div
   150→                initial={{ opacity: 0, y: -8 }}
   151→                animate={{ opacity: 1, y: 0 }}
   152→                exit={{ opacity: 0, y: -8 }}
   153→                transition={{ duration: 0.15 }}
   154→                className="absolute right-0 top-full mt-2 w-36 py-2 rounded-xl
   155→                           bg-neutral-900 border border-white/10 shadow-xl shadow-black/50 z-50"
   156→              >
   157→                {filterOptions.map((option) => (
   158→                  <button
   159→                    key={option.value}
   160→                    onClick={() => {
   161→                      setSearchType(option.value);
   162→                      setShowFilterDropdown(false);
   163→                    }}
   164→                    className={`w-full px-4 py-2.5 text-left text-[11px] font-bold uppercase tracking-wider
   165→                               transition-colors ${
   166→                                 searchType === option.value
   167→                                   ? 'text-cyan-400 bg-cyan-500/10'
   168→                                   : 'text-neutral-400 hover:text-white hover:bg-white/5'
   169→                               }`}
   170→                  >
   171→                    {option.label}
   172→                  </button>
   173→                ))}
   174→              </motion.div>
   175→            )}
   176→          </AnimatePresence>
   177→        </div>
   178→      )}
   179→
   180→      {/* Search Button */}
   181→      <button
   182→        onClick={handleSearch}
   183→        disabled={query.trim().length < 2}
   184→        className="h-11 px-5 rounded-xl bg-cyan-500 text-black text-[10px] font-black uppercase tracking-widest
   185→                   flex items-center gap-2 hover:bg-cyan-400 transition-all shadow-lg shadow-cyan-500/20
   186→                   disabled:opacity-50 disabled:cursor-not-allowed disabled:hover:bg-cyan-500"
   187→      >
   188→        <Search size={14} />
   189→        <span className="hidden sm:inline">Search</span>
   190→      </button>
   191→    </div>
   192→  );
   193→}
