   745→class ApiClient {
   746→  private token: string | null = null;
   747→
   748→  setToken(token: string | null) {
   749→    this.token = token;
   750→  }
   751→
   752→  getToken(): string | null {
   753→    if (typeof window !== 'undefined') {
   754→      return this.token || localStorage.getItem('auth_token');
   755→    }
   756→    return this.token;
   757→  }
   758→
   759→  private async request<T>(endpoint: string, options: ApiOptions = {}): Promise<T> {
   760→    const { method = 'GET', body, token } = options;
   761→    const authToken = token || this.getToken();
   762→
   763→    const headers: Record<string, string> = {
   764→      'Content-Type': 'application/json',
   765→    };
   766→
   767→    if (authToken) {
   768→      headers['Authorization'] = `Bearer ${authToken}`;
   769→    }
   770→
   771→    try {
   772→      const response = await fetch(`${API_URL}${endpoint}`, {
   773→        method,
   774→        headers,
   775→        body: body ? JSON.stringify(body) : undefined,
   776→      });
   777→
   778→      if (!response.ok) {
   779→        const error = await response.json().catch(() => ({ detail: 'Request failed' }));
   780→        throw new Error(error.detail || `HTTP ${response.status}`);
   781→      }
   782→
   783→      // Handle empty responses (204 No Content)
   784→      if (response.status === 204) {
   785→        return undefined as T;
   786→      }
   787→
   788→      return response.json();
   789→    } catch (err) {
   790→      // Network errors, CORS issues, etc.
   791→      if (err instanceof TypeError && err.message.includes('fetch')) {
   792→        throw new Error('Network error - please check your internet connection');
   793→      }
   794→      throw err;
   795→    }
   796→  }
