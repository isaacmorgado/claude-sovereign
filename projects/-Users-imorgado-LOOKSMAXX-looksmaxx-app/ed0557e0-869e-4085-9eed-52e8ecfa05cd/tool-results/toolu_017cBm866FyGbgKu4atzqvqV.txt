    60→@router.get("/categories", response_model=List[CategoryResponse])
    61→async def list_categories(db: AsyncSession = Depends(get_db)):
    62→    """List all active forum categories with sub-forums and post counts.
    63→
    64→    Optimized to use only 2 queries instead of N+1 pattern.
    65→    """
    66→    # Query 1: Get all categories with sub-forums
    67→    query = (
    68→        select(ForumIssueCategory)
    69→        .options(selectinload(ForumIssueCategory.sub_forums))
    70→        .where(ForumIssueCategory.is_active == True)
    71→        .order_by(ForumIssueCategory.display_order)
    72→    )
    73→    result = await db.execute(query)
    74→    categories = result.scalars().all()
    75→
    76→    # Query 2: Get ALL sub-forum post counts in ONE query (batched)
    77→    sf_counts_query = (
    78→        select(
    79→            ForumPost.sub_forum_id,
    80→            func.count(ForumPost.id).label('post_count')
    81→        )
    82→        .where(
    83→            ForumPost.is_deleted == False,
    84→            ForumPost.is_approved == True
    85→        )
    86→        .group_by(ForumPost.sub_forum_id)
    87→    )
    88→    sf_counts_result = await db.execute(sf_counts_query)
    89→    sf_counts = {row[0]: row[1] for row in sf_counts_result}
    90→
    91→    # Build response using pre-fetched counts (no additional queries)
    92→    response = []
    93→    for cat in categories:
    94→        cat_post_count = 0
    95→        sub_forums = []
    96→
    97→        for sf in sorted(cat.sub_forums, key=lambda x: x.display_order):
    98→            if sf.is_active:
    99→                sf_post_count = sf_counts.get(sf.id, 0)
   100→                cat_post_count += sf_post_count
   101→
   102→                sub_forums.append(SubForumResponse(
   103→                    id=sf.id,
   104→                    name=sf.name,
   105→                    slug=sf.slug,
   106→                    description=sf.description,
   107→                    icon=sf.icon,
   108→                    display_order=sf.display_order,
   109→                    post_count=sf_post_count
   110→                ))
   111→
   112→        response.append(CategoryResponse(
   113→            id=cat.id,
   114→            name=cat.name,
   115→            slug=cat.slug,
   116→            description=cat.description,
   117→            icon=cat.icon,
   118→            display_order=cat.display_order,
   119→            post_count=cat_post_count,
   120→            sub_forums=sub_forums
   121→        ))
   122→
   123→    return response

   126→@router.get("/categories/{slug}", response_model=CategoryResponse)
   127→async def get_category(
   128→    slug: str,
   129→    db: AsyncSession = Depends(get_db)
   130→):
   131→    """Get a single category by slug with its sub-forums.
   132→
   133→    Optimized to use only 2 queries instead of N+1 pattern.
   134→    """
   135→    # Query 1: Get category with sub-forums
   136→    query = (
   137→        select(ForumIssueCategory)
   138→        .options(selectinload(ForumIssueCategory.sub_forums))
   139→        .where(ForumIssueCategory.slug == slug, ForumIssueCategory.is_active == True)
   140→    )
   141→    result = await db.execute(query)
   142→    cat = result.scalar_one_or_none()
   143→
   144→    if not cat:
   145→        raise HTTPException(status_code=404, detail="Category not found")
   146→
   147→    # Get sub-forum IDs for this category
   148→    sf_ids = [sf.id for sf in cat.sub_forums if sf.is_active]
   149→
   150→    # Query 2: Get post counts for all sub-forums in this category (batched)
   151→    sf_counts = {}
   152→    if sf_ids:
   153→        sf_counts_query = (
   154→            select(
   155→                ForumPost.sub_forum_id,
   156→                func.count(ForumPost.id).label('post_count')
   157→            )
   158→            .where(
   159→                ForumPost.sub_forum_id.in_(sf_ids),
   160→                ForumPost.is_deleted == False,
   161→                ForumPost.is_approved == True
   162→            )
   163→            .group_by(ForumPost.sub_forum_id)
   164→        )
   165→        sf_counts_result = await db.execute(sf_counts_query)
   166→        sf_counts = {row[0]: row[1] for row in sf_counts_result}
   167→
   168→    # Build sub-forums list using pre-fetched counts
   169→    cat_post_count = 0
   170→    sub_forums = []
   171→    for sf in sorted(cat.sub_forums, key=lambda x: x.display_order):
   172→        if sf.is_active:
   173→            sf_post_count = sf_counts.get(sf.id, 0)
   174→            cat_post_count += sf_post_count
   175→
   176→            sub_forums.append(SubForumResponse(
   177→                id=sf.id,
   178→                name=sf.name,
   179→                slug=sf.slug,
   180→                description=sf.description,
   181→                icon=sf.icon,
   182→                display_order=sf.display_order,
   183→                post_count=sf_post_count
   184→            ))
   185→
   186→    return CategoryResponse(
   187→        id=cat.id,
   188→        name=cat.name,
   189→        slug=cat.slug,
   190→        description=cat.description,
   191→        icon=cat.icon,
   192→        display_order=cat.display_order,
   193→        post_count=cat_post_count,
   194→        sub_forums=sub_forums
   195→    )

   200→@router.get("/search", response_model=ForumSearchResponse)
   201→async def search_forum(
   202→    q: str = Query(..., min_length=2, max_length=100, description="Search query"),
   203→    type: str = Query("all", description="Search type: posts, comments, or all"),
   204→    limit: int = Query(20, ge=1, le=50),
   205→    offset: int = Query(0, ge=0),
   206→    db: AsyncSession = Depends(get_db)
   207→):
   208→    """Search forum posts and comments using ILIKE pattern matching.
   209→
   210→    - Search by title, content for posts
   211→    - Search by content for comments
   212→    - Results sorted by vote_count DESC, created_at DESC
   213→    """
   214→    search_pattern = f"%{q}%"
   215→    results = []
   216→    total_count = 0
   217→
   218→    if type in ("posts", "all"):
   219→        # Search posts by title and content
   220→        post_query = (
   221→            select(ForumPost)
   222→            .options(
   223→                selectinload(ForumPost.author),
   224→                selectinload(ForumPost.sub_forum).selectinload(ForumSubForum.issue_category)
   225→            )
   226→            .where(
   227→                ForumPost.is_deleted == False,
   228→                ForumPost.is_approved == True,
   229→                or_(
   230→                    ForumPost.title.ilike(search_pattern),
   231→                    ForumPost.content.ilike(search_pattern)
   232→                )
   233→            )
   234→            .order_by(ForumPost.vote_count.desc(), ForumPost.created_at.desc())
   235→        )
   236→
   237→        # Count posts
   238→        post_count_query = (
   239→            select(func.count(ForumPost.id))
   240→            .where(
   241→                ForumPost.is_deleted == False,
   242→                ForumPost.is_approved == True,
   243→                or_(
   244→                    ForumPost.title.ilike(search_pattern),
   245→                    ForumPost.content.ilike(search_pattern)
   246→                )
   247→            )
   248→        )
   249→        post_count_result = await db.execute(post_count_query)
   250→        post_count = post_count_result.scalar() or 0
   251→
   252→        if type == "posts":
   253→            total_count = post_count
   254→            post_query = post_query.offset(offset).limit(limit)
   255→        else:
   256→            total_count += post_count
   257→            # For "all", we get all posts first then merge with comments
   258→            post_query = post_query.limit(limit * 2)  # Get more to merge
   259→
   260→        post_result = await db.execute(post_query)
   261→        posts = post_result.scalars().all()
   262→
   263→        for post in posts:
   264→            content_preview = post.content[:200] + "..." if len(post.content) > 200 else post.content
   265→            results.append(ForumSearchResult(
   266→                id=post.id,
   267→                type=SearchResultType.POST,
   268→                title=post.title,
   269→                content_preview=content_preview,
   270→                post_id=None,
   271→                post_title=None,
   272→                author=PostAuthor(id=post.author.id, username=post.author.username),
   273→                vote_count=post.vote_count,
   274→                category_slug=post.sub_forum.issue_category.slug if post.sub_forum and post.sub_forum.issue_category else "",
   275→                sub_forum_slug=post.sub_forum.slug if post.sub_forum else "",
   276→                created_at=post.created_at
   277→            ))
   278→
   279→    if type in ("comments", "all"):
   280→        # Search comments by content
   281→        comment_query = (
   282→            select(ForumComment)
   283→            .options(
   284→                selectinload(ForumComment.author),
   285→                selectinload(ForumComment.post).selectinload(ForumPost.sub_forum).selectinload(ForumSubForum.issue_category)
   286→            )
   287→            .where(
   288→                ForumComment.is_deleted == False,
   289→                ForumComment.content.ilike(search_pattern)
   290→            )
   291→            .order_by(ForumComment.vote_count.desc(), ForumComment.created_at.desc())
   292→        )
   293→
   294→        # Count comments
   295→        comment_count_query = (
   296→            select(func.count(ForumComment.id))
   297→            .where(
   298→                ForumComment.is_deleted == False,
   299→                ForumComment.content.ilike(search_pattern)
   300→            )
   301→        )
   302→        comment_count_result = await db.execute(comment_count_query)
   303→        comment_count = comment_count_result.scalar() or 0
   304→
   305→        if type == "comments":
   306→            total_count = comment_count
   307→            comment_query = comment_query.offset(offset).limit(limit)
   308→        else:
   309→            total_count += comment_count
   310→            comment_query = comment_query.limit(limit * 2)
   311→
   312→        comment_result = await db.execute(comment_query)
   313→        comments = comment_result.scalars().all()
   314→
   315→        for comment in comments:
   316→            content_preview = comment.content[:200] + "..." if len(comment.content) > 200 else comment.content
   317→            post = comment.post
   318→            if post and post.sub_forum:
   319→                category_slug = post.sub_forum.issue_category.slug if post.sub_forum.issue_category else ""
   320→                sub_forum_slug = post.sub_forum.slug
   321→            else:
   322→                category_slug = ""
   323→                sub_forum_slug = ""
   324→
   325→            results.append(ForumSearchResult(
   326→                id=comment.id,
   327→                type=SearchResultType.COMMENT,
   328→                title=None,
   329→                content_preview=content_preview,
   330→                post_id=comment.post_id,
   331→                post_title=post.title if post else None,
   332→                author=PostAuthor(id=comment.author.id, username=comment.author.username),
   333→                vote_count=comment.vote_count,
   334→                category_slug=category_slug,
   335→                sub_forum_slug=sub_forum_slug,
   336→                created_at=comment.created_at
   337→            ))
   338→
   339→    # For "all" type, sort merged results by vote_count and created_at, then paginate
   340→    if type == "all":
   341→        results.sort(key=lambda x: (-x.vote_count, x.created_at), reverse=False)
   342→        results.sort(key=lambda x: x.vote_count, reverse=True)
   343→        results = results[offset:offset + limit]
   344→
   345→    return ForumSearchResponse(
   346→        results=results,
   347→        total_count=total_count,
   348→        query=q,
   349→        search_type=type,
   350→        has_more=offset + limit < total_count
   351→    )

   354→@router.get("/recommended", response_model=List[RecommendedForumResponse])
   355→async def get_recommended_forums(
   356→    flaws: str = Query(..., description="Comma-separated list of flaw IDs"),
   357→    db: AsyncSession = Depends(get_db)
   358→):
   359→    """Get recommended forums based on detected flaws."""
   360→    flaw_list = [f.strip() for f in flaws.split(",") if f.strip()]
   361→
   362→    if not flaw_list:
   363→        return []
   364→
   365→    # Find mappings for these flaws
   366→    query = (
   367→        select(FlawToForumMapping)
   368→        .options(selectinload(FlawToForumMapping.issue_category))
   369→        .where(FlawToForumMapping.flaw_id.in_(flaw_list))
   370→        .order_by(FlawToForumMapping.priority)
   371→    )
   372→
   373→    result = await db.execute(query)
   374→    mappings = result.scalars().all()
   375→
   376→    # Group by category
   377→    category_flaws = {}
   378→    for mapping in mappings:
   379→        cat_id = mapping.issue_category_id
   380→        if cat_id not in category_flaws:
   381→            category_flaws[cat_id] = {
   382→                "category": mapping.issue_category,
   383→                "flaws": [],
   384→                "priority": mapping.priority
   385→            }
   386→        category_flaws[cat_id]["flaws"].append(mapping.flaw_id)
   387→        # Use lowest priority (1 = primary)
   388→        if mapping.priority < category_flaws[cat_id]["priority"]:
   389→            category_flaws[cat_id]["priority"] = mapping.priority
   390→
   391→    # Build response sorted by priority
   392→    response = []
   393→    for cat_data in sorted(category_flaws.values(), key=lambda x: x["priority"]):
   394→        cat = cat_data["category"]
   395→        response.append(RecommendedForumResponse(
   396→            category=CategoryListResponse(
   397→                id=cat.id,
   398→                name=cat.name,
   399→                slug=cat.slug,
   400→                description=cat.description,
   401→                icon=cat.icon,
   402→                display_order=cat.display_order,
   403→                post_count=0
   404→            ),
   405→            matched_flaws=cat_data["flaws"],
   406→            priority=cat_data["priority"]
   407→        ))
   408→
   409→    return response

   522→@router.get("/sub-forums/{sub_forum_id}/posts", response_model=PostListResponse)
   523→async def list_sub_forum_posts(
   524→    sub_forum_id: UUID,
   525→    sort: SortOrder = Query(SortOrder.HOT),
   526→    limit: int = Query(PAGE_SIZE, ge=1, le=50),
   527→    offset: int = Query(0, ge=0),
   528→    current_user: Optional[User] = Depends(get_current_user_optional),
   529→    db: AsyncSession = Depends(get_db),
   530→):
   531→    """List posts in a sub-forum with sorting and pagination."""
   532→    # Verify sub-forum exists and get category
   533→    sf_result = await db.execute(
   534→        select(ForumSubForum)
   535→        .options(selectinload(ForumSubForum.issue_category))
   536→        .where(ForumSubForum.id == sub_forum_id)
   537→    )
   538→    sub_forum = sf_result.scalar_one_or_none()
   539→    if not sub_forum:
   540→        raise HTTPException(status_code=404, detail="Sub-forum not found")
   541→
   542→    category_slug = sub_forum.issue_category.slug
   543→
   544→    # Base query
   545→    query = (
   546→        select(ForumPost)
   547→        .options(selectinload(ForumPost.author))
   548→        .where(
   549→            ForumPost.sub_forum_id == sub_forum_id,
   550→            ForumPost.is_deleted == False,
   551→            ForumPost.is_approved == True
   552→        )
   553→    )
   554→
   555→    # Sorting - pinned always first
   556→    if sort == SortOrder.NEW:
   557→        query = query.order_by(ForumPost.is_pinned.desc(), ForumPost.created_at.desc())
   558→    elif sort == SortOrder.TOP:
   559→        query = query.order_by(ForumPost.is_pinned.desc(), ForumPost.vote_count.desc())
   560→    else:  # HOT - combination of votes and recency
   561→        query = query.order_by(ForumPost.is_pinned.desc(), ForumPost.vote_count.desc(), ForumPost.created_at.desc())
   562→
   563→    # Get total count
   564→    count_query = (
   565→        select(func.count(ForumPost.id))
   566→        .where(
   567→            ForumPost.sub_forum_id == sub_forum_id,
   568→            ForumPost.is_deleted == False,
   569→            ForumPost.is_approved == True
   570→        )
   571→    )
   572→    total_result = await db.execute(count_query)
   573→    total_count = total_result.scalar() or 0
   574→

   843→@router.post("/posts/{post_id}/vote", response_model=VoteResponse)
   844→@limiter.limit(ForumRateLimits.VOTES)
   845→async def vote_on_post(
   846→    request: Request,
   847→    post_id: UUID,
   848→    data: VoteRequest,
   849→    current_user: User = Depends(get_current_user),
   850→    db: AsyncSession = Depends(get_db),
   851→):
   852→    """Vote on a post (upvote/downvote). Voting again removes vote, changing vote type switches it.
   853→
   854→    Rate limit: 30 votes per minute per user.
   855→    Also updates the post author's karma.
   856→    """
   857→    # Verify post exists and get author
   858→    post_result = await db.execute(
   859→        select(ForumPost).where(ForumPost.id == post_id, ForumPost.is_deleted == False)
   860→    )
   861→    post = post_result.scalar_one_or_none()
   862→    if not post:
   863→        raise HTTPException(status_code=404, detail="Post not found")
   864→
   865→    # Prevent self-voting
   866→    if post.author_id == current_user.id:
   867→        raise HTTPException(status_code=400, detail="Cannot vote on your own content")
   868→
   869→    # Get the post author for karma update
   870→    author_result = await db.execute(select(User).where(User.id == post.author_id))
   871→    post_author = author_result.scalar_one_or_none()
   872→
   873→    # Check existing vote
   874→    vote_result = await db.execute(
   875→        select(ForumVote).where(
   876→            ForumVote.user_id == current_user.id,
   877→            ForumVote.target_type == TargetType.POST,
   878→            ForumVote.target_id == post_id
   879→        )
   880→    )
   881→    existing_vote = vote_result.scalar_one_or_none()
   882→
   883→    new_user_vote = None
   884→    karma_delta = 0
   885→
   886→    if existing_vote:
   887→        # Robustly get value (handle if DB returns string or Enum)
   888→        current_val = existing_vote.vote_type.value if hasattr(existing_vote.vote_type, 'value') else existing_vote.vote_type
   889→        # Request is from Pydantic Schema Enum
   890→        request_val = data.vote_type.value
   891→
   892→        if current_val == request_val:
   893→            # Same vote - remove it
   894→            await db.delete(existing_vote)
   895→            post.vote_count += -1 if request_val == "up" else 1
   896→            karma_delta = -1 if request_val == "up" else 1
   897→        else:
   898→            # Different vote - switch it
   899→            # Ensure we assign the Model Enum (or string if that's what it wants, but Model Enum is safer if Typed)
   900→            existing_vote.vote_type = VoteType(request_val)
   901→            # Swing by 2 (remove old, add new)
   902→            post.vote_count += 2 if request_val == "up" else -2
   903→            karma_delta = 2 if request_val == "up" else -2
   904→            new_user_vote = data.vote_type
   905→    else:
   906→        # New vote
   907→        vote = ForumVote(
   908→            user_id=current_user.id,
   909→            target_type=TargetType.POST,
   910→            target_id=post_id,
   911→            # Explicitly cast to Model Enum
   912→            vote_type=VoteType(data.vote_type.value)
   913→        )
   914→        db.add(vote)
   915→        post.vote_count += 1 if data.vote_type.value == "up" else -1
   916→        karma_delta = 1 if data.vote_type.value == "up" else -1
   917→        new_user_vote = data.vote_type
   918→
   919→    # Update post author's karma
   920→    if post_author and karma_delta != 0:
   921→        post_author.forum_karma = (post_author.forum_karma or 0) + karma_delta
   922→
   923→    await db.commit()
   924→
   925→    return VoteResponse(success=True, new_vote_count=post.vote_count, user_vote=new_user_vote)
