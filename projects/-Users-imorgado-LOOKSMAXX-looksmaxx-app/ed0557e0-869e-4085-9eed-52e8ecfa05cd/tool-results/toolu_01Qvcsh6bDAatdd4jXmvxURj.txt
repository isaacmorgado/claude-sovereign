    61→@router.get("/categories", response_model=List[CategoryResponse])
    62→async def list_categories(db: AsyncSession = Depends(get_db)):
    63→    """List all active forum categories with sub-forums and post counts.
    64→
    65→    Optimized to use only 2 queries instead of N+1 pattern.
    66→    """
    67→    # Query 1: Get all categories with sub-forums
    68→    query = (
    69→        select(ForumIssueCategory)
    70→        .options(selectinload(ForumIssueCategory.sub_forums))
    71→        .where(ForumIssueCategory.is_active == True)
    72→        .order_by(ForumIssueCategory.display_order)
    73→    )
    74→    result = await db.execute(query)
    75→    categories = result.scalars().all()
    76→
    77→    # Query 2: Get ALL sub-forum post counts in ONE query (batched)
    78→    sf_counts_query = (
    79→        select(
    80→            ForumPost.sub_forum_id,
    81→            func.count(ForumPost.id).label('post_count')
    82→        )
    83→        .where(
    84→            ForumPost.is_deleted == False,
    85→            ForumPost.is_approved == True
    86→        )
    87→        .group_by(ForumPost.sub_forum_id)
    88→    )
    89→    sf_counts_result = await db.execute(sf_counts_query)
    90→    sf_counts = {row[0]: row[1] for row in sf_counts_result}
    91→
    92→    # Build response using pre-fetched counts (no additional queries)
    93→    response = []
    94→    for cat in categories:
    95→        cat_post_count = 0
    96→        sub_forums = []
    97→
    98→        for sf in sorted(cat.sub_forums, key=lambda x: x.display_order):
    99→            if sf.is_active:
   100→                sf_post_count = sf_counts.get(sf.id, 0)
   101→                cat_post_count += sf_post_count
   102→
   103→                sub_forums.append(SubForumResponse(
   104→                    id=sf.id,
   105→                    name=sf.name,
   106→                    slug=sf.slug,
   107→                    description=sf.description,
   108→                    icon=sf.icon,
   109→                    display_order=sf.display_order,
   110→                    post_count=sf_post_count
   111→                ))
   112→
   113→        response.append(CategoryResponse(
   114→            id=cat.id,
   115→            name=cat.name,
   116→            slug=cat.slug,
   117→            description=cat.description,
   118→            icon=cat.icon,
   119→            display_order=cat.display_order,
   120→            post_count=cat_post_count,
   121→            sub_forums=sub_forums
   122→        ))
   123→
   124→    return response

   127→@router.get("/categories/{slug}", response_model=CategoryResponse)
   128→async def get_category(
   129→    slug: str,
   130→    db: AsyncSession = Depends(get_db)
   131→):
   132→    """Get a single category by slug with its sub-forums.
   133→
   134→    Optimized to use only 2 queries instead of N+1 pattern.
   135→    """
   136→    # Query 1: Get category with sub-forums
   137→    query = (
   138→        select(ForumIssueCategory)
   139→        .options(selectinload(ForumIssueCategory.sub_forums))
   140→        .where(ForumIssueCategory.slug == slug, ForumIssueCategory.is_active == True)
   141→    )
   142→    result = await db.execute(query)
   143→    cat = result.scalar_one_or_none()
   144→
   145→    if not cat:
   146→        raise HTTPException(status_code=404, detail="Category not found")
   147→
   148→    # Get sub-forum IDs for this category
   149→    sf_ids = [sf.id for sf in cat.sub_forums if sf.is_active]
   150→
   151→    # Query 2: Get post counts for all sub-forums in this category (batched)
   152→    sf_counts = {}
   153→    if sf_ids:
   154→        sf_counts_query = (
   155→            select(
   156→                ForumPost.sub_forum_id,
   157→                func.count(ForumPost.id).label('post_count')
   158→            )
   159→            .where(
   160→                ForumPost.sub_forum_id.in_(sf_ids),
   161→                ForumPost.is_deleted == False,
   162→                ForumPost.is_approved == True
   163→            )
   164→            .group_by(ForumPost.sub_forum_id)
   165→        )
   166→        sf_counts_result = await db.execute(sf_counts_query)
   167→        sf_counts = {row[0]: row[1] for row in sf_counts_result}
   168→
   169→    # Build sub-forums list using pre-fetched counts
   170→    cat_post_count = 0
   171→    sub_forums = []
   172→    for sf in sorted(cat.sub_forums, key=lambda x: x.display_order):
   173→        if sf.is_active:
   174→            sf_post_count = sf_counts.get(sf.id, 0)
   175→            cat_post_count += sf_post_count
   176→
   177→            sub_forums.append(SubForumResponse(
   178→                id=sf.id,
   179→                name=sf.name,
   180→                slug=sf.slug,
   181→                description=sf.description,
   182→                icon=sf.icon,
   183→                display_order=sf.display_order,
   184→                post_count=sf_post_count
   185→            ))
   186→
   187→    return CategoryResponse(
   188→        id=cat.id,
   189→        name=cat.name,
   190→        slug=cat.slug,
   191→        description=cat.description,
   192→        icon=cat.icon,
   193→        display_order=cat.display_order,
   194→        post_count=cat_post_count,
   195→        sub_forums=sub_forums
   196→    )

   201→@router.get("/search", response_model=ForumSearchResponse)
   202→async def search_forum(
   203→    q: str = Query(..., min_length=2, max_length=100, description="Search query"),
   204→    type: str = Query("all", description="Search type: posts, comments, or all"),
   205→    limit: int = Query(20, ge=1, le=50),
   206→    offset: int = Query(0, ge=0),
   207→    db: AsyncSession = Depends(get_db)
   208→):
   209→    """Search forum posts and comments using ILIKE pattern matching.
   210→
   211→    - Search by title, content for posts
   212→    - Search by content for comments
   213→    - Results sorted by vote_count DESC, created_at DESC
   214→    """
   215→    search_pattern = f"%{q}%"
   216→    results = []
   217→    total_count = 0
   218→
   219→    if type in ("posts", "all"):
   220→        # Search posts by title and content
   221→        post_query = (
   222→            select(ForumPost)
   223→            .options(
   224→                selectinload(ForumPost.author),
   225→                selectinload(ForumPost.sub_forum).selectinload(ForumSubForum.issue_category)
   226→            )
   227→            .where(
   228→                ForumPost.is_deleted == False,
   229→                ForumPost.is_approved == True,
   230→                or_(
   231→                    ForumPost.title.ilike(search_pattern),
   232→                    ForumPost.content.ilike(search_pattern)
   233→                )
   234→            )
   235→            .order_by(ForumPost.vote_count.desc(), ForumPost.created_at.desc())
   236→        )
   237→
   238→        # Count posts
   239→        post_count_query = (
   240→            select(func.count(ForumPost.id))
   241→            .where(
   242→                ForumPost.is_deleted == False,
   243→                ForumPost.is_approved == True,
   244→                or_(
   245→                    ForumPost.title.ilike(search_pattern),
   246→                    ForumPost.content.ilike(search_pattern)
   247→                )
   248→            )
   249→        )
   250→        post_count_result = await db.execute(post_count_query)
   251→        post_count = post_count_result.scalar() or 0
   252→
   253→        if type == "posts":
   254→            total_count = post_count
   255→            post_query = post_query.offset(offset).limit(limit)
   256→        else:
   257→            total_count += post_count
   258→            # For "all", we get all posts first then merge with comments
   259→            post_query = post_query.limit(limit * 2)  # Get more to merge
   260→
   261→        post_result = await db.execute(post_query)
   262→        posts = post_result.scalars().all()
   263→
   264→        for post in posts:
   265→            content_preview = post.content[:200] + "..." if len(post.content) > 200 else post.content
   266→            results.append(ForumSearchResult(
   267→                id=post.id,
   268→                type=SearchResultType.POST,
   269→                title=post.title,
   270→                content_preview=content_preview,
   271→                post_id=None,
   272→                post_title=None,
   273→                author=PostAuthor(id=post.author.id, username=post.author.username),
   274→                vote_count=post.vote_count,
   275→                category_slug=post.sub_forum.issue_category.slug if post.sub_forum and post.sub_forum.issue_category else "",
   276→                sub_forum_slug=post.sub_forum.slug if post.sub_forum else "",
   277→                created_at=post.created_at
   278→            ))
   279→
   280→    if type in ("comments", "all"):
   281→        # Search comments by content
   282→        comment_query = (
   283→            select(ForumComment)
   284→            .options(
   285→                selectinload(ForumComment.author),
   286→                selectinload(ForumComment.post).selectinload(ForumPost.sub_forum).selectinload(ForumSubForum.issue_category)
   287→            )
   288→            .where(
   289→                ForumComment.is_deleted == False,
   290→                ForumComment.content.ilike(search_pattern)
   291→            )
   292→            .order_by(ForumComment.vote_count.desc(), ForumComment.created_at.desc())
   293→        )
   294→
   295→        # Count comments
   296→        comment_count_query = (
   297→            select(func.count(ForumComment.id))
   298→            .where(
   299→                ForumComment.is_deleted == False,
   300→                ForumComment.content.ilike(search_pattern)
   301→            )
   302→        )
   303→        comment_count_result = await db.execute(comment_count_query)
   304→        comment_count = comment_count_result.scalar() or 0
   305→
   306→        if type == "comments":
   307→            total_count = comment_count
   308→            comment_query = comment_query.offset(offset).limit(limit)
   309→        else:
   310→            total_count += comment_count
   311→            comment_query = comment_query.limit(limit * 2)
   312→
   313→        comment_result = await db.execute(comment_query)
   314→        comments = comment_result.scalars().all()
   315→
   316→        for comment in comments:
   317→            content_preview = comment.content[:200] + "..." if len(comment.content) > 200 else comment.content
   318→            post = comment.post
   319→            if post and post.sub_forum:
   320→                category_slug = post.sub_forum.issue_category.slug if post.sub_forum.issue_category else ""
   321→                sub_forum_slug = post.sub_forum.slug
   322→            else:
   323→                category_slug = ""
   324→                sub_forum_slug = ""
   325→
   326→            results.append(ForumSearchResult(
   327→                id=comment.id,
   328→                type=SearchResultType.COMMENT,
   329→                title=None,
   330→                content_preview=content_preview,
   331→                post_id=comment.post_id,
   332→                post_title=post.title if post else None,
   333→                author=PostAuthor(id=comment.author.id, username=comment.author.username),
   334→                vote_count=comment.vote_count,
   335→                category_slug=category_slug,
   336→                sub_forum_slug=sub_forum_slug,
   337→                created_at=comment.created_at
   338→            ))
   339→
   340→    # For "all" type, sort merged results by vote_count and created_at, then paginate
   341→    if type == "all":
   342→        results.sort(key=lambda x: (-x.vote_count, x.created_at), reverse=False)
   343→        results.sort(key=lambda x: x.vote_count, reverse=True)
   344→        results = results[offset:offset + limit]
   345→
   346→    return ForumSearchResponse(
   347→        results=results,
   348→        total_count=total_count,
   349→        query=q,
   350→        search_type=type,
   351→        has_more=offset + limit < total_count
   352→    )

   355→@router.get("/recommended", response_model=List[RecommendedForumResponse])
   356→async def get_recommended_forums(
   357→    flaws: str = Query(..., description="Comma-separated list of flaw IDs"),
   358→    db: AsyncSession = Depends(get_db)
   359→):
   360→    """Get recommended forums based on detected flaws."""
   361→    flaw_list = [f.strip() for f in flaws.split(",") if f.strip()]
   362→
   363→    if not flaw_list:
   364→        return []
   365→
   366→    # Find mappings for these flaws
   367→    query = (
   368→        select(FlawToForumMapping)
   369→        .options(selectinload(FlawToForumMapping.issue_category))
   370→        .where(FlawToForumMapping.flaw_id.in_(flaw_list))
   371→        .order_by(FlawToForumMapping.priority)
   372→    )
   373→
   374→    result = await db.execute(query)
   375→    mappings = result.scalars().all()
   376→
   377→    # Group by category
   378→    category_flaws = {}
   379→    for mapping in mappings:
   380→        cat_id = mapping.issue_category_id
   381→        if cat_id not in category_flaws:
   382→            category_flaws[cat_id] = {
   383→                "category": mapping.issue_category,
   384→                "flaws": [],
   385→                "priority": mapping.priority
   386→            }
   387→        category_flaws[cat_id]["flaws"].append(mapping.flaw_id)
   388→        # Use lowest priority (1 = primary)
   389→        if mapping.priority < category_flaws[cat_id]["priority"]:
   390→            category_flaws[cat_id]["priority"] = mapping.priority
   391→
   392→    # Build response sorted by priority
   393→    response = []
   394→    for cat_data in sorted(category_flaws.values(), key=lambda x: x["priority"]):
   395→        cat = cat_data["category"]
   396→        response.append(RecommendedForumResponse(
   397→            category=CategoryListResponse(
   398→                id=cat.id,
   399→                name=cat.name,
   400→                slug=cat.slug,
   401→                description=cat.description,
   402→                icon=cat.icon,
   403→                display_order=cat.display_order,
   404→                post_count=0
   405→            ),
   406→            matched_flaws=cat_data["flaws"],
   407→            priority=cat_data["priority"]
   408→        ))
   409→
   410→    return response

   523→@router.get("/sub-forums/{sub_forum_id}/posts", response_model=PostListResponse)
   524→async def list_sub_forum_posts(
   525→    sub_forum_id: UUID,
   526→    sort: SortOrder = Query(SortOrder.HOT),
   527→    limit: int = Query(PAGE_SIZE, ge=1, le=50),
   528→    offset: int = Query(0, ge=0),
   529→    current_user: Optional[User] = Depends(get_current_user_optional),
   530→    db: AsyncSession = Depends(get_db),
   531→):
   532→    """List posts in a sub-forum with sorting and pagination."""
   533→    # Verify sub-forum exists and get category
   534→    sf_result = await db.execute(
   535→        select(ForumSubForum)
   536→        .options(selectinload(ForumSubForum.issue_category))
   537→        .where(ForumSubForum.id == sub_forum_id)
   538→    )
   539→    sub_forum = sf_result.scalar_one_or_none()
   540→    if not sub_forum:
   541→        raise HTTPException(status_code=404, detail="Sub-forum not found")
   542→
   543→    category_slug = sub_forum.issue_category.slug
   544→
   545→    # Base query
   546→    query = (
   547→        select(ForumPost)
   548→        .options(selectinload(ForumPost.author))
   549→        .where(
   550→            ForumPost.sub_forum_id == sub_forum_id,
   551→            ForumPost.is_deleted == False,
   552→            ForumPost.is_approved == True
   553→        )
   554→    )
   555→
   556→    # Sorting - pinned always first
   557→    if sort == SortOrder.NEW:
   558→        query = query.order_by(ForumPost.is_pinned.desc(), ForumPost.created_at.desc())
   559→    elif sort == SortOrder.TOP:
   560→        query = query.order_by(ForumPost.is_pinned.desc(), ForumPost.vote_count.desc())
   561→    else:  # HOT - combination of votes and recency
   562→        query = query.order_by(ForumPost.is_pinned.desc(), ForumPost.vote_count.desc(), ForumPost.created_at.desc())
   563→
   564→    # Get total count
   565→    count_query = (
   566→        select(func.count(ForumPost.id))
   567→        .where(
   568→            ForumPost.sub_forum_id == sub_forum_id,
   569→            ForumPost.is_deleted == False,
   570→            ForumPost.is_approved == True
   571→        )
   572→    )
   573→    total_result = await db.execute(count_query)

   844→@router.post("/posts/{post_id}/vote", response_model=VoteResponse)
   845→@limiter.limit(ForumRateLimits.VOTES)
   846→async def vote_on_post(
   847→    request: Request,
   848→    post_id: UUID,
   849→    data: VoteRequest,
   850→    current_user: User = Depends(get_current_user),
   851→    db: AsyncSession = Depends(get_db),
   852→):
   853→    """Vote on a post (upvote/downvote). Voting again removes vote, changing vote type switches it.
   854→
   855→    Rate limit: 30 votes per minute per user.
   856→    Also updates the post author's karma.
   857→    """
   858→    # Verify post exists and get author
   859→    post_result = await db.execute(
   860→        select(ForumPost).where(ForumPost.id == post_id, ForumPost.is_deleted == False)
   861→    )
   862→    post = post_result.scalar_one_or_none()
   863→    if not post:
   864→        raise HTTPException(status_code=404, detail="Post not found")
   865→
   866→    # Prevent self-voting
   867→    if post.author_id == current_user.id:
   868→        raise HTTPException(status_code=400, detail="Cannot vote on your own content")
   869→
   870→    # Get the post author for karma update
   871→    author_result = await db.execute(select(User).where(User.id == post.author_id))
   872→    post_author = author_result.scalar_one_or_none()
   873→
   874→    # Check existing vote
   875→    vote_result = await db.execute(
   876→        select(ForumVote).where(
   877→            ForumVote.user_id == current_user.id,
   878→            ForumVote.target_type == TargetType.POST,
   879→            ForumVote.target_id == post_id
   880→        )
   881→    )
   882→    existing_vote = vote_result.scalar_one_or_none()
   883→
   884→    new_user_vote = None
   885→    karma_delta = 0
   886→
   887→    if existing_vote:
   888→        # Robustly get value (handle if DB returns string or Enum)
   889→        current_val = existing_vote.vote_type.value if hasattr(existing_vote.vote_type, 'value') else existing_vote.vote_type
   890→        # Request is from Pydantic Schema Enum
   891→        request_val = data.vote_type.value
   892→
   893→        if current_val == request_val:
   894→            # Same vote - remove it
   895→            await db.delete(existing_vote)
   896→            post.vote_count += -1 if request_val == "up" else 1
   897→            karma_delta = -1 if request_val == "up" else 1
   898→        else:
   899→            # Different vote - switch it
   900→            # Ensure we assign the Model Enum (or string if that's what it wants, but Model Enum is safer if Typed)
   901→            existing_vote.vote_type = VoteType(request_val)
   902→            # Swing by 2 (remove old, add new)
   903→            post.vote_count += 2 if request_val == "up" else -2
   904→            karma_delta = 2 if request_val == "up" else -2
   905→            new_user_vote = data.vote_type
   906→    else:
   907→        # New vote
   908→        vote = ForumVote(
   909→            user_id=current_user.id,
   910→            target_type=TargetType.POST,
   911→            target_id=post_id,
   912→            # Explicitly cast to Model Enum
   913→            vote_type=VoteType(data.vote_type.value)
   914→        )
   915→        db.add(vote)
   916→        post.vote_count += 1 if data.vote_type.value == "up" else -1
   917→        karma_delta = 1 if data.vote_type.value == "up" else -1
   918→        new_user_vote = data.vote_type
   919→
   920→    # Update post author's karma
   921→    if post_author and karma_delta != 0:
   922→        post_author.forum_karma = (post_author.forum_karma or 0) + karma_delta
   923→
   924→    await db.commit()
   925→
   926→    return VoteResponse(success=True, new_vote_count=post.vote_count, user_vote=new_user_vote)
