   218→// ============================================
   219→// SEARCH CONTENT COMPONENT
   220→// ============================================
   221→function SearchContent() {
   222→  const searchParams = useSearchParams();
   223→  const [results, setResults] = useState<ForumSearchResult[]>([]);
   224→  const [totalCount, setTotalCount] = useState(0);
   225→  const [isLoading, setIsLoading] = useState(false);
   226→  const [error, setError] = useState<string | null>(null);
   227→  const [currentPage, setCurrentPage] = useState(1);
   228→
   229→  const query = searchParams.get('q') || '';
   230→  const type = (searchParams.get('type') || 'all') as 'posts' | 'comments' | 'all';
   231→  const PAGE_SIZE = 20;
   232→
   233→  const totalPages = Math.ceil(totalCount / PAGE_SIZE);
   234→
   235→  // Perform search
   236→  const performSearch = useCallback(async (searchQuery: string, searchType: 'posts' | 'comments' | 'all', page: number) => {
   237→    if (!searchQuery || searchQuery.length < 2) {
   238→      setResults([]);
   239→      setTotalCount(0);
   240→      return;
   241→    }
   242→
   243→    setIsLoading(true);
   244→    setError(null);
   245→
   246→    try {
   247→      const response = await api.searchForum({
   248→        query: searchQuery,
   249→        type: searchType,
   250→        limit: PAGE_SIZE,
   251→        offset: (page - 1) * PAGE_SIZE,
   252→      });
   253→      setResults(response.results);
   254→      setTotalCount(response.totalCount);
   255→    } catch (err) {
   256→      console.error('Search error:', err);
   257→      setError(err instanceof Error ? err.message : 'Search failed. Please try again.');
   258→      setResults([]);
   259→      setTotalCount(0);
   260→    } finally {
   261→      setIsLoading(false);
   262→    }
   263→  }, []);
   264→
   265→  // Search when params change
   266→  useEffect(() => {
   267→    setCurrentPage(1);
   268→    performSearch(query, type, 1);
   269→  }, [query, type, performSearch]);
   270→
   271→  // Handle page change
   272→  const handlePageChange = (page: number) => {
   273→    setCurrentPage(page);
   274→    performSearch(query, type, page);
   275→    window.scrollTo({ top: 0, behavior: 'smooth' });
   276→  };
   277→
   278→  // Handle new search from SearchBar
   279→  const handleSearch = (newQuery: string, newType: 'posts' | 'comments' | 'all') => {
   280→    const params = new URLSearchParams();
   281→    params.set('q', newQuery);
   282→    if (newType !== 'all') params.set('type', newType);
   283→    window.history.pushState({}, '', `/forum/search?${params.toString()}`);
   284→    setCurrentPage(1);
   285→    performSearch(newQuery, newType, 1);
   286→  };
   287→
   288→  return (
   289→    <>
   290→      {/* Search Header */}
   291→      <section className="border-b border-white/5">
   292→        <div className="max-w-4xl mx-auto px-6 py-8">
   293→          <h1 className="text-3xl md:text-4xl font-black tracking-tighter italic uppercase mb-6">
   294→            Search <span className="text-cyan-400">Results</span>
   295→          </h1>
   296→
   297→          <SearchBar
   298→            initialQuery={query}
   299→            initialType={type}
   300→            onSearch={handleSearch}
   301→            placeholder="Search posts and comments..."
   302→          />
   303→
   304→          {query && !isLoading && (
   305→            <p className="mt-4 text-sm text-neutral-500">
   306→              {totalCount > 0 ? (
   307→                <>Found <span className="text-white font-bold">{totalCount}</span> result{totalCount !== 1 ? 's' : ''} for &quot;{query}&quot;</>
   308→              ) : (
   309→                <>Searching for &quot;{query}&quot;...</>
   310→              )}
   311→            </p>
   312→          )}
   313→        </div>
   314→      </section>
   315→
   316→      {/* Results */}
   317→      <main className="max-w-4xl mx-auto px-6 py-8">
   318→        {/* Error State */}
   319→        {error && (
   320→          <div className="p-4 rounded-xl bg-red-500/10 border border-red-500/20 mb-6 flex items-center gap-3">
   321→            <AlertCircle size={18} className="text-red-400" />
   322→            <p className="text-red-400 text-sm">{error}</p>
   323→          </div>
   324→        )}
   325→
   326→        {/* Loading State */}
   327→        {isLoading && (
   328→          <div className="space-y-4">
   329→            {[1, 2, 3, 4, 5].map((i) => (
   330→              <SearchResultSkeleton key={i} />
   331→            ))}
   332→          </div>
   333→        )}
   334→
   335→        {/* Results List */}
   336→        {!isLoading && results.length > 0 && (
   337→          <>
   338→            <div className="space-y-4">
   339→              {results.map((result, index) => (
   340→                <SearchResultCard key={`${result.type}-${result.id}`} result={result} index={index} />
   341→              ))}
   342→            </div>
   343→
   344→            <Pagination
   345→              currentPage={currentPage}
   346→              totalPages={totalPages}
   347→              onPageChange={handlePageChange}
   348→            />
   349→          </>
   350→        )}
   351→
   352→        {/* Empty State */}
   353→        {!isLoading && !error && query && results.length === 0 && (
   354→          <EmptyState query={query} />
   355→        )}
   356→
   357→        {/* No Query State */}
   358→        {!isLoading && !query && (
   359→          <div className="text-center py-16 rounded-2xl bg-neutral-900/30 border border-white/5">
   360→            <div className="w-14 h-14 rounded-2xl bg-neutral-900/50 flex items-center justify-center mx-auto mb-4">
   361→              <Search size={24} className="text-neutral-600" />
   362→            </div>
   363→            <h3 className="text-lg font-bold text-white mb-2">Search the forum</h3>
   364→            <p className="text-neutral-500 text-sm max-w-md mx-auto">
   365→              Enter a search term above to find posts and comments across all communities.
   366→            </p>
   367→          </div>
   368→        )}
   369→      </main>
   370→    </>
   371→  );
   372→}

   402→// ============================================
   403→// HELPER FUNCTIONS
   404→// ============================================
   405→function getTimeAgo(date: Date): string {
   406→  const now = new Date();
   407→  const seconds = Math.floor((now.getTime() - date.getTime()) / 1000);
   408→
   409→  if (seconds < 60) return 'just now';
   410→  if (seconds < 3600) return `${Math.floor(seconds / 60)}m ago`;
   411→  if (seconds < 86400) return `${Math.floor(seconds / 3600)}h ago`;
   412→  if (seconds < 604800) return `${Math.floor(seconds / 86400)}d ago`;
   413→  if (seconds < 2592000) return `${Math.floor(seconds / 604800)}w ago`;
   414→  if (seconds < 31536000) return `${Math.floor(seconds / 2592000)}mo ago`;
   415→  return `${Math.floor(seconds / 31536000)}y ago`;
   416→}
