   228→ * Get guide by ID

   230→export function getGuideById(id: string): Guide | undefined {
   231→  return GUIDES_BY_ID[id];
   232→}
   233→
   234→/**
   235→ * Get guide by slug
   236→ */
   237→export function getGuideBySlug(slug: string): Guide | undefined {
   238→  return GUIDES_BY_SLUG[slug];
   239→}

   241→/**
   242→ * Get all guides in a category
   243→ */
   244→export function getGuidesByCategory(categoryId: string): Guide[] {
   245→  const category = GUIDE_CATEGORIES.find(c => c.id === categoryId);
   246→  if (!category) return [];
   247→  return category.guideIds
   248→    .map(id => GUIDES_BY_ID[id])
   249→    .filter((g): g is Guide => g !== undefined);
   250→}
   251→
   252→/**
   253→ * Get related guides for a given guide
   254→ */
   255→export function getRelatedGuides(guideId: string): Guide[] {
   256→  const guide = GUIDES_BY_ID[guideId];
   257→  if (!guide || !guide.relatedGuides) return [];
   258→  return guide.relatedGuides
   259→    .map(id => GUIDES_BY_ID[id])
   260→    .filter((g): g is Guide => g !== undefined);
   261→}
   262→
   263→/**
   264→ * Search guides by title or description
   265→ */
   266→export function searchGuides(query: string): Guide[] {
   267→  const lowered = query.toLowerCase();
   268→  return ALL_GUIDES.filter(
   269→    g =>
   270→      g.title.toLowerCase().includes(lowered) ||
   271→      g.description.toLowerCase().includes(lowered) ||
   272→      (g.subtitle && g.subtitle.toLowerCase().includes(lowered)) ||
   273→      (g.tags && g.tags.some(tag => tag.toLowerCase().includes(lowered)))
   274→  );
   275→}

   291→/**
   292→ * Get all unique product IDs across all guides
   293→ */
   294→export function getAllGuideProductIds(): string[] {
   295→  const ids = new Set<string>();
   296→  ALL_GUIDES.forEach(guide => {
   297→    guide.productIds?.forEach(id => ids.add(id));
   298→    guide.sections.forEach(section => {
   299→      section.products?.forEach(id => ids.add(id));
   300→    });
   301→  });
   302→  return Array.from(ids);
   303→}

   305→/**
   306→ * Get guides by gender filter
   307→ */
   308→export function getGuidesByGender(gender: 'male' | 'female' | 'all'): Guide[] {
   309→  if (gender === 'all') return ALL_GUIDES;
   310→
   311→  const maleGuideIds = ['hair-loss', 'beard-growth', 'mewing-jawline', 'gynecomastia', 'neck-training'];
   312→  const femaleGuideIds = ['anti-aging-women', 'glass-skin', 'hormonal-skin', 'body-contouring-women', 'postpartum-recovery', 'feminine-features'];
   313→
   314→  if (gender === 'male') {
   315→    return ALL_GUIDES.filter(g => !femaleGuideIds.includes(g.id));
   316→  } else {
   317→    return ALL_GUIDES.filter(g => !maleGuideIds.includes(g.id));
   318→  }
   319→}
