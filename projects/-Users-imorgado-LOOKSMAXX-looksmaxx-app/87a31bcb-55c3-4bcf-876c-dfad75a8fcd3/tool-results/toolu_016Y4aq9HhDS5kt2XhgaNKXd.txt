    10→// ============================================
    11→// AMAZON ASIN MAPPING BY REGION
    12→// ============================================
    13→
    14→interface ProductASINs {
    15→  us?: string;
    16→  uk?: string;
    17→  de?: string;
    18→  fr?: string;
    19→  au?: string;
    20→}
    21→
    22→// Regions that have Amazon stores with ASINs
    23→type AmazonRegion = 'us' | 'uk' | 'de' | 'fr' | 'au';
    24→const AMAZON_REGIONS: AmazonRegion[] = ['us', 'uk', 'de', 'fr', 'au'];
    25→
    26→// Helper to build region links from ASINs
    27→function buildRegionLinks(asins: ProductASINs): Record<string, string> {
    28→  const links: Record<string, string> = {};
    29→
    30→  AMAZON_REGIONS.forEach(region => {
    31→    const asin = asins[region] || asins.us; // Fallback to US ASIN
    32→    if (asin) {
    33→      links[region] = buildAmazonUrl(asin, region);
    34→    }
    35→  });
    36→
    37→  // Asia uses US ASIN on amzn.asia domain
    38→  if (asins.us) {
    39→    links['asia'] = buildAmazonUrl(asins.us, 'asia');
    40→  }
    41→
    42→  return links;
    43→}
