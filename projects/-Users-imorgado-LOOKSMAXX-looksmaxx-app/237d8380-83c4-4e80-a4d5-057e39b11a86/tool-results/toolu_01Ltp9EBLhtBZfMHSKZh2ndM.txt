   174→// ============================================

   176→/**
   177→ * Exponential Decay Scoring Algorithm with Directional/Dimorphic Support
   178→ *
   179→ * Standard: score = maxScore * e^(-decayRate * deviation)
   180→ *
   181→ * Directional scoring (polarity):
   182→ * - 'higher_is_better': Values above safeFloor but below ideal get softZoneScore
   183→ *   Example: Canthal Tilt of 3 degrees is still positive/good, just not peak ideal (6-8 degrees)
   184→ * - 'lower_is_better': Values below safeCeiling but above ideal get softZoneScore
   185→ *   Example: Short philtrum is still attractive even if shorter than "ideal"
   186→ */
   187→export function calculateMetricScore(
   188→  value: number,
   189→  config: MetricConfig
   190→): number {
   191→  const {
   192→    id,
   193→    idealMin,
   194→    idealMax,
   195→    decayRate,
   196→    maxScore,
   197→    customCurve,
   198→    polarity = 'balanced',
   199→    safeFloor,
   200→    safeCeiling,
   201→    softZoneScore = 8.0, // Default "Good" score for acceptable-but-not-ideal values
   202→  } = config;
   203→
   204→  // IMPORTANT: Check demographic-adjusted ideal range FIRST
   205→  // This ensures that demographic overrides are respected even when Bezier curves exist
   206→  // (Bezier curves are optimized for base ranges and don't auto-adjust for demographics)
   207→  if (value >= idealMin && value <= idealMax) {
   208→    return maxScore;
   209→  }
   210→
   211→  // Use custom curve if available in config
   212→  if (customCurve && customCurve.mode === 'custom') {
   213→    return interpolateCustomCurve(value, customCurve.points, maxScore);
   214→  }
   215→
   216→  // Check for pre-defined Bezier curve from harmony curves (66 metrics)
   217→  const bezierCurve = BEZIER_CURVES[id];
   218→  if (bezierCurve && bezierCurve.mode === 'custom') {
   219→    return interpolateCustomCurve(value, bezierCurve.points, maxScore);
   220→  }
   221→
   222→  // Handle directional/dimorphic scoring
   223→  if (polarity === 'higher_is_better' && safeFloor !== undefined) {
   224→    // Higher values are good. Only values below safeFloor are true weaknesses.
   225→    if (value >= safeFloor && value < idealMin) {
   226→      // Value is in the "acceptable but not ideal" zone
   227→      // Give a passing score that decreases linearly as we approach the floor
   228→      const zoneRange = idealMin - safeFloor;
   229→      const distanceFromIdeal = idealMin - value;
   230→      const t = zoneRange > 0 ? distanceFromIdeal / zoneRange : 0;
   231→      // Linear interpolation: idealMin -> maxScore, safeFloor -> softZoneScore
   232→      return maxScore - t * (maxScore - softZoneScore);
   233→    }
   234→    if (value >= idealMax) {
   235→      // Values above ideal are still excellent for 'higher_is_better'
   236→      // Apply gentler decay (1/3 the normal rate)
   237→      const deviation = value - idealMax;
   238→      const gentleDecay = decayRate / 3;
   239→      return Math.max(softZoneScore, maxScore * Math.exp(-gentleDecay * deviation));
   240→    }
   241→    // Below safeFloor - apply normal (or harsher) exponential decay
   242→    if (value < safeFloor) {
   243→      const deviation = safeFloor - value;
   244→      return Math.max(0, softZoneScore * Math.exp(-decayRate * deviation));
   245→    }
   246→  }
   247→
   248→  if (polarity === 'lower_is_better' && safeCeiling !== undefined) {
   249→    // Lower values are good. Only values above safeCeiling are true weaknesses.
   250→    if (value <= safeCeiling && value > idealMax) {
   251→      // Value is in the "acceptable but not ideal" zone
   252→      const zoneRange = safeCeiling - idealMax;
   253→      const distanceFromIdeal = value - idealMax;
   254→      const t = zoneRange > 0 ? distanceFromIdeal / zoneRange : 0;
   255→      // Linear interpolation: idealMax -> maxScore, safeCeiling -> softZoneScore
   256→      return maxScore - t * (maxScore - softZoneScore);
   257→    }
   258→    if (value <= idealMin) {
   259→      // Values below ideal are still excellent for 'lower_is_better'
   260→      // Apply gentler decay (1/3 the normal rate)
   261→      const deviation = idealMin - value;
   262→      const gentleDecay = decayRate / 3;
   263→      return Math.max(softZoneScore, maxScore * Math.exp(-gentleDecay * deviation));
   264→    }
   265→    // Above safeCeiling - apply normal exponential decay
   266→    if (value > safeCeiling) {
   267→      const deviation = value - safeCeiling;
   268→      return Math.max(0, softZoneScore * Math.exp(-decayRate * deviation));
   269→    }
   270→  }
   271→
   272→  // Default balanced scoring: deviation in either direction is equally bad
   273→  const deviation = value < idealMin
   274→    ? idealMin - value
   275→    : value - idealMax;
   276→
   277→  // Exponential decay
   278→  const score = maxScore * Math.exp(-decayRate * deviation);
   279→
   280→  return Math.max(0, Math.min(maxScore, score));
   281→}

   283→/**
   284→ * Check if a metric value is in an "acceptable" zone (not a true weakness).
   285→ * Used by InsightsEngine to prevent false positives.
   286→ */
   287→export function isValueAcceptable(
   288→  value: number,
   289→  config: MetricConfig
   290→): { acceptable: boolean; reason: string } {
   291→  const {
   292→    idealMin,
   293→    idealMax,
   294→    polarity = 'balanced',
   295→    safeFloor,
   296→    safeCeiling,
   297→  } = config;
   298→
   299→  // Within ideal range is always acceptable
   300→  if (value >= idealMin && value <= idealMax) {
   301→    return { acceptable: true, reason: 'within_ideal' };
   302→  }
   303→
   304→  // Check directional acceptability
   305→  if (polarity === 'higher_is_better') {
   306→    if (value >= idealMax) {
   307→      return { acceptable: true, reason: 'above_ideal_higher_is_better' };
   308→    }
   309→    if (safeFloor !== undefined && value >= safeFloor) {
   310→      return { acceptable: true, reason: 'above_safe_floor' };
   311→    }
   312→    // Below safe floor - true weakness
   313→    return { acceptable: false, reason: 'below_safe_floor' };
   314→  }
   315→
   316→  if (polarity === 'lower_is_better') {
   317→    if (value <= idealMin) {
   318→      return { acceptable: true, reason: 'below_ideal_lower_is_better' };
   319→    }
   320→    if (safeCeiling !== undefined && value <= safeCeiling) {
   321→      return { acceptable: true, reason: 'below_safe_ceiling' };
   322→    }
   323→    // Above safe ceiling - true weakness
   324→    return { acceptable: false, reason: 'above_safe_ceiling' };
   325→  }
   326→
   327→  // Balanced polarity - outside ideal is not acceptable
   328→  return { acceptable: false, reason: 'outside_ideal_balanced' };
   329→}

   331→/**
   332→ * Interpolate custom Bezier curve for scoring
   333→ * Uses cubic Bezier interpolation with control handles for smooth curves
   334→ */
   335→export function interpolateCustomCurve(
   336→  value: number,
   337→  points: CurvePoint[],
   338→  maxScore: number
   339→): number {
   340→  void maxScore; // Reserved for future custom curve implementations
   341→  if (points.length === 0) return 0;
   342→
   343→  // Sort points by x value
   344→  const sortedPoints = [...points].sort((a, b) => a.x - b.x);
   345→
   346→  // Clamp to range
   347→  if (value <= sortedPoints[0].x) return sortedPoints[0].y;
   348→  if (value >= sortedPoints[sortedPoints.length - 1].x) {
   349→    return sortedPoints[sortedPoints.length - 1].y;
   350→  }
   351→
   352→  // Find bracketing points
   353→  let lowerIndex = 0;
   354→  for (let i = 0; i < sortedPoints.length - 1; i++) {
   355→    if (value >= sortedPoints[i].x && value <= sortedPoints[i + 1].x) {
   356→      lowerIndex = i;
   357→      break;
   358→    }
   359→  }
   360→
   361→  const p0 = sortedPoints[lowerIndex];
   362→  const p3 = sortedPoints[lowerIndex + 1];
   363→
   364→  // Check if we have Bezier control handles
   365→  const hasHandles = p0.rightHandleX !== undefined && p3.leftHandleX !== undefined;
   366→
   367→  if (hasHandles) {
   368→    // Cubic Bezier interpolation with control points
   369→    const p1x = p0.rightHandleX!;
   370→    const p1y = p0.rightHandleY ?? p0.y;
   371→    const p2x = p3.leftHandleX!;
   372→    const p2y = p3.leftHandleY ?? p3.y;
   373→
   374→    // Find t parameter for given x value using Newton-Raphson iteration
   375→    let t = (value - p0.x) / (p3.x - p0.x); // Initial guess
   376→
   377→    // Newton-Raphson iterations to find t where B_x(t) = value
   378→    for (let iter = 0; iter < 10; iter++) {
   379→      const bx = cubicBezier(p0.x, p1x, p2x, p3.x, t);
   380→      const bxDerivative = cubicBezierDerivative(p0.x, p1x, p2x, p3.x, t);
   381→
   382→      if (Math.abs(bxDerivative) < 1e-10) break;
   383→
   384→      const newT = t - (bx - value) / bxDerivative;
   385→      if (Math.abs(newT - t) < 1e-10) break;
   386→      t = Math.max(0, Math.min(1, newT));
   387→    }
   388→
   389→    // Calculate y using the found t parameter
   390→    return cubicBezier(p0.y, p1y, p2y, p3.y, t);
   391→  } else {
   392→    // Smooth interpolation using Catmull-Rom spline
   393→    // This provides smoother curves than linear interpolation
   394→    const t = (value - p0.x) / (p3.x - p0.x);
   395→
   396→    // Get neighboring points for spline calculation
   397→    const pMinus1 = lowerIndex > 0 ? sortedPoints[lowerIndex - 1] : p0;
   398→    const p4 = lowerIndex + 2 < sortedPoints.length ? sortedPoints[lowerIndex + 2] : p3;
   399→
   400→    // Catmull-Rom spline interpolation
   401→    return catmullRomSpline(pMinus1.y, p0.y, p3.y, p4.y, t);
   402→  }
   403→}
   404→
   405→/**
   406→ * Cubic Bezier curve evaluation: B(t) = (1-t)^3*P0 + 3*(1-t)^2*t*P1 + 3*(1-t)*t^2*P2 + t^3*P3
   407→ */
   408→export function cubicBezier(p0: number, p1: number, p2: number, p3: number, t: number): number {
   409→  const mt = 1 - t;
   410→  const mt2 = mt * mt;
   411→  const mt3 = mt2 * mt;
   412→  const t2 = t * t;
   413→  const t3 = t2 * t;
   414→  return mt3 * p0 + 3 * mt2 * t * p1 + 3 * mt * t2 * p2 + t3 * p3;
   415→}
   416→
   417→/**
   418→ * Cubic Bezier derivative: B'(t) = 3*(1-t)^2*(P1-P0) + 6*(1-t)*t*(P2-P1) + 3*t^2*(P3-P2)
   419→ */
   420→export function cubicBezierDerivative(p0: number, p1: number, p2: number, p3: number, t: number): number {
   421→  const mt = 1 - t;
   422→  return 3 * mt * mt * (p1 - p0) + 6 * mt * t * (p2 - p1) + 3 * t * t * (p3 - p2);
   423→}

   425→/**
   426→ * Catmull-Rom spline interpolation for smooth curves without explicit handles
   427→ */
   428→export function catmullRomSpline(p0: number, p1: number, p2: number, p3: number, t: number): number {
   429→  const t2 = t * t;
   430→  const t3 = t2 * t;
   431→
   432→  // Catmull-Rom coefficients with tension = 0.5
   433→  const a = -0.5 * p0 + 1.5 * p1 - 1.5 * p2 + 0.5 * p3;
   434→  const b = p0 - 2.5 * p1 + 2 * p2 - 0.5 * p3;
   435→  const c = -0.5 * p0 + 0.5 * p2;
   436→  const d = p1;
   437→
   438→  return a * t3 + b * t2 + c * t + d;
   439→}

   530→/**
   531→ * Calculate angle at vertex point (in degrees)
   532→ */
   533→export function calculateAngle(p1: Point, vertex: Point, p2: Point): number {
   534→  const v1 = { x: p1.x - vertex.x, y: p1.y - vertex.y };
   535→  const v2 = { x: p2.x - vertex.x, y: p2.y - vertex.y };
   536→
   537→  const dot = v1.x * v2.x + v1.y * v2.y;
   538→  const cross = v1.x * v2.y - v1.y * v2.x;
   539→
   540→  const angle = Math.atan2(cross, dot) * (180 / Math.PI);
   541→  return Math.abs(angle);
   542→}

   544→/**
   545→ * Calculate perpendicular distance from point to line (signed)
   546→ */
   547→export function perpendicularDistance(
   548→  point: Point,
   549→  lineStart: Point,
   550→  lineEnd: Point
   551→): number {
   552→  const A = point.x - lineStart.x;
   553→  const B = point.y - lineStart.y;
   554→  const C = lineEnd.x - lineStart.x;
   555→  const D = lineEnd.y - lineStart.y;
   556→
   557→  const dot = A * C + B * D;
   558→  const lenSq = C * C + D * D;
   559→  const param = lenSq !== 0 ? dot / lenSq : -1;
   560→
   561→  let xx: number, yy: number;
   562→
   563→  if (param < 0) {
   564→    xx = lineStart.x;
   565→    yy = lineStart.y;
   566→  } else if (param > 1) {
   567→    xx = lineEnd.x;
   568→    yy = lineEnd.y;
   569→  } else {
   570→    xx = lineStart.x + param * C;
   571→    yy = lineStart.y + param * D;
   572→  }
   573→
   574→  const dx = point.x - xx;
   575→  const dy = point.y - yy;
   576→
   577→  const sign =
   578→    (lineEnd.y - lineStart.y) * point.x -
   579→      (lineEnd.x - lineStart.x) * point.y +
   580→      lineEnd.x * lineStart.y -
   581→      lineEnd.y * lineStart.x >=
   582→    0
   583→      ? 1
   584→      : -1;
   585→
   586→  return sign * Math.sqrt(dx * dx + dy * dy);
   587→}

   593→/**
   594→ * Calculate percentile from harmony score
   595→ */
   596→export function calculateHarmonyPercentile(score: number): number {
   597→  // Based on normal distribution with mean=5, stdDev=1.5
   598→  const mean = 5;
   599→  const stdDev = 1.5;
   600→  const z = (score - mean) / stdDev;
   601→  return normalCDF(z) * 100;
   602→}
   603→
   604→/**
   605→ * Standard normal CDF approximation
   606→ */
   607→export function normalCDF(z: number): number {
   608→  const a1 = 0.254829592;
   609→  const a2 = -0.284496736;
   610→  const a3 = 1.421413741;
   611→  const a4 = -1.453152027;
   612→  const a5 = 1.061405429;
   613→  const p = 0.3275911;
   614→
   615→  const sign = z < 0 ? -1 : 1;
   616→  const absZ = Math.abs(z) / Math.sqrt(2);
   617→
   618→  const t = 1.0 / (1.0 + p * absZ);
   619→  const y = 1.0 - ((((a5 * t + a4) * t + a3) * t + a2) * t + a1) * t * Math.exp(-absZ * absZ);
   620→
   621→  return 0.5 * (1.0 + sign * y);
   622→}
