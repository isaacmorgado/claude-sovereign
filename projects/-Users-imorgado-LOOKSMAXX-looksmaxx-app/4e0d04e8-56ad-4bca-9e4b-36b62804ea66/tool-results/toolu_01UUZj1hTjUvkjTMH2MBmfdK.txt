   278→
   279→export function getScoreColor(score: number): string {
   280→  if (score >= 8) return '#67e8f9';  // Cyan - ideal
   281→  if (score >= 6) return '#22c55e';  // Green - good
   282→  if (score >= 4) return '#fbbf24';  // Yellow - fair
   283→  return 'rgb(227, 67, 67)';         // Red - poor
   284→}
   285→
   286→export function getQualityColor(quality: QualityTier): string {
   287→  switch (quality) {
   288→    case 'ideal': return '#67e8f9';
   289→    case 'excellent': return '#22c55e';
   290→    case 'good': return '#fbbf24';
   291→    case 'below_average': return 'rgb(227, 67, 67)';
   292→    default: return '#6b7280';
   293→  }
   294→}
   295→
   296→export function getSeverityColor(severity: SeverityLevel): string {
   297→  switch (severity) {
   298→    case 'optimal': return '#67e8f9';
   299→    case 'minor': return '#22c55e';
   300→    case 'moderate': return '#fbbf24';
   301→    case 'major': return '#f97316';
   302→    case 'severe': return '#ef4444';
   303→    case 'extremely_severe': return '#dc2626';
   304→    default: return '#6b7280';
   305→  }
   306→}
   307→
   308→export function getCategoryColor(categoryName: string): string {
   309→  const cat = MEASUREMENT_CATEGORIES.find(c =>
   310→    c.name.toLowerCase().includes(categoryName.toLowerCase()) ||
   311→    categoryName.toLowerCase().includes(c.name.toLowerCase())
   312→  );
   313→  return cat?.color || '#6b7280';
   314→}

   316→// ============================================
   317→// UNIT FORMATTING
   318→// ============================================
   319→
   320→export function formatUnit(unit: 'x' | 'mm' | '%' | '°' | MeasurementUnit): string {
   321→  switch (unit) {
   322→    case 'x':
   323→    case 'ratio': return 'x';
   324→    case 'mm': return 'mm';
   325→    case '%':
   326→    case 'percent': return '%';
   327→    case '°':
   328→    case 'degrees': return '°';
   329→    default: return '';
   330→  }
   331→}
   332→
   333→export function formatValue(value: number, unit: 'x' | 'mm' | '%' | '°' | MeasurementUnit): string {
   334→  const unitStr = formatUnit(unit);
   335→  if (unit === '%' || unit === 'percent') {
   336→    return `${value.toFixed(1)}${unitStr}`;
   337→  }
   338→  if (unit === '°' || unit === 'degrees') {
   339→    return `${value.toFixed(1)}${unitStr}`;
   340→  }
   341→  return `${value.toFixed(2)}${unitStr}`;
   342→}
