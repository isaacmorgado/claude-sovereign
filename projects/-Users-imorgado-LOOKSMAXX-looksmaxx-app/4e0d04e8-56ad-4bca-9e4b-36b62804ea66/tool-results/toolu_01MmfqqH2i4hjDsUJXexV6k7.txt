    59→// =============================================================================
    60→// WEIGHTED HARMONY SCORE CALCULATION
    61→// =============================================================================
    62→
    63→export interface HarmonyScoreResult {
    64→  /** Weighted harmony percentage (0-100%) */
    65→  harmonyPercentage: number;
    66→  /** Weighted average score (0-10 scale) */
    67→  weightedAverage: number;
    68→  /** Unweighted average for comparison */
    69→  unweightedAverage: number;
    70→  /** Breakdown by weight tier */
    71→  weightDistribution: {
    72→    highImpact: { count: number; avgScore: number };
    73→    mediumImpact: { count: number; avgScore: number };
    74→    standard: { count: number; avgScore: number };
    75→  };
    76→}
    77→
    78→/**
    79→ * Calculate the weighted Harmony Score from scored measurements.
    80→ * Uses 3-tier weighting: High (3x), Medium (2x), Standard (1x)
    81→ *
    82→ * Formula: Σ(Score × Weight) / Σ(Weights) → percentage
    83→ */
    84→export function calculateHarmonyScore(
    85→  measurements: Array<{ metricId: string; standardizedScore: number }>
    86→): HarmonyScoreResult {
    87→  let totalWeightedScore = 0;
    88→  let totalWeight = 0;
    89→  let unweightedTotal = 0;
    90→
    91→  const tiers = {
    92→    high: { total: 0, count: 0 },
    93→    medium: { total: 0, count: 0 },
    94→    standard: { total: 0, count: 0 },
    95→  };
    96→
    97→  for (const m of measurements) {
    98→    const weight = getMetricWeight(m.metricId);
    99→    const tier = getWeightTier(m.metricId);
   100→
   101→    totalWeightedScore += m.standardizedScore * weight;
   102→    totalWeight += weight;
   103→    unweightedTotal += m.standardizedScore;
   104→
   105→    tiers[tier].total += m.standardizedScore;
   106→    tiers[tier].count++;
   107→  }
   108→
   109→  const rawWeightedAverage = totalWeight > 0 ? totalWeightedScore / totalWeight : 0;
   110→  const rawUnweightedAverage = measurements.length > 0 ? unweightedTotal / measurements.length : 0;
   111→
   112→  // Clamp averages to 0-10 range to prevent any edge case overflows
   113→  const weightedAverage = Math.max(0, Math.min(10, rawWeightedAverage));
   114→  const unweightedAverage = Math.max(0, Math.min(10, rawUnweightedAverage));
   115→  const harmonyPercentage = Math.max(0, Math.min(100, (weightedAverage / 10) * 100));
   116→
   117→  // Helper to clamp tier averages to 0-10
   118→  const clampAvg = (total: number, count: number): number => {
   119→    if (count === 0) return 0;
   120→    return Math.max(0, Math.min(10, total / count));
   121→  };
   122→
   123→  return {
   124→    harmonyPercentage,
   125→    weightedAverage,
   126→    unweightedAverage,
   127→    weightDistribution: {
   128→      highImpact: {
   129→        count: tiers.high.count,
   130→        avgScore: clampAvg(tiers.high.total, tiers.high.count),
   131→      },
   132→      mediumImpact: {
   133→        count: tiers.medium.count,
   134→        avgScore: clampAvg(tiers.medium.total, tiers.medium.count),
   135→      },
   136→      standard: {
   137→        count: tiers.standard.count,
   138→        avgScore: clampAvg(tiers.standard.total, tiers.standard.count),
   139→      },
   140→    },
   141→  };
   142→}

   367→/**
   368→ * Generate personalized advice based on metric score and value position.
   369→ *
   370→ * @param metricId - The metric identifier
   371→ * @param score - Score from 0-10
   372→ * @param value - Actual measured value
   373→ * @param idealMin - Minimum of ideal range
   374→ * @param idealMax - Maximum of ideal range
   375→ * @returns Advice string for the metric
   376→ */
   377→export function generateAdvice(
   378→  metricId: string,
   379→  score: number,
   380→  value: number,
   381→  idealMin: number,
   382→  idealMax: number
   383→): string {
   384→  // Perfect score - no action needed
   385→  if (score >= 9.0) {
   386→    return "Ideal proportions. No action needed.";
   387→  }
   388→
   389→  // Get advice from database
   390→  const advice = ADVICE_DATABASE[metricId];
   391→  if (advice) {
   392→    if (value > idealMax) {
   393→      return advice.tooHigh;
   394→    } else if (value < idealMin) {
   395→      return advice.tooLow;
   396→    }
   397→  }
   398→
   399→  // Default fallback
   400→  if (value > idealMax) {
   401→    return `Value (${value.toFixed(2)}) is above the ideal range of ${idealMin.toFixed(1)}-${idealMax.toFixed(1)}.`;
   402→  } else if (value < idealMin) {
   403→    return `Value (${value.toFixed(2)}) is below the ideal range of ${idealMin.toFixed(1)}-${idealMax.toFixed(1)}.`;
   404→  } else {
   405→    return "Value is within acceptable range but not optimal.";
   406→  }
   407→}

   410→ * Generate advice for a metric score result
   411→ */
   412→export function generateAdviceFromResult(result: MetricScoreResult): string {
   413→  return generateAdvice(

   415→    result.standardizedScore,
   416→    result.value,
   417→    result.idealMin,
   418→    result.idealMax
   419→  );
   420→}

   422→// =============================================================================
   423→// GAUSSIAN SCORING (matches Python implementation)
   424→// =============================================================================
   425→
   426→/**
   427→ * Calculate a score from 0-10 using Gaussian decay.
   428→ * Matches the Python looksmax_engine.py implementation.
   429→ *
   430→ * @param value - The measured value
   431→ * @param minRange - Minimum of ideal range
   432→ * @param maxRange - Maximum of ideal range
   433→ * @returns Score from 0-10
   434→ */
   435→export function calculateGaussianScore(
   436→  value: number,
   437→  minRange: number,
   438→  maxRange: number
   439→): number {
   440→  // Within ideal range = perfect score
   441→  if (value >= minRange && value <= maxRange) {
   442→    return 10.0;
   443→  }
   444→
   445→  let deviation: number;
   446→  let target: number;
   447→
   448→  if (value < minRange) {
   449→    deviation = minRange - value;
   450→    target = minRange;
   451→  } else {
   452→    deviation = value - maxRange;
   453→    target = maxRange;
   454→  }
   455→
   456→  // Prevent division by zero
   457→  if (target === 0) target = 1.0;
   458→
   459→  // Sigma is 10% of target value
   460→  let sigma = Math.abs(target) * 0.1;
   461→  if (sigma === 0) sigma = 1.0;
   462→
   463→  // Gaussian decay
   464→  const score = 10.0 * Math.exp(-0.5 * Math.pow(deviation / sigma, 2));
   465→  return Math.max(0.0, Math.min(10.0, score));
   466→}

   484→/**
   485→ * Get the top N metrics (strengths) sorted by score descending
   486→ */
   487→export function getTopMetrics(
   488→  measurements: MetricScoreResult[],
   489→  count: number = 3
   490→): RankedMetric[] {
   491→  return [...measurements]
   492→    .sort((a, b) => b.standardizedScore - a.standardizedScore)
   493→    .slice(0, count)
   494→    .map((m) => ({
   495→      metricId: m.metricId,
   496→      name: m.name,
   497→      value: m.value,
   498→      score: m.standardizedScore,
   499→      idealMin: m.idealMin,
   500→      idealMax: m.idealMax,
   501→      advice: generateAdviceFromResult(m),
   502→      weightTier: getWeightTier(m.metricId),
   503→      category: m.category,
   504→    }));
   505→}
   506→
   507→/**
   508→ * Get the bottom N metrics (areas to improve) sorted by score ascending
   509→ */
   510→export function getBottomMetrics(
   511→  measurements: MetricScoreResult[],
   512→  count: number = 3
   513→): RankedMetric[] {
   514→  return [...measurements]
   515→    .sort((a, b) => a.standardizedScore - b.standardizedScore)
   516→    .slice(0, count)
   517→    .map((m) => ({
   518→      metricId: m.metricId,
   519→      name: m.name,
   520→      value: m.value,
   521→      score: m.standardizedScore,
   522→      idealMin: m.idealMin,
   523→      idealMax: m.idealMax,
   524→      advice: generateAdviceFromResult(m),
   525→      weightTier: getWeightTier(m.metricId),
   526→      category: m.category,
   527→    }));
   528→}
