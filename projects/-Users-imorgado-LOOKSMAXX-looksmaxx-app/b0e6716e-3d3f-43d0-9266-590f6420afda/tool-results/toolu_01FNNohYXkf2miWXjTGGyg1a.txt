   456→/**
   457→ * Get all guides in a category
   458→ */
   459→export function getGuidesByCategory(categoryId: string): Guide[] {
   460→  const category = GUIDE_CATEGORIES.find(c => c.id === categoryId);
   461→  if (!category) return [];
   462→  return category.guideIds
   463→    .map(id => GUIDES_BY_ID[id])
   464→    .filter((g): g is Guide => g !== undefined);
   465→}
   466→
   467→/**
   468→ * Get related guides for a given guide
   469→ */
   470→export function getRelatedGuides(guideId: string): Guide[] {
   471→  const guide = GUIDES_BY_ID[guideId];
   472→  if (!guide || !guide.relatedGuides) return [];
   473→  return guide.relatedGuides
   474→    .map(id => GUIDES_BY_ID[id])
   475→    .filter((g): g is Guide => g !== undefined);
   476→}

   478→/**
   479→ * Search guides by title or description
   480→ */
   481→export function searchGuides(query: string): Guide[] {
   482→  const lowered = query.toLowerCase();
   483→  return ALL_GUIDES.filter(
   484→    g =>
   485→      g.title.toLowerCase().includes(lowered) ||
   486→      g.description.toLowerCase().includes(lowered) ||
   487→      (g.subtitle && g.subtitle.toLowerCase().includes(lowered)) ||
   488→      (g.tags && g.tags.some(tag => tag.toLowerCase().includes(lowered)))
   489→  );
   490→}

   506→/**
   507→ * Get all unique product IDs across all guides
   508→ */
   509→export function getAllGuideProductIds(): string[] {
   510→  const ids = new Set<string>();
   511→  ALL_GUIDES.forEach(guide => {
   512→    guide.productIds?.forEach(id => ids.add(id));
   513→    guide.sections.forEach(section => {
   514→      section.products?.forEach(id => ids.add(id));
   515→    });
   516→  });
   517→  return Array.from(ids);
   518→}

   529→  if (gender === 'male') {
   530→    return ALL_GUIDES.filter(g => !femaleGuideIds.includes(g.id));
   531→  } else {
   532→    return ALL_GUIDES.filter(g => !maleGuideIds.includes(g.id));
   533→  }
   534→}
