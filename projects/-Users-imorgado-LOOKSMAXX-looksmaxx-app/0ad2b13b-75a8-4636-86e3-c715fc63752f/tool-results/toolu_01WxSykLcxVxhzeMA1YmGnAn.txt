    79→function convertUnitToRatioUnit(unit: string): 'x' | 'mm' | '%' | '°' {
    80→  switch (unit) {
    81→    case 'ratio': return 'x';
    82→    case 'percent': return '%';
    83→    case 'degrees': return '°';
    84→    case 'mm': return 'mm';
    85→    default: return 'x';
    86→  }
    87→}
    88→
    89→function transformToRatio(result: FaceIQScoreResult, landmarks: LandmarkPoint[]): Ratio {
    90→  const metricConfig = FACEIQ_METRICS[result.metricId];
    91→
    92→  // Generate illustration based on metric
    93→  const illustration = generateIllustration(result.metricId, landmarks);
    94→
    95→  // Get flaw/strength mappings
    96→  const { mayIndicateFlaws, mayIndicateStrengths } = getFlawStrengthMappings(result);
    97→
    98→  return {
    99→    id: result.metricId,
   100→    name: result.name,
   101→    value: result.value,
   102→    score: result.score,
   103→    standardizedScore: result.standardizedScore,
   104→    unit: convertUnitToRatioUnit(result.unit),
   105→    idealMin: result.idealMin,
   106→    idealMax: result.idealMax,
   107→    rangeMin: metricConfig?.rangeMin || result.idealMin - 0.5,
   108→    rangeMax: metricConfig?.rangeMax || result.idealMax + 0.5,
   109→    description: metricConfig?.description || '',
   110→    category: result.category,
   111→    qualityLevel: result.qualityTier,
   112→    severity: result.severity,
   113→    illustration,
   114→    mayIndicateFlaws,
   115→    mayIndicateStrengths,
   116→    usedLandmarks: getUsedLandmarks(result.metricId),
   117→    scoringCurveConfig: metricConfig ? {
   118→      decayRate: metricConfig.decayRate,
   119→      maxScore: metricConfig.maxScore,
   120→    } : undefined,
   121→  };
   122→}

   693→function generateIllustration(metricId: string, landmarks: LandmarkPoint[]): Ratio['illustration'] {
   694→  const landmarkMap: Record<string, LandmarkPoint> = {};
   695→  landmarks.forEach(l => { landmarkMap[l.id] = l; });
   696→
   697→  const config = ILLUSTRATION_CONFIGS[metricId];
   698→  const points: Record<string, { type: 'landmark'; landmarkId: string }> = {};
   699→  const lines: Record<string, { from: string; to: string; color: string; label?: string; labelPosition?: 'start' | 'middle' | 'end' }> = {};
   700→
   701→  if (config) {
   702→    config.points.forEach((pointId, i) => {
   703→      if (landmarkMap[pointId]) {
   704→        points[`point_${i}`] = { type: 'landmark', landmarkId: pointId };
   705→      }
   706→    });
   707→    config.lines.forEach((line, i) => {
   708→      lines[`line_${i}`] = {
   709→        from: line.from,
   710→        to: line.to,
   711→        color: line.color || '#67e8f9',
   712→        label: line.label,
   713→        labelPosition: line.labelPosition,
   714→      };
   715→    });
   716→  }
   717→
   718→  return { points, lines };
   719→}

   781→function generateStrengthsFromAnalysis(analysis: HarmonyAnalysis): Strength[] {
   782→  const strengths: Strength[] = [];
   783→
   784→  analysis.strengths.forEach((s, i) => {
   785→    const matchingMeasurement = analysis.measurements.find(m => m.metricId === s.metricId);
   786→
   787→    strengths.push({
   788→      id: `strength_${i}`,
   789→      strengthName: s.metricName,
   790→      summary: s.reasoning,
   791→      avgScore: matchingMeasurement?.score || 8,
   792→      qualityLevel: s.qualityTier,
   793→      categoryName: s.category,
   794→      responsibleRatios: [{
   795→        ratioName: s.metricName,
   796→        ratioId: s.metricId,
   797→        score: matchingMeasurement?.score || 8,
   798→        value: s.value,
   799→      }],
   800→    });
   801→  });
   802→
   803→  return strengths;
   804→}

   806→function generateFlawsFromAnalysis(analysis: HarmonyAnalysis): Flaw[] {
   807→  const flaws: Flaw[] = [];
   808→  let rollingLost = 0;
   809→
   810→  analysis.flaws.forEach((f, i) => {
   811→    const matchingMeasurement = analysis.measurements.find(m => m.metricId === f.metricId);
   812→    const impact = matchingMeasurement ? (10 - matchingMeasurement.score) * 0.5 : 2;
   813→    rollingLost += impact;
   814→
   815→    flaws.push({
   816→      id: `flaw_${i}`,
   817→      flawName: f.metricName,
   818→      summary: f.reasoning,
   819→      harmonyPercentageLost: impact,
   820→      standardizedImpact: impact / 10,
   821→      categoryName: f.category,
   822→      responsibleRatios: [{
   823→        ratioName: f.metricName,
   824→        ratioId: f.metricId,
   825→        score: matchingMeasurement?.score || 3,
   826→        value: matchingMeasurement?.value || 0,
   827→      }],
   828→      rollingPointsDeducted: rollingLost,
   829→      rollingHarmonyPercentageLost: rollingLost,
   830→      rollingStandardizedImpact: rollingLost / 10,
   831→    });
   832→  });
   833→
   834→  return flaws.sort((a, b) => b.harmonyPercentageLost - a.harmonyPercentageLost);
   835→}

  1213→// Calculate relevance score based on multiple factors including metric scores
  1214→function calculateRelevanceScore(
  1215→  proc: ProcedureConfig,
  1216→  flaw: Flaw,
  1217→  allFlaws: Flaw[],
  1218→  allRatios: Ratio[] = []
  1219→): number {
  1220→  let score = 0;
  1221→
  1222→  // Check metric ID match (highest priority)
  1223→  const metricId = flaw.responsibleRatios[0]?.ratioId || '';
  1224→  const metricScore = flaw.responsibleRatios[0]?.score || 5;
  1225→
  1226→  if (proc.targetMetrics.includes(metricId)) {
  1227→    // Base score for metric match
  1228→    score += 40;
  1229→
  1230→    // Boost based on how poor the metric score is (lower score = more improvement potential)
  1231→    // Score of 2 adds +20, score of 8 adds +5
  1232→    const metricPoorness = Math.max(0, 10 - metricScore) * 2;
  1233→    score += metricPoorness;
  1234→  }
  1235→
  1236→  // Check category match
  1237→  if (proc.targetCategories.some(cat =>
  1238→    flaw.categoryName.toLowerCase().includes(cat.toLowerCase()) ||
  1239→    cat.toLowerCase().includes(flaw.categoryName.toLowerCase())
  1240→  )) {
  1241→    score += 25;
  1242→  }
  1243→
  1244→  // Check keyword match with weighted scoring
  1245→  const flawLower = flaw.flawName.toLowerCase();
  1246→  const summaryLower = flaw.summary.toLowerCase();
  1247→  const matchedKeywords = proc.targetKeywords.filter(kw =>
  1248→    flawLower.includes(kw.toLowerCase()) || summaryLower.includes(kw.toLowerCase())
  1249→  );
  1250→  // Primary keyword match (in flaw name) worth more
  1251→  const primaryMatches = matchedKeywords.filter(kw => flawLower.includes(kw.toLowerCase()));
  1252→  score += primaryMatches.length * 15;
  1253→  score += (matchedKeywords.length - primaryMatches.length) * 8;
  1254→
  1255→  // Adjust by flaw severity/impact (more severe = higher priority)
  1256→  score += Math.min(flaw.harmonyPercentageLost * 4, 20);
  1257→
  1258→  // Bonus if this procedure addresses multiple flaws
  1259→  const otherFlawsAddressed = allFlaws.filter(f =>
  1260→    f.id !== flaw.id && (
  1261→      proc.targetMetrics.includes(f.responsibleRatios[0]?.ratioId || '') ||
  1262→      proc.targetCategories.some(cat => f.categoryName.toLowerCase().includes(cat.toLowerCase()))
  1263→    )
  1264→  );
  1265→  score += otherFlawsAddressed.length * 7;
  1266→
  1267→  // Check if procedure targets any below-average ratios directly
  1268→  const matchingRatios = allRatios.filter(r =>
  1269→    proc.targetMetrics.includes(r.id) && r.score < 6
  1270→  );
  1271→  score += matchingRatios.length * 8;
  1272→
  1273→  return Math.min(score, 100);
  1274→}

  1276→// Calculate expected score improvement based on metric deviation and procedure effectiveness
  1277→function calculateExpectedImprovement(
  1278→  proc: ProcedureConfig,
  1279→  matchedRatios: Ratio[],
  1280→  avgFlawImpact: number
  1281→): { min: number; max: number; potential: number } {
  1282→  // Base improvement from procedure
  1283→  const baseMin = proc.expectedImprovementRange.min;
  1284→  const baseMax = proc.expectedImprovementRange.max;
  1285→
  1286→  // Severity multiplier: more severe issues have more room for improvement
  1287→  const severityMultiplier = Math.min(avgFlawImpact / 2.5, 1.8);
  1288→
  1289→  // Metric deviation bonus: further from ideal = more potential improvement
  1290→  let deviationBonus = 1.0;
  1291→  if (matchedRatios.length > 0) {
  1292→    const avgScore = matchedRatios.reduce((sum, r) => sum + r.score, 0) / matchedRatios.length;
  1293→    // Score of 3 gives 1.4x, score of 7 gives 1.0x
  1294→    deviationBonus = 1 + Math.max(0, (7 - avgScore) * 0.1);
  1295→  }
  1296→
  1297→  const adjustedMin = baseMin * severityMultiplier * deviationBonus;
  1298→  const adjustedMax = baseMax * severityMultiplier * deviationBonus;
  1299→
  1300→  // Calculate potential new score if treatment is applied
  1301→  const avgCurrentScore = matchedRatios.length > 0
  1302→    ? matchedRatios.reduce((sum, r) => sum + r.score, 0) / matchedRatios.length
  1303→    : 5;
  1304→  const potentialScore = Math.min(10, avgCurrentScore + (adjustedMax * 0.8));
  1305→
  1306→  return {
  1307→    min: Math.round(adjustedMin * 100) / 100,
  1308→    max: Math.round(adjustedMax * 100) / 100,
  1309→    potential: Math.round(potentialScore * 10) / 10,
  1310→  };
  1311→}

  1313→function generateRecommendations(flaws: Flaw[], frontRatios: Ratio[] = [], sideRatios: Ratio[] = []): Recommendation[] {
  1314→  const recommendations: Recommendation[] = [];
  1315→  const allRatios = [...frontRatios, ...sideRatios];
  1316→
  1317→  // Track which procedures have been added and their matched flaws
  1318→  const procMatches: Map<string, {
  1319→    flaws: string[];
  1320→    ratios: string[];
  1321→    ratioIds: string[];
  1322→    maxScore: number;
  1323→    totalFlawImpact: number;
  1324→  }> = new Map();
  1325→
  1326→  flaws.forEach(flaw => {
  1327→    PROCEDURE_DATABASE.forEach(proc => {
  1328→      // Pass allRatios to the enhanced relevance scoring
  1329→      const relevanceScore = calculateRelevanceScore(proc, flaw, flaws, allRatios);
  1330→
  1331→      if (relevanceScore >= 25) {  // Minimum threshold
  1332→        const existing = procMatches.get(proc.ref_id);
  1333→        if (existing) {
  1334→          existing.flaws.push(flaw.flawName);
  1335→          existing.ratios.push(...flaw.responsibleRatios.map(r => r.ratioName));
  1336→          existing.ratioIds.push(...flaw.responsibleRatios.map(r => r.ratioId));
  1337→          existing.maxScore = Math.max(existing.maxScore, relevanceScore);
  1338→          existing.totalFlawImpact += flaw.harmonyPercentageLost;
  1339→        } else {
  1340→          procMatches.set(proc.ref_id, {
  1341→            flaws: [flaw.flawName],
  1342→            ratios: flaw.responsibleRatios.map(r => r.ratioName),
  1343→            ratioIds: flaw.responsibleRatios.map(r => r.ratioId),
  1344→            maxScore: relevanceScore,
  1345→            totalFlawImpact: flaw.harmonyPercentageLost,
  1346→          });
  1347→        }
  1348→      }
  1349→    });
  1350→  });
  1351→
  1352→  // Build recommendations from matched procedures
  1353→  procMatches.forEach((match, procId) => {
  1354→    const proc = PROCEDURE_DATABASE.find(p => p.ref_id === procId);
  1355→    if (!proc) return;
  1356→
  1357→    // Get affected ratios from the actual data (including directly targeted metrics)
  1358→    const affectedRatios = allRatios.filter(r =>
  1359→      proc.targetMetrics.includes(r.id) ||
  1360→      match.ratioIds.includes(r.id) ||
  1361→      match.flaws.some(f => f.toLowerCase().includes(r.category.toLowerCase()))
  1362→    );
  1363→
  1364→    // Calculate expected improvement with the new function
  1365→    const avgFlawImpact = match.totalFlawImpact / match.flaws.length;
  1366→    const improvement = calculateExpectedImprovement(proc, affectedRatios, avgFlawImpact);
  1367→
  1368→    // Calculate priority score based on multiple factors
  1369→    const priorityScore =
  1370→      (match.maxScore * 0.4) +  // Relevance weight
  1371→      (match.flaws.length * 8) +  // Multi-flaw bonus
  1372→      (avgFlawImpact * 3) +  // Severity weight
  1373→      (proc.baseImpact * 25);  // Procedure effectiveness
  1374→
  1375→    // Adjust impact based on how well this procedure addresses the specific issues
  1376→    const adjustedImpact = Math.min(
  1377→      1.0,
  1378→      proc.baseImpact * (0.85 + (match.maxScore / 400)) * (1 + (match.flaws.length * 0.05))
  1379→    );
  1380→
  1381→    // Calculate direction for each affected ratio
  1382→    const ratiosImpacted = affectedRatios.map(r => {
  1383→      // Determine direction based on deviation
  1384→      let direction: 'increase' | 'decrease' | 'both' = 'both';
  1385→      if (r.value < r.idealMin) {
  1386→        direction = 'increase';
  1387→      } else if (r.value > r.idealMax) {
  1388→        direction = 'decrease';
  1389→      }
  1390→
  1391→      // Calculate percentage effect based on how far from ideal and procedure strength
  1392→      const deviation = r.value < r.idealMin
  1393→        ? r.idealMin - r.value
  1394→        : r.value > r.idealMax
  1395→          ? r.value - r.idealMax
  1396→          : 0;
  1397→      const normalizedDeviation = deviation / ((r.rangeMax - r.rangeMin) / 2);
  1398→      const percentageEffect = proc.baseImpact * Math.min(normalizedDeviation * 1.5, 1.5);
  1399→
  1400→      return {
  1401→        ratioId: r.id,
  1402→        ratioName: r.name,
  1403→        direction,
  1404→        percentageEffect: Math.round(percentageEffect * 100) / 100,
  1405→      };
  1406→    });
  1407→
  1408→    recommendations.push({
  1409→      ref_id: proc.ref_id,
  1410→      name: proc.name,
  1411→      description: proc.description,
  1412→      phase: proc.phase,
  1413→      impact: adjustedImpact,
  1414→      coverage: match.flaws.length + Math.min(match.ratios.length, 5),
  1415→      percentage: proc.percentage,
  1416→      expectedImprovementRange: {
  1417→        min: improvement.min,
  1418→        max: improvement.max,
  1419→      },
  1420→      matchedFlaws: Array.from(new Set(match.flaws)),
  1421→      matchedRatios: Array.from(new Set(match.ratios)),
  1422→      ratios_impacted: ratiosImpacted,
  1423→      timeline: proc.timeline,
  1424→      cost: proc.cost,
  1425→      risks_side_effects: proc.risks_side_effects,
  1426→      warnings: proc.warnings,
  1427→      gender: proc.gender,
  1428→      // Add priority score as a sorting key (stored in coverage field calculation)
  1429→      _priorityScore: priorityScore,
  1430→    } as Recommendation & { _priorityScore: number });
  1431→  });
  1432→
  1433→  // Sort by priority score (combines impact, coverage, and severity)
  1434→  return recommendations.sort((a, b) => {
  1435→    const aScore = (a as Recommendation & { _priorityScore?: number })._priorityScore || (a.impact * 100);
  1436→    const bScore = (b as Recommendation & { _priorityScore?: number })._priorityScore || (b.impact * 100);
  1437→    const scoreDiff = bScore - aScore;
  1438→    if (Math.abs(scoreDiff) > 5) return scoreDiff;
  1439→    // Tiebreaker: prefer procedures that address more flaws
  1440→    return b.matchedFlaws.length - a.matchedFlaws.length;
  1441→  });
  1442→}
