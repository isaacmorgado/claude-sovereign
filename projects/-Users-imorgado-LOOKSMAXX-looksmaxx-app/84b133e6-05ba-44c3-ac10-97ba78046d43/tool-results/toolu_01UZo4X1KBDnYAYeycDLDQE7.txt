   683→// ADVICE ENGINE CLASS
   684→// ============================================
   685→
   686→export class AdviceEngine {
   687→  private plans: RawPlan[];
   688→
   689→  constructor(customPlans?: RawPlan[]) {
   690→    this.plans = customPlans || PLANS;
   691→  }
   692→
   693→  /**
   694→   * Check if a single threshold condition is met
   695→   */
   696→  private checkThreshold(userValue: number, operator: '<' | '>', thresholdValue: number): boolean {
   697→    if (operator === '<') {
   698→      return userValue < thresholdValue;
   699→    }
   700→    return userValue > thresholdValue;
   701→  }
   702→
   703→  /**
   704→   * Evaluate trigger rules against user metrics
   705→   */
   706→  private evaluateTriggerRules(
   707→    triggerRules: TriggerRules,
   708→    metricsDict: Record<string, number>
   709→  ): { isTriggered: boolean; triggeredMetrics: TriggerResult[] } {
   710→    const { metrics, thresholds, condition } = triggerRules;
   711→    const triggeredMetrics: TriggerResult[] = [];
   712→    const results: boolean[] = [];
   713→
   714→    for (const metricName of metrics) {
   715→      if (!(metricName in metricsDict)) continue;
   716→      if (!(metricName in thresholds)) continue;
   717→
   718→      const userValue = metricsDict[metricName];
   719→      const { operator, value: thresholdValue } = thresholds[metricName];
   720→      const isMet = this.checkThreshold(userValue, operator, thresholdValue);
   721→
   722→      if (isMet) {
   723→        triggeredMetrics.push({
   724→          metric: metricName,
   725→          value: userValue,
   726→          threshold: thresholdValue,
   727→          operator
   728→        });
   729→      }
   730→
   731→      results.push(isMet);
   732→    }
   733→
   734→    const isTriggered = condition === 'AND'
   735→      ? results.length > 0 && results.every(r => r)
   736→      : results.some(r => r);
   737→
   738→    return { isTriggered, triggeredMetrics };
   739→  }
   740→
   741→  /**
   742→   * Get all recommendations that match the user's metrics
   743→   *
   744→   * @param metricsDict - Raw metric values
   745→   * @param severityDict - Severity classifications (ideal/good/moderate/severe) for each metric
   746→   *                       If a metric is 'ideal', it will NOT trigger remediation plans
   747→   */
   748→  getRecommendations(
   749→    metricsDict: Record<string, number>,
   750→    severityDict?: Record<string, string>
   751→  ): Plan[] {
   752→    const recommendations: Plan[] = [];
   753→
   754→    for (const rawPlan of this.plans) {
   755→      const { isTriggered, triggeredMetrics } = this.evaluateTriggerRules(
   756→        rawPlan.trigger_rules,
   757→        metricsDict
   758→      );
   759→
   760→      // If severity data is provided, filter out metrics that are already ideal
   761→      if (isTriggered && severityDict) {
   762→        // Check if ALL triggered metrics are outside ideal range
   763→        const hasActualFlaw = triggeredMetrics.some(tm => {
   764→          const severity = severityDict[tm.metric];
   765→          // Only trigger if severity is NOT ideal (i.e., it's good/moderate/severe)
   766→          return severity && severity !== 'ideal';
   767→        });
   768→
   769→        // Skip this plan if all triggered metrics are already ideal
   770→        if (!hasActualFlaw) {
   771→          continue;
   772→        }
   773→      }
   774→
   775→      if (isTriggered) {
   776→        recommendations.push({
   777→          id: rawPlan.id,
   778→          title: rawPlan.title,
   779→          content: {
   780→            description: rawPlan.content.description,
   781→            cost_min: rawPlan.content.cost_min,
   782→            cost_max: rawPlan.content.cost_max,
   783→            time_min: rawPlan.content.time_min || 'Unknown',
   784→            time_max: rawPlan.content.time_max || 'Unknown',
   785→            risks: rawPlan.content.risks || 'Consult a professional.',
   786→            citations: rawPlan.content.citations || [],
   787→            tags: rawPlan.content.tags
   788→          },
   789→          trigger_reason: triggeredMetrics
   790→        });
   791→      }
   792→    }
   793→
   794→    return recommendations;
   795→  }
   796→
   797→  /**
   798→   * Get recommendations sorted by cost (cheapest first)
   799→   */
   800→  getRecommendationsByCost(
   801→    metricsDict: Record<string, number>,
   802→    severityDict?: Record<string, string>
   803→  ): Plan[] {
   804→    return this.getRecommendations(metricsDict, severityDict).sort(
   805→      (a, b) => a.content.cost_min - b.content.cost_min
   806→    );
   807→  }
   808→
   809→  /**
   810→   * Get recommendations grouped by phase/tag
   811→   */
   812→  getRecommendationsByPhase(
   813→    metricsDict: Record<string, number>,
   814→    severityDict?: Record<string, string>
   815→  ): Record<string, Plan[]> {
   816→    const recommendations = this.getRecommendations(metricsDict, severityDict);
   817→    const grouped: Record<string, Plan[]> = {
   818→      'Foundational': [],
   819→      'Minimally Invasive': [],
   820→      'Surgical': []
   821→    };
   822→
   823→    for (const rec of recommendations) {
   824→      const tags = rec.content.tags || [];
   825→      if (tags.includes('Foundational')) {
   826→        grouped['Foundational'].push(rec);
   827→      } else if (tags.includes('Surgical')) {
   828→        grouped['Surgical'].push(rec);
   829→      } else if (tags.includes('Minimally Invasive')) {
   830→        grouped['Minimally Invasive'].push(rec);
   831→      } else {
   832→        grouped['Minimally Invasive'].push(rec); // Default
   833→      }
   834→    }
   835→
   836→    return grouped;
   837→  }
   838→}
