   436â†’async function runDetection() {
   437â†’    if (isOperationInProgress) return;
   438â†’
   439â†’    // Check sequence
   440â†’    const hasSequence = await checkSequenceOpen();
   441â†’    if (!hasSequence) {
   442â†’        setStatus('Please open a sequence first', true);
   443â†’        return;
   444â†’    }
   445â†’
   446â†’    // Check online
   447â†’    if (!isOnline()) {
   448â†’        setStatus('Offline - Check your connection', true);
   449â†’        return;
   450â†’    }
   451â†’
   452â†’    isOperationInProgress = true;
   453â†’    ui.goBtn.disabled = true;
   454â†’    showProgress('Detecting silences...');
   455â†’
   456â†’    try {
   457â†’        const settings = getSettings();
   458â†’
   459â†’        // Get sequence info
   460â†’        const seqInfo = await jsx.call('getActiveSequence');
   461â†’        console.log('[SPLICE] Sequence info:', seqInfo);
   462â†’
   463â†’        // Export audio via JSX for silence detection
   464â†’        updateProgress(10, 'Exporting audio...');
   465â†’
   466â†’        let audioFilePath = null;
   467â†’        try {
   468â†’            // Get sequence info for audio export
   469â†’            const exportResult = await jsx.call('exportSequenceAudioForAnalysis');
   470â†’            if (exportResult && exportResult.success && exportResult.outputPath) {
   471â†’                audioFilePath = exportResult.outputPath;
   472â†’                console.log('[SPLICE] Audio exported to:', audioFilePath);
   473â†’            } else if (exportResult && exportResult.error) {
   474â†’                throw new Error(exportResult.error);
   475â†’            }
   476â†’        } catch (exportError) {
   477â†’            console.warn('[SPLICE] Audio export failed, trying fallback:', exportError.message);
   478â†’            // Try to get audio from first clip in timeline
   479â†’            try {
   480â†’                const clipInfo = await jsx.call('getFirstClipAudioPath');
   481â†’                if (clipInfo && clipInfo.path) {
   482â†’                    audioFilePath = clipInfo.path;
   483â†’                    console.log('[SPLICE] Using clip audio path:', audioFilePath);
   484â†’                }
   485â†’            } catch (clipError) {
   486â†’                console.warn('[SPLICE] Fallback audio path failed:', clipError.message);
   487â†’            }
   488â†’        }
   489â†’
   490â†’        if (!audioFilePath) {
   491â†’            throw new Error('Could not export or locate audio file. Please ensure sequence has audio tracks.');
   492â†’        }
   493â†’
   494â†’        // Call backend for silence detection with audio file path
   495â†’        updateProgress(30, 'Detecting silences...');
   496â†’        const silenceResponse = await fetchWithTimeout(
   497â†’            `${getBackendUrl()}/silences-rms`,
   498â†’            {
   499â†’                method: 'POST',
   500â†’                headers: getAuthHeaders(),
   501â†’                body: JSON.stringify({
   502â†’                    wavPath: audioFilePath,
   503â†’                    sensitivity: settings.sensitivity,
   504â†’                    minSilenceLength: settings.minSilenceLength || 0.5
   505â†’                })
   506â†’            }
   507â†’        );
   508â†’
   509â†’        if (!silenceResponse.ok) {
   510â†’            throw new Error(await parseErrorResponse(silenceResponse));
   511â†’        }
   512â†’
   513â†’        const silenceData = await silenceResponse.json();
   514â†’        previewSilences = silenceData.silences || [];
   515â†’
   516â†’        // Detect takes if enabled
   517â†’        if (settings.enableTakesDetection) {
   518â†’            updateProgress(60, 'Detecting takes...');
   519â†’            const analyzeResponse = await fetchWithTimeout(
   520â†’                `${getBackendUrl()}/analyze`,
   521â†’                {
   522â†’                    method: 'POST',
   523â†’                    headers: getAuthHeaders(),
   524â†’                    body: JSON.stringify({
   525â†’                        detectTakes: true
   526â†’                    })
   527â†’                }
   528â†’            );
   529â†’
   530â†’            if (analyzeResponse.ok) {
   531â†’                const analyzeData = await analyzeResponse.json();
   532â†’                previewTakes = analyzeData.takes || [];
   533â†’            }
   534â†’        }
   535â†’
   536â†’        // Detect chapters if enabled
   537â†’        if (settings.enableChapters) {
   538â†’            updateProgress(80, 'Detecting chapters...');
   539â†’            try {
   540â†’                const chapterResponse = await fetchWithTimeout(
   541â†’                    `${getBackendUrl()}/chapters`,
   542â†’                    {
   543â†’                        method: 'POST',
