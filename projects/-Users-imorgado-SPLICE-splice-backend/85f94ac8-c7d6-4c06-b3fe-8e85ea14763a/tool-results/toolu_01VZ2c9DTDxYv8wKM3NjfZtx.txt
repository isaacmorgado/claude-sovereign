   169→function scheduleUIUpdate(callback) {
   170→  if (pendingUIUpdate) {
   171→    cancelAnimationFrame(pendingUIUpdate);
   172→  }
   173→  pendingUIUpdate = requestAnimationFrame(() => {
   174→    callback();
   175→    pendingUIUpdate = null;
   176→  });
   177→}

   192→function initKeyboardShortcuts() {
   193→  document.addEventListener('keydown', (e) => {
   194→    // Only handle when combined preview is visible
   195→    const previewVisible = ui.combinedPreview?.style.display !== 'none' &&
   196→                           ui.combinedPreview?.style.display !== '';
   197→
   198→    if (!previewVisible) return;
   199→
   200→    const isMac = navigator.platform.toUpperCase().indexOf('MAC') >= 0;
   201→    const cmdKey = isMac ? e.metaKey : e.ctrlKey;
   202→
   203→    // Enter - Apply selected
   204→    if (e.key === 'Enter' && !e.shiftKey) {
   205→      e.preventDefault();
   206→      if (selectedSilenceIndices.size > 0) {
   207→        ui.applyPreviewBtn?.click();
   208→      }
   209→    }
   210→
   211→    // Escape - Cancel
   212→    if (e.key === 'Escape') {
   213→      e.preventDefault();
   214→      ui.cancelPreviewBtn?.click();
   215→    }
   216→
   217→    // Cmd/Ctrl+A - Select all
   218→    if (e.key === 'a' && cmdKey) {
   219→      e.preventDefault();
   220→      if (ui.selectAllSilences) {
   221→        ui.selectAllSilences.checked = true;
   222→        selectAllSilences(true);
   223→      }
   224→    }
   225→
   226→    // Cmd/Ctrl+D - Deselect all
   227→    if (e.key === 'd' && cmdKey) {
   228→      e.preventDefault();
   229→      if (ui.selectAllSilences) {
   230→        ui.selectAllSilences.checked = false;
   231→        selectAllSilences(false);
   232→      }
   233→    }
   234→
   235→    // Cmd/Ctrl+I - Invert selection
   236→    if (e.key === 'i' && cmdKey) {
   237→      e.preventDefault();
   238→      invertSilenceSelection();
   239→    }
   240→  });

   243→// ============================================================================
   244→// UNIFIED WORKFLOW (Silence Detection + Takes Detection)
   245→// ============================================================================
   246→function initUnifiedWorkflow() {
   247→  if (!ui.goBtn) return;
   248→
   249→  ui.goBtn.addEventListener('click', async () => {
   250→    // Prevent double-click
   251→    if (isOperationInProgress) {
   252→      setStatus('Operation in progress...');
   253→      return;
   254→    }
   255→
   256→    isOperationInProgress = true;
   257→    ui.goBtn.disabled = true;
   258→
   259→    try {
   260→      showProgress('Preparing audio...');
   261→
   262→      // Step 1: Export audio
   263→      const exportSuccess = await exportAudioInternal();
   264→      if (!exportSuccess) {
   265→        throw new Error('Audio export failed');
   266→      }
   267→
   268→      // Step 2: Get settings
   269→      const sensitivity = parseInt(ui.sensitivitySlider?.value || 50);
   270→      const useIsolated = ui.sourceIsolated?.checked ?? false;
   271→      const detectTakes = ui.enableTakesDetection?.checked ?? true;
   272→
   273→      // Step 3: Vocal isolation if needed
   274→      let audioPath = WAV_PATH;
   275→      if (useIsolated) {
   276→        // Check isolation access before proceeding
   277→        const credits = getCurrentCredits();
   278→        if (!credits || !credits.hasIsolationAccess) {
   279→          throw new Error('Vocal isolation requires Pro or Team tier. Upgrade to access this feature.');
   280→        }
   281→
   282→        // Show overage warning if applicable
   283→        const isolationMins = credits.isolationHoursRemaining * 60;
   284→        if (isolationMins <= 0) {
   285→          const overageRate = credits.isolationOverageRate || 0.08;
   286→          showProgress(`Isolating vocals (overage: $${overageRate}/min)...`);
   287→        } else {
   288→          showProgress(`Isolating vocals (${isolationMins.toFixed(0)} min remaining)...`);
   289→        }
   290→
   291→        const isolateResult = await isolateVocals();
   292→        if (isolateResult.success) {
   293→          audioPath = isolateResult.outputPath;
   294→          // Update credits display after isolation
   295→          await refreshCredits();
   296→        } else {
   297→          throw new Error(isolateResult.error || 'Vocal isolation failed');
   298→        }
   299→      }
   300→
   301→      // Step 4: Run detection (parallel if takes enabled)
   302→      const params = mapSensitivity(sensitivity);
   303→      let silences = [];
   304→      let takes = [];
   305→
   306→      if (detectTakes) {
   307→        // Parallel detection for performance
   308→        showProgress('Detecting silences and analyzing takes...');
   309→        const [silenceResult, transcriptResult] = await Promise.all([
   310→          detectSilences(audioPath, params),
   311→          transcribeAudio()
   312→        ]);
   313→
   314→        silences = silenceResult;
   315→
   316→        // Extract takes from transcript result
   317→        if (transcriptResult.success && transcriptResult.takes) {
   318→          if (transcriptResult.takes.takes) {
   319→            takes = transcriptResult.takes.takes;
   320→          } else if (Array.isArray(transcriptResult.takes)) {
   321→            takes = transcriptResult.takes;
   322→          }
   323→        }
   324→
   325→        // Store transcript for reference
   326→        window.currentTranscript = transcriptResult.transcript;
   327→      } else {
   328→        // Just detect silences
   329→        showProgress('Detecting silences...');
   330→        silences = await detectSilences(audioPath, params);
   331→      }
   332→
   333→      if (silences.length === 0) {
   334→        showEmptyState();
   335→        setStatus('No silences detected. Try increasing sensitivity.');
   336→        return;
   337→      }
   338→
   339→      // Step 5: Filter silences by takes (protect speech)
   340→      const filtered = filterSilencesByTakes(silences, takes);
   341→
   342→      // Step 6: Add timeline markers (clear old ones first)
   343→      showProgress('Adding timeline markers...');
   344→      await clearSpliceMarkers();
   345→      const markerResult = await addTimelineMarkers(filtered.safe, filtered.protected, takes);
   346→
   347→      // Step 7: Show combined preview
   348→      showCombinedPreview(filtered.safe, filtered.protected, takes);
   349→
   350→      const markerInfo = markerResult.added > 0 ? ` (${markerResult.added} markers added)` : '';
   351→      const statusMsg = takes.length > 0
   352→        ? `Found ${silences.length} silences, ${filtered.protected.length} protected by ${takes.length} takes${markerInfo}`
   353→        : `Found ${silences.length} silence(s) - review and apply${markerInfo}`;
   354→      setStatus(statusMsg);
   355→
   356→    } catch (err) {
   357→      showEmptyState();
   358→      setStatus('Error: ' + err.message);
   359→      console.error('[SPLICE] Unified workflow error:', err);
   360→    } finally {
   361→      isOperationInProgress = false;
   362→      ui.goBtn.disabled = false;
   363→    }
   364→  });
   365→}

   367→// ============================================================================
   368→// TAKE-AWARE SILENCE FILTERING
   369→// ============================================================================
   370→/**
   371→ * Filter silences to protect those that overlap with takes (speech segments).
   372→ * Returns { safe: [], protected: [] }
   373→ */
   374→function filterSilencesByTakes(silences, takes) {
   375→  if (!takes || takes.length === 0) {
   376→    // No takes = all silences are safe to remove
   377→    return { safe: silences, protected: [] };
   378→  }
   379→
   380→  const safe = [];
   381→  const protected_ = [];
   382→
   383→  silences.forEach(silence => {
   384→    // Check if this silence overlaps with any take
   385→    const overlaps = takes.some(take => {
   386→      const takeStart = take.startTime || take.start;
   387→      const takeEnd = take.endTime || take.end;
   388→      // Overlap exists if silence starts before take ends AND silence ends after take starts
   389→      return silence.start < takeEnd && silence.end > takeStart;
   390→    });
   391→
   392→    if (overlaps) {
   393→      protected_.push(silence);
   394→    } else {
   395→      safe.push(silence);
   396→    }
   397→  });
   398→
   399→  console.log(`[SPLICE] Filtered silences: ${safe.length} safe, ${protected_.length} protected`);
   400→  return { safe, protected: protected_ };
   401→}

   403→// ============================================================================
   404→// TIMELINE MARKERS (Visual indicators on Premiere Pro timeline)
   405→// ============================================================================
   406→/**
   407→ * Add colored markers to the timeline for silences and takes.
   408→ * - Red markers: silences safe to remove
   409→ * - Yellow markers: silences protected by takes
   410→ * - Blue markers: takes (speech segments)
   411→ *
   412→ * @param {Array} safeToRemove - Silences safe to remove
   413→ * @param {Array} protectedSilences - Silences overlapping speech
   414→ * @param {Array} takes - Detected speech segments
   415→ * @returns {Promise<{added: number, errors: number}>}
   416→ */
   417→async function addTimelineMarkers(safeToRemove, protectedSilences, takes) {
   418→  const results = { added: 0, errors: 0 };
   419→
   420→  try {
   421→    const context = await getActiveSequence();
   422→    if (!context) {
   423→      console.warn('[SPLICE] No sequence available for markers');
   424→      return results;
   425→    }
   426→
   427→    const { sequence, project } = context;
   428→
   429→    // Get markers object for the sequence
   430→    const markers = await ppro.Markers.getMarkers(sequence);
   431→    if (!markers) {
   432→      console.warn('[SPLICE] Markers API not available');
   433→      return results;
   434→    }
   435→
   436→    // Collect all marker actions
   437→    const markerActions = [];
   438→
   439→    // Red markers for safe-to-remove silences
   440→    for (const silence of safeToRemove) {
   441→      try {
   442→        const duration = silence.end - silence.start;
   443→
   444→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   445→        const startTime = ppro.TickTime.createWithSeconds(silence.start);
   446→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   447→
   448→        const action = markers.createAddMarkerAction(
   449→          `SPLICE: Remove (${duration.toFixed(2)}s)`,
   450→          ppro.Marker.MARKER_TYPE_COMMENT,
   451→          startTime,
   452→          durationTime,
   453→          `Silence to remove: ${formatTime(silence.start)} - ${formatTime(silence.end)}`
   454→        );
   455→        markerActions.push({ action, colorIndex: 1, type: 'silence' }); // Red = index 1
   456→      } catch (err) {
   457→        console.error('[SPLICE] Error creating silence marker:', err);
   458→        results.errors++;
   459→      }
   460→    }
   461→
   462→    // Yellow markers for protected silences
   463→    for (const silence of protectedSilences) {
   464→      try {
   465→        const duration = silence.end - silence.start;
   466→
   467→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   468→        const startTime = ppro.TickTime.createWithSeconds(silence.start);
   469→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   470→
   471→        const action = markers.createAddMarkerAction(
   472→          `SPLICE: Protected (${duration.toFixed(2)}s)`,
   473→          ppro.Marker.MARKER_TYPE_COMMENT,
   474→          startTime,
   475→          durationTime,
   476→          `Protected silence (overlaps speech): ${formatTime(silence.start)} - ${formatTime(silence.end)}`
   477→        );
   478→        markerActions.push({ action, colorIndex: 4, type: 'protected' }); // Yellow = index 4
   479→      } catch (err) {
   480→        console.error('[SPLICE] Error creating protected marker:', err);
   481→        results.errors++;
   482→      }
   483→    }
   484→
   485→    // Blue markers for takes (speech segments)
   486→    for (const take of takes) {
   487→      try {
   488→        const start = take.startTime || take.start;
   489→        const end = take.endTime || take.end;
   490→        const duration = end - start;
   491→        const textPreview = take.text?.substring(0, 30) || 'Speech';
   492→
   493→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   494→        const startTime = ppro.TickTime.createWithSeconds(start);
   495→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   496→
   497→        const action = markers.createAddMarkerAction(
   498→          `SPLICE: Take`,
   499→          ppro.Marker.MARKER_TYPE_COMMENT,
   500→          startTime,
   501→          durationTime,
   502→          `Speech segment: ${textPreview}${take.text?.length > 30 ? '...' : ''}`
   503→        );
   504→        markerActions.push({ action, colorIndex: 5, type: 'take' }); // Blue = index 5
   505→      } catch (err) {
   506→        console.error('[SPLICE] Error creating take marker:', err);
   507→        results.errors++;
   508→      }
   509→    }
   510→
   511→    // Execute all marker actions in a single transaction
   512→    if (markerActions.length > 0) {
   513→      await project.lockedAccess(async () => {
   514→        await project.executeTransaction((compoundAction) => {
   515→          for (const { action } of markerActions) {
   516→            compoundAction.addAction(action);
   517→          }
   518→        }, 'SPLICE: Add Timeline Markers');
   519→      });
   520→
   521→      results.added = markerActions.length;
   522→      console.log(`[SPLICE] Added ${results.added} timeline markers`);
   523→    }
   524→
   525→  } catch (err) {
   526→    console.error('[SPLICE] Timeline markers error:', err);
   527→    results.errors++;
   528→  }
   529→
   530→  return results;
   531→}

   533→/**
   534→ * Read current SPLICE marker positions from timeline.
   535→ * This allows users to manually adjust markers before applying.
   536→ * Returns silences based on current marker positions (not original detection).
   537→ */
   538→async function readMarkersAssilences() {
   539→  const silences = [];
   540→
   541→  try {
   542→    const context = await getActiveSequence();
   543→    if (!context) return silences;
   544→
   545→    const { sequence } = context;
   546→    const markers = await ppro.Markers.getMarkers(sequence);
   547→    if (!markers) return silences;
   548→
   549→    const allMarkers = await markers.getMarkers();
   550→    if (!allMarkers || allMarkers.length === 0) return silences;
   551→
   552→    for (const marker of allMarkers) {
   553→      const name = await marker.getName();
   554→
   555→      // Only read "Remove" markers (red ones that will be cut)
   556→      if (name && name.startsWith('SPLICE: Remove')) {
   557→        const startTickTime = await marker.getStart();
   558→        const durationTickTime = await marker.getDuration();
   559→
   560→        // Use TickTime.seconds property to get seconds directly (no TICKS_PER_SECOND needed)
   561→        const start = startTickTime.seconds;
   562→        const duration = durationTickTime.seconds;
   563→        const end = start + duration;
   564→
   565→        silences.push({
   566→          start,
   567→          end,
   568→          duration,
   569→          fromMarker: true  // Flag to indicate this came from marker adjustment
   570→        });
   571→      }
   572→    }
   573→
   574→    // Sort by start time
   575→    silences.sort((a, b) => a.start - b.start);
   576→    console.log(`[SPLICE] Read ${silences.length} silences from adjusted markers`);
   577→
   578→  } catch (err) {
   579→    console.error('[SPLICE] Error reading markers:', err);
   580→  }
   581→
   582→  return silences;
   583→}

   585→/**
   586→ * Remove all SPLICE markers from the timeline.
   587→ * Call this before adding new markers to avoid duplicates.
   588→ */
   589→async function clearSpliceMarkers() {
   590→  try {
   591→    const context = await getActiveSequence();
   592→    if (!context) return 0;
   593→
   594→    const { sequence, project } = context;
   595→    const markers = await ppro.Markers.getMarkers(sequence);
   596→    if (!markers) return 0;
   597→
   598→    // Get all markers
   599→    const allMarkers = await markers.getMarkers();
   600→    if (!allMarkers || allMarkers.length === 0) return 0;
   601→
   602→    // Filter SPLICE markers
   603→    const spliceMarkers = [];
   604→    for (const marker of allMarkers) {
   605→      const name = await marker.getName();
   606→      if (name && name.startsWith('SPLICE:')) {
   607→        spliceMarkers.push(marker);
   608→      }
   609→    }
   610→
   611→    if (spliceMarkers.length === 0) return 0;
   612→
   613→    // Remove all SPLICE markers in a single transaction
   614→    await project.lockedAccess(async () => {

  1199→ */
  1200→async function findFirstMediaItem(item) {
  1201→  if (!item) return null;
  1202→
  1203→  // Fix: Use local Constant instead of app.Constants.ProjectItemType
  1204→  if (item.type === ProjectItemType.CLIP) {
  1205→    return item;
  1206→  }
  1207→
  1208→  // Search Children (Bin or Root)
  1209→  // Note: item.children is a collection, check numItems
  1210→  if (item.children && item.children.numItems > 0) {
  1211→    for (let i = 0; i < item.children.numItems; i++) {
  1212→      const child = item.children[i];
  1213→      const found = await findFirstMediaItem(child);
  1214→      if (found) return found;
  1215→    }
  1216→  }
  1217→
  1218→  return null;
  1219→}

  1639→/**
  1640→ * Poll batch job status
  1641→ */
  1642→async function pollBatchStatus(jobId, progressBar, progressText) {
  1643→  const pollInterval = 2000; // 2 seconds
  1644→  const maxPolls = 300; // 10 minutes max
  1645→  let polls = 0;
  1646→
  1647→  while (polls < maxPolls) {
  1648→    try {
  1649→      const response = await fetch(`${BACKEND_URL}/batch/status/${jobId}`, {
  1650→        headers: getAuthHeaders()
  1651→      });
  1652→
  1653→      if (!response.ok) {
  1654→        throw new Error('Failed to get batch status');
  1655→      }
  1656→
  1657→      const data = await response.json();
  1658→      const job = data.job;
  1659→
  1660→      // Update progress
  1661→      if (progressBar) progressBar.value = job.progress.percentage;
  1662→      if (progressText) progressText.textContent = `${job.progress.completed + job.progress.failed} / ${job.progress.total} files`;
  1663→
  1664→      // Check if complete
  1665→      if (job.status !== 'processing') {
  1666→        if (job.status === 'completed') {
  1667→          setStatus(`Batch complete: ${job.progress.completed} files processed`);
  1668→        } else if (job.status === 'completed_with_errors') {
  1669→          setStatus(`Batch complete with ${job.progress.failed} error(s)`);
  1670→        } else {
  1671→          setStatus('Batch failed');
  1672→        }
  1673→
  1674→        // Clear queue on completion
  1675→        batchQueue = [];
  1676→        updateBatchQueueDisplay();
  1677→        return;
  1678→      }
  1679→
  1680→      await new Promise(resolve => setTimeout(resolve, pollInterval));
  1681→      polls++;
  1682→    } catch (err) {
  1683→      setStatus('Polling error: ' + err.message);
  1684→      return;
  1685→    }
  1686→  }
  1687→
  1688→  setStatus('Batch timed out');
  1689→}

  1693→  window.removeBatchItem = removeBatchItem;
