   164→   164→        customerId: license.customerId,

   359→   359→  // Simulate retry mechanism

   362→   362→    let retryCount = 0;
   363→   363→
   364→   364→    while (retryCount < maxRetries) {
   365→   365→      attempts++;
   366→   366→      // Simulate 2 failures then success

   370→   370→        continue;
   371→   371→      }

   373→   373→      break;
   374→   374→    }
   375→   375→
   376→   376→    return { result, attempts: retryCount };
   377→   377→  }

   380→   380→
