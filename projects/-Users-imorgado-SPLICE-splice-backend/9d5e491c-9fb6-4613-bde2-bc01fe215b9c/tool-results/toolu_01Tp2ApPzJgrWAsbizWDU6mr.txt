    34→const DEFAULT_SETTINGS = {
    35→  sensitivity: 50,
    36→  audioSource: 'original',
    37→  autoMarkBest: true,
    38→  enableTakesDetection: true,
    39→  rememberOptions: false,
    40→  optionsExpanded: false,

    46→
    47→let currentSettings = { ...DEFAULT_SETTINGS };
    48→
    49→function loadSettings() {
    50→  try {
    51→    const saved = localStorage.getItem('spliceSettings');
    52→    if (saved) {
    53→      currentSettings = { ...DEFAULT_SETTINGS, ...JSON.parse(saved) };
    54→    }
    55→  } catch (e) {
    56→    currentSettings = { ...DEFAULT_SETTINGS };
    57→  }
    58→  return currentSettings;
    59→}
    60→
    61→function saveSettings(settings) {
    62→  try {
    63→    currentSettings = { ...currentSettings, ...settings };
    64→    localStorage.setItem('spliceSettings', JSON.stringify(currentSettings));
    65→  } catch (e) {
    66→    // Silent fail
    67→  }
    68→}
    69→
    70→function getSettings() {
    71→  return { ...currentSettings };
    72→}

    74→function isLoggedIn() {
    75→  const settings = getSettings();
    76→  return !!settings.customerId;
    77→}

    81→  return settings.customerId || null;

   105→function formatBuildError(error) {
   106→  if (!error) return 'Unknown error occurred';
   107→
   108→  const errorMap = {
   109→    'Builder not initialized': 'Builder module failed to load. Try restarting the plugin.',
   110→    'No active project': 'No project is open. Please open or create a project first.',
   111→    'No media items found': 'No media files in project. Import media before building.',
   112→    'No detection data': 'Run analysis first (click GO button).',
   113→    'Failed to generate cut list': 'Server error generating cuts. Check your connection.',
   114→    'Failed to build sequence': 'Premiere Pro sequence build failed. Try again.',
   115→    'fetch failed': 'Network error. Check if backend server is running.',
   116→    'NetworkError': 'Cannot connect to server. Check your internet connection.',
   117→    'Insufficient credits': 'Out of credits. Upgrade your plan to continue.',
   118→    'Authentication required': 'Login required. Click the credits badge to log in.'
   119→  };
   120→
   121→  for (const [pattern, friendly] of Object.entries(errorMap)) {
   122→    if (error.includes(pattern)) {
   123→      return friendly;
   124→    }
   125→  }
   126→
   127→  return error;
   128→}

   149→function assertEqual(actual, expected, message = '') {
   150→  if (actual !== expected) {
   151→    throw new Error(`${message}: Expected ${JSON.stringify(expected)}, got ${JSON.stringify(actual)}`);
   152→  }
   153→}

   155→function assertTrue(condition, message = '') {
   156→  if (!condition) {
   157→    throw new Error(`${message}: Expected true, got false`);
   158→  }
   159→}
   160→
   161→function assertFalse(condition, message = '') {
   162→  if (condition) {
   163→    throw new Error(`${message}: Expected false, got true`);
   164→  }
   165→}
   166→
   167→function assertNotNull(value, message = '') {
   168→  if (value === null || value === undefined) {
   169→    throw new Error(`${message}: Expected non-null value`);
   170→  }
   171→}
   172→
   173→function assertIncludes(arr, item, message = '') {
   174→  if (!arr || !arr.includes) {
   175→    throw new Error(`${message}: Not an array`);
   176→  }
   177→  if (!arr.includes(item)) {
   178→    throw new Error(`${message}: Array does not include ${item}`);
   179→  }
   180→}
