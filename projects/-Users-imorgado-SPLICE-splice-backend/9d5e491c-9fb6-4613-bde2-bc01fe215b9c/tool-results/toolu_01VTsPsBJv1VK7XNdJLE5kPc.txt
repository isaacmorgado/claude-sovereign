    22→    22→/**
    23→    23→ * Helper: Make HTTPS request
    24→    24→ */
    25→    25→function request(method, path, body = null) {
    26→    26→  return new Promise((resolve, reject) => {
    27→    27→    const url = new URL(path, BASE_URL);
    28→    28→    const options = {
    29→    29→      method,
    30→    30→      hostname: url.hostname,
    31→    31→      port: url.port,
    32→    32→      path: url.pathname,
    33→    33→      headers: {
    34→    34→        'Content-Type': 'application/json',
    35→    35→        'x-stripe-customer-id': TEST_CUSTOMER_ID
    36→    36→      },
    37→    37→      agent
    38→    38→    };
    39→    39→
    40→    40→    const req = https.request(options, (res) => {
    41→    41→      let data = '';
    42→    42→      res.on('data', (chunk) => data += chunk);
    43→    43→      res.on('end', () => {
    44→    44→        try {
    45→    45→          resolve({ status: res.statusCode, data: JSON.parse(data) });
    46→    46→        } catch (e) {
    47→    47→          resolve({ status: res.statusCode, data: data });
    48→    48→        }
    49→    49→      });
    50→    50→    });
    51→    51→
    52→    52→    req.on('error', reject);
    53→    53→    if (body) req.write(JSON.stringify(body));
    54→    54→    req.end();
    55→    55→  });
    56→    56→}
