    23→function request(method, path, body = null, headers = {}) {
    24→  return new Promise((resolve, reject) => {
    25→    const url = new URL(path, BASE_URL);
    26→    const options = {
    27→      method,
    28→      hostname: url.hostname,
    29→      port: url.port,
    30→      path: url.pathname,
    31→      headers: {
    32→        'Content-Type': 'application/json',
    33→        ...headers
    34→      },
    35→      agent
    36→    };
    37→
    38→    const req = https.request(options, (res) => {
    39→      let data = '';
    40→      res.on('data', (chunk) => data += chunk);
    41→      res.on('end', () => {
    42→        try {
    43→          resolve({ status: res.statusCode, data: JSON.parse(data) });
    44→        } catch (e) {
    45→          resolve({ status: res.statusCode, data: data });
    46→        }
    47→      });
    48→    });
    49→
    50→    req.on('error', reject);
    51→    if (body) req.write(JSON.stringify(body));
    52→    req.end();
    53→  });
    54→}
