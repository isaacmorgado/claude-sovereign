   113→let confirmModalCallback = null;

   115→/**
   116→ * Show a confirmation modal dialog (UXP-compatible replacement for confirm())
   117→ * @param {string} title - Modal title
   118→ * @param {string} message - Confirmation message
   119→ * @param {Function} onConfirm - Callback when confirmed
   120→ * @param {Function} [onCancel] - Optional callback when cancelled
   121→ * @param {Object} [options] - Optional customization { confirmText, cancelText, confirmStyle }
   122→ */
   123→function showConfirmModal(title, message, onConfirm, onCancel = null, options = {}) {
   124→  if (!ui.confirmModal) return;
   125→
   126→  // Set content
   127→  if (ui.confirmModalTitle) ui.confirmModalTitle.textContent = title;
   128→  if (ui.confirmModalMessage) ui.confirmModalMessage.textContent = message;
   129→
   130→  // Customize button text if provided
   131→  if (ui.confirmModalConfirmBtn) {
   132→    ui.confirmModalConfirmBtn.textContent = options.confirmText || 'Confirm';
   133→    ui.confirmModalConfirmBtn.style.background = options.confirmStyle || '#dc3545';
   134→  }
   135→  if (ui.confirmModalCancelBtn) {
   136→    ui.confirmModalCancelBtn.textContent = options.cancelText || 'Cancel';
   137→  }
   138→
   139→  // Store callbacks
   140→  confirmModalCallback = { onConfirm, onCancel };
   141→
   142→  // Show modal
   143→  ui.confirmModal.classList.remove('hidden');
   144→
   145→  // Focus cancel button for safety (user must explicitly confirm)
   146→  if (ui.confirmModalCancelBtn) ui.confirmModalCancelBtn.focus();
   147→}
   148→
   149→function hideConfirmModal(confirmed = false) {
   150→  if (!ui.confirmModal) return;
   151→
   152→  ui.confirmModal.classList.add('hidden');
   153→
   154→  // Execute callback
   155→  if (confirmModalCallback) {
   156→    if (confirmed && confirmModalCallback.onConfirm) {
   157→      confirmModalCallback.onConfirm();
   158→    } else if (!confirmed && confirmModalCallback.onCancel) {
   159→      confirmModalCallback.onCancel();
   160→    }
   161→    confirmModalCallback = null;
   162→  }
   163→}

   284→// ============================================================================
   285→// DEBOUNCED UI UPDATES (Performance: batch DOM updates)
   286→// ============================================================================
   287→function scheduleUIUpdate(callback) {
   288→  if (pendingUIUpdate) {
   289→    cancelAnimationFrame(pendingUIUpdate);
   290→  }
   291→  pendingUIUpdate = requestAnimationFrame(() => {
   292→    callback();
   293→    pendingUIUpdate = null;
   294→  });
   295→}

   310→function initKeyboardShortcuts() {
   311→  document.addEventListener('keydown', (e) => {
   312→    // Only handle when combined preview is visible
   313→    const previewVisible = ui.combinedPreview?.style.display !== 'none' &&
   314→                           ui.combinedPreview?.style.display !== '';
   315→
   316→    if (!previewVisible) return;
   317→
   318→    const isMac = navigator.platform.toUpperCase().indexOf('MAC') >= 0;
   319→    const cmdKey = isMac ? e.metaKey : e.ctrlKey;
   320→
   321→    // Enter - Apply selected
   322→    if (e.key === 'Enter' && !e.shiftKey) {
   323→      e.preventDefault();
   324→      if (selectedSilenceIndices.size > 0) {
   325→        ui.applyPreviewBtn?.click();
   326→      }
   327→    }
   328→
   329→    // Escape - Cancel
   330→    if (e.key === 'Escape') {
   331→      e.preventDefault();
   332→      ui.cancelPreviewBtn?.click();
   333→    }
   334→
   335→    // Cmd/Ctrl+A - Select all
   336→    if (e.key === 'a' && cmdKey) {
   337→      e.preventDefault();
   338→      if (ui.selectAllSilences) {
   339→        ui.selectAllSilences.checked = true;
   340→        selectAllSilences(true);
   341→      }
   342→    }
   343→
   344→    // Cmd/Ctrl+D - Deselect all
   345→    if (e.key === 'd' && cmdKey) {
   346→      e.preventDefault();
   347→      if (ui.selectAllSilences) {
   348→        ui.selectAllSilences.checked = false;
   349→        selectAllSilences(false);
   350→      }
   351→    }
   352→
   353→    // Cmd/Ctrl+I - Invert selection
   354→    if (e.key === 'i' && cmdKey) {
   355→      e.preventDefault();
   356→      invertSilenceSelection();
   357→    }
   358→  });
   359→}

   361→// ============================================================================
   362→// UNIFIED WORKFLOW (Silence Detection + Takes Detection)
   363→// ============================================================================
   364→function initUnifiedWorkflow() {
   365→  if (!ui.goBtn) return;
   366→
   367→  ui.goBtn.addEventListener('click', async () => {
   368→    // Prevent double-click
   369→    if (isOperationInProgress) {
   370→      setStatus('Operation in progress...');
   371→      return;
   372→    }
   373→
   374→    isOperationInProgress = true;
   375→    ui.goBtn.disabled = true;
   376→
   377→    try {
   378→      showProgress('Preparing audio...');
   379→
   380→      // Step 1: Export audio
   381→      const exportSuccess = await exportAudioInternal();
   382→      if (!exportSuccess) {
   383→        throw new Error('Audio export failed');
   384→      }
   385→
   386→      // Step 2: Get settings
   387→      const sensitivity = parseInt(ui.sensitivitySlider?.value || 50);
   388→      const useIsolated = ui.sourceIsolated?.checked ?? false;
   389→      const detectTakes = ui.enableTakesDetection?.checked ?? true;
   390→
   391→      // Get initialized paths (cross-platform)
   392→      const paths = await getPaths();
   393→
   394→      // Step 3: Vocal isolation if needed
   395→      let audioPath = paths.wavPath;
   396→      if (useIsolated) {
   397→        // Check isolation access before proceeding
   398→        const credits = getCurrentCredits();
   399→        if (!credits || !credits.hasIsolationAccess) {
   400→          throw new Error('Vocal isolation requires Pro or Team tier. Upgrade to access this feature.');
   401→        }
   402→
   403→        // Show overage warning if applicable
   404→        const isolationMins = credits.isolationHoursRemaining * 60;
   405→        if (isolationMins <= 0) {
   406→          const overageRate = credits.isolationOverageRate || 0.08;
   407→          showProgress(`Isolating vocals (overage: $${overageRate}/min)...`);
   408→        } else {
   409→          showProgress(`Isolating vocals (${isolationMins.toFixed(0)} min remaining)...`);
   410→        }
   411→
   412→        const isolateResult = await isolateVocals();
   413→        if (isolateResult.success) {
   414→          audioPath = isolateResult.outputPath;
   415→          // Update credits display after isolation
   416→          await refreshCredits();
   417→        } else {
   418→          throw new Error(isolateResult.error || 'Vocal isolation failed');
   419→        }
   420→      }
   421→
   422→      // Step 4: Run detection (parallel if takes enabled)
   423→      const params = mapSensitivity(sensitivity);
   424→      let silences = [];
   425→      let takes = [];
   426→
   427→      if (detectTakes) {
   428→        // Parallel detection for performance
   429→        showProgress('Detecting silences and analyzing takes...');
   430→        const [silenceResult, transcriptResult] = await Promise.all([
   431→          detectSilences(audioPath, params),
   432→          transcribeAudio()
   433→        ]);
   434→
   435→        silences = silenceResult;
   436→
   437→        // Extract takes from transcript result
   438→        if (transcriptResult.success && transcriptResult.takes) {
   439→          if (transcriptResult.takes.takes) {
   440→            takes = transcriptResult.takes.takes;
   441→          } else if (Array.isArray(transcriptResult.takes)) {
   442→            takes = transcriptResult.takes;
   443→          }
   444→        }
   445→
   446→        // Store transcript for reference
   447→        window.currentTranscript = transcriptResult.transcript;
   448→      } else {
   449→        // Just detect silences
   450→        showProgress('Detecting silences...');
   451→        silences = await detectSilences(audioPath, params);
   452→      }
   453→
   454→      if (silences.length === 0) {
   455→        showEmptyState();
   456→        setStatus('No silences detected. Try increasing sensitivity.');
   457→        return;
   458→      }
   459→
   460→      // Step 5: Filter silences by takes (protect speech)
   461→      const filtered = filterSilencesByTakes(silences, takes);
   462→
   463→      // Step 6: Add timeline markers (clear old ones first)
   464→      showProgress('Adding timeline markers...');
   465→      await clearSpliceMarkers();
   466→      const markerResult = await addTimelineMarkers(filtered.safe, filtered.protected, takes);
   467→
   468→      // Step 7: Show combined preview
   469→      showCombinedPreview(filtered.safe, filtered.protected, takes);
   470→
   471→      const markerInfo = markerResult.added > 0 ? ` (${markerResult.added} markers added)` : '';
   472→      const statusMsg = takes.length > 0
   473→        ? `Found ${silences.length} silences, ${filtered.protected.length} protected by ${takes.length} takes${markerInfo}`
   474→        : `Found ${silences.length} silence(s) - review and apply${markerInfo}`;
   475→      setStatus(statusMsg);
   476→
   477→    } catch (err) {
   478→      showEmptyState();
   479→      setStatus('Error: ' + err.message);
   480→      console.error('[SPLICE] Unified workflow error:', err);
   481→    } finally {
   482→      isOperationInProgress = false;
   483→      ui.goBtn.disabled = false;
   484→    }
   485→  });
   486→}

   488→// ============================================================================
   489→// TAKE-AWARE SILENCE FILTERING
   490→// ============================================================================
   491→/**
   492→ * Filter silences to protect those that overlap with takes (speech segments).
   493→ * Returns { safe: [], protected: [] }
   494→ */
   495→function filterSilencesByTakes(silences, takes) {
   496→  if (!takes || takes.length === 0) {
   497→    // No takes = all silences are safe to remove
   498→    return { safe: silences, protected: [] };
   499→  }
   500→
   501→  const safe = [];
   502→  const protected_ = [];
   503→
   504→  silences.forEach(silence => {
   505→    // Check if this silence overlaps with any take
   506→    const overlaps = takes.some(take => {
   507→      const takeStart = take.startTime || take.start;
   508→      const takeEnd = take.endTime || take.end;
   509→      // Overlap exists if silence starts before take ends AND silence ends after take starts
   510→      return silence.start < takeEnd && silence.end > takeStart;
   511→    });
   512→
   513→    if (overlaps) {
   514→      protected_.push(silence);
   515→    } else {
   516→      safe.push(silence);
   517→    }
   518→  });
   519→
   520→  console.log(`[SPLICE] Filtered silences: ${safe.length} safe, ${protected_.length} protected`);
   521→  return { safe, protected: protected_ };
   522→}

   524→// ============================================================================
   525→// TIMELINE MARKERS (Visual indicators on Premiere Pro timeline)
   526→// ============================================================================
   527→/**
   528→ * Add colored markers to the timeline for silences and takes.
   529→ * - Red markers: silences safe to remove
   530→ * - Yellow markers: silences protected by takes
   531→ * - Blue markers: takes (speech segments)
   532→ *
   533→ * @param {Array} safeToRemove - Silences safe to remove
   534→ * @param {Array} protectedSilences - Silences overlapping speech
   535→ * @param {Array} takes - Detected speech segments
   536→ * @returns {Promise<{added: number, errors: number}>}
   537→ */
   538→async function addTimelineMarkers(safeToRemove, protectedSilences, takes) {
   539→  const results = { added: 0, errors: 0 };
   540→
   541→  try {
   542→    const context = await getActiveSequence();
   543→    if (!context) {
   544→      console.warn('[SPLICE] No sequence available for markers');
   545→      return results;
   546→    }
   547→
   548→    const { sequence, project } = context;
   549→
   550→    // Get markers object for the sequence
   551→    const markers = await ppro.Markers.getMarkers(sequence);
   552→    if (!markers) {
   553→      console.warn('[SPLICE] Markers API not available');
   554→      return results;
   555→    }
   556→
   557→    // Collect all marker actions
   558→    const markerActions = [];
   559→
   560→    // Red markers for safe-to-remove silences
   561→    for (const silence of safeToRemove) {
   562→      try {
   563→        const duration = silence.end - silence.start;
   564→
   565→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   566→        const startTime = ppro.TickTime.createWithSeconds(silence.start);
   567→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   568→
   569→        const action = markers.createAddMarkerAction(
   570→          `SPLICE: Remove (${duration.toFixed(2)}s)`,
   571→          ppro.Marker.MARKER_TYPE_COMMENT,
   572→          startTime,
   573→          durationTime,
   574→          `Silence to remove: ${formatTime(silence.start)} - ${formatTime(silence.end)}`
   575→        );
   576→        markerActions.push({ action, colorIndex: 1, type: 'silence' }); // Red = index 1
   577→      } catch (err) {
   578→        console.error('[SPLICE] Error creating silence marker:', err);
   579→        results.errors++;
   580→      }
   581→    }
   582→
   583→    // Yellow markers for protected silences
   584→    for (const silence of protectedSilences) {
   585→      try {
   586→        const duration = silence.end - silence.start;
   587→
   588→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   589→        const startTime = ppro.TickTime.createWithSeconds(silence.start);
   590→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   591→
   592→        const action = markers.createAddMarkerAction(
   593→          `SPLICE: Protected (${duration.toFixed(2)}s)`,
   594→          ppro.Marker.MARKER_TYPE_COMMENT,
   595→          startTime,
   596→          durationTime,
   597→          `Protected silence (overlaps speech): ${formatTime(silence.start)} - ${formatTime(silence.end)}`
   598→        );
   599→        markerActions.push({ action, colorIndex: 4, type: 'protected' }); // Yellow = index 4
   600→      } catch (err) {
   601→        console.error('[SPLICE] Error creating protected marker:', err);
   602→        results.errors++;
   603→      }
   604→    }
   605→
   606→    // Blue markers for takes (speech segments)
   607→    for (const take of takes) {
   608→      try {
   609→        const start = take.startTime || take.start;
   610→        const end = take.endTime || take.end;
   611→        const duration = end - start;
   612→        const textPreview = take.text?.substring(0, 30) || 'Speech';
   613→
   614→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   615→        const startTime = ppro.TickTime.createWithSeconds(start);
   616→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   617→
   618→        const action = markers.createAddMarkerAction(
   619→          `SPLICE: Take`,
   620→          ppro.Marker.MARKER_TYPE_COMMENT,
   621→          startTime,
   622→          durationTime,
   623→          `Speech segment: ${textPreview}${take.text?.length > 30 ? '...' : ''}`
   624→        );
   625→        markerActions.push({ action, colorIndex: 5, type: 'take' }); // Blue = index 5
   626→      } catch (err) {
   627→        console.error('[SPLICE] Error creating take marker:', err);
   628→        results.errors++;
   629→      }
   630→    }
   631→
   632→    // Execute all marker actions in a single transaction
   633→    if (markerActions.length > 0) {
   634→      await project.lockedAccess(async () => {
   635→        await project.executeTransaction((compoundAction) => {
   636→          for (const { action } of markerActions) {
   637→            compoundAction.addAction(action);
   638→          }
   639→        }, 'SPLICE: Add Timeline Markers');
   640→      });
   641→
   642→      results.added = markerActions.length;
   643→      console.log(`[SPLICE] Added ${results.added} timeline markers`);
   644→    }
   645→
   646→  } catch (err) {
   647→    console.error('[SPLICE] Timeline markers error:', err);
   648→    results.errors++;
   649→  }
   650→
   651→  return results;
   652→}

   654→/**
   655→ * Read current SPLICE marker positions from timeline.
   656→ * This allows users to manually adjust markers before applying.
   657→ * Returns silences based on current marker positions (not original detection).
   658→ */
   659→async function readMarkersAssilences() {
   660→  const silences = [];
   661→
   662→  try {
   663→    const context = await getActiveSequence();
   664→    if (!context) return silences;
   665→
   666→    const { sequence } = context;
   667→    const markers = await ppro.Markers.getMarkers(sequence);
   668→    if (!markers) return silences;
   669→
   670→    const allMarkers = await markers.getMarkers();
   671→    if (!allMarkers || allMarkers.length === 0) return silences;
   672→
   673→    for (const marker of allMarkers) {
   674→      const name = await marker.getName();
   675→
   676→      // Only read "Remove" markers (red ones that will be cut)
   677→      if (name && name.startsWith('SPLICE: Remove')) {
   678→        const startTickTime = await marker.getStart();
   679→        const durationTickTime = await marker.getDuration();
   680→
   681→        // Use TickTime.seconds property to get seconds directly (no TICKS_PER_SECOND needed)
   682→        const start = startTickTime.seconds;
   683→        const duration = durationTickTime.seconds;
   684→        const end = start + duration;
   685→
   686→        silences.push({
   687→          start,
   688→          end,
   689→          duration,
   690→          fromMarker: true  // Flag to indicate this came from marker adjustment
   691→        });
   692→      }
   693→    }
   694→
   695→    // Sort by start time
   696→    silences.sort((a, b) => a.start - b.start);
   697→    console.log(`[SPLICE] Read ${silences.length} silences from adjusted markers`);
   698→
   699→  } catch (err) {
   700→    console.error('[SPLICE] Error reading markers:', err);
   701→  }
   702→
   703→  return silences;
   704→}

   706→/**

   710→async function clearSpliceMarkers() {
   711→  try {
   712→    const context = await getActiveSequence();
   713→    if (!context) return 0;
   714→
   715→    const { sequence, project } = context;
   716→    const markers = await ppro.Markers.getMarkers(sequence);
   717→    if (!markers) return 0;
   718→
   719→    // Get all markers
   720→    const allMarkers = await markers.getMarkers();
   721→    if (!allMarkers || allMarkers.length === 0) return 0;
   722→
   723→    // Filter SPLICE markers
   724→    const spliceMarkers = [];
   725→    for (const marker of allMarkers) {
   726→      const name = await marker.getName();
   727→      if (name && name.startsWith('SPLICE:')) {
   728→        spliceMarkers.push(marker);
   729→      }
   730→    }
   731→
   732→    if (spliceMarkers.length === 0) return 0;
   733→
   734→    // Remove all SPLICE markers in a single transaction
   735→    await project.lockedAccess(async () => {
   736→      await project.executeTransaction((compoundAction) => {
   737→        for (const marker of spliceMarkers) {
   738→          const removeAction = markers.createRemoveMarkerAction(marker);
   739→          compoundAction.addAction(removeAction);
   740→        }
   741→      }, 'SPLICE: Clear Markers');
   742→    });
   743→
   744→    console.log(`[SPLICE] Cleared ${spliceMarkers.length} previous markers`);
   745→    return spliceMarkers.length;
   746→
   747→  } catch (err) {
   748→    console.error('[SPLICE] Error clearing markers:', err);
   749→    return 0;
   750→  }

  1350→/**

  1355→async function findFirstMediaItem(item) {

  1358→  // Fix: Use local Constant instead of app.Constants.ProjectItemType
  1359→  if (item.type === ProjectItemType.CLIP) {

  1363→  // Search Children (Bin or Root)
  1364→  // Note: item.children is a collection, check numItems
  1365→  if (item.children && item.children.numItems > 0) {

  1367→      const child = item.children[i];

  1372→
  1373→  return null;

  1797→          paddingEnd: params.padding / 2
  1798→        }
  1799→      })
  1800→    }, FETCH_TIMEOUT_PROCESSING);
  1801→
  1802→    if (!response.ok) {
  1803→      const errorMsg = await parseErrorResponse(response);
  1804→      throw new Error(errorMsg);
  1805→    }
  1806→
  1807→    const data = await response.json();
  1808→    currentBatchJobId = data.jobId;
  1809→    setStatus(`Batch job started: ${data.jobId}`);
  1810→
  1811→    // Poll for status
  1812→    await pollBatchStatus(currentBatchJobId, batchProgressBar, batchProgressText);
  1813→
  1814→  } catch (err) {
  1815→    setStatus('Batch error: ' + err.message);
  1816→  } finally {
  1817→    if (batchProgress) batchProgress.style.display = 'none';
  1818→  }
  1819→}

  1821→/**
  1822→ * Poll batch job status
  1823→ */
  1824→async function pollBatchStatus(jobId, progressBar, progressText) {
  1825→  const pollInterval = 2000; // 2 seconds
  1826→  const maxPolls = 300; // 10 minutes max
  1827→  let polls = 0;
  1828→
  1829→  while (polls < maxPolls) {
  1830→    try {
  1831→      const response = await fetchWithTimeout(`${getBackendUrl()}/batch/status/${jobId}`, {
  1832→        headers: getAuthHeaders()
  1833→      }, FETCH_TIMEOUT_DEFAULT);
  1834→
  1835→      if (!response.ok) {
  1836→        throw new Error('Failed to get batch status');
  1837→      }
  1838→
  1839→      const data = await response.json();
  1840→      const job = data.job;
  1841→
  1842→      // Update progress
  1843→      if (progressBar) progressBar.value = job.progress.percentage;
  1844→      if (progressText) progressText.textContent = `${job.progress.completed + job.progress.failed} / ${job.progress.total} files`;
  1845→
  1846→      // Check if complete
  1847→      if (job.status !== 'processing') {
  1848→        if (job.status === 'completed') {
  1849→          setStatus(`Batch complete: ${job.progress.completed} files processed`);
  1850→        } else if (job.status === 'completed_with_errors') {
  1851→          setStatus(`Batch complete with ${job.progress.failed} error(s)`);
  1852→        } else {
  1853→          setStatus('Batch failed');
  1854→        }
  1855→
  1856→        // Clear queue on completion
  1857→        batchQueue = [];
  1858→        updateBatchQueueDisplay();
  1859→        return;
  1860→      }
  1861→
  1862→      await new Promise(resolve => setTimeout(resolve, pollInterval));
  1863→      polls++;
  1864→    } catch (err) {
  1865→      setStatus('Polling error: ' + err.message);
  1866→      return;
  1867→    }
  1868→  }
  1869→
  1870→  setStatus('Batch timed out');
  1871→}

  1874→if (typeof window !== 'undefined') {
  1875→  window.removeBatchItem = removeBatchItem;
