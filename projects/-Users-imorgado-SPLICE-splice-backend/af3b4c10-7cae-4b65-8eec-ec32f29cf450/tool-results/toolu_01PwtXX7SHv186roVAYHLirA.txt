   159→function scheduleUIUpdate(callback) {
   160→  if (pendingUIUpdate) {
   161→    cancelAnimationFrame(pendingUIUpdate);
   162→  }
   163→  pendingUIUpdate = requestAnimationFrame(() => {
   164→    callback();
   165→    pendingUIUpdate = null;
   166→  });

   182→function initKeyboardShortcuts() {
   183→  document.addEventListener('keydown', (e) => {
   184→    // Only handle when combined preview is visible
   185→    const previewVisible = ui.combinedPreview?.style.display !== 'none' &&
   186→                           ui.combinedPreview?.style.display !== '';
   187→
   188→    if (!previewVisible) return;
   189→
   190→    const isMac = navigator.platform.toUpperCase().indexOf('MAC') >= 0;
   191→    const cmdKey = isMac ? e.metaKey : e.ctrlKey;
   192→
   193→    // Enter - Apply selected
   194→    if (e.key === 'Enter' && !e.shiftKey) {
   195→      e.preventDefault();
   196→      if (selectedSilenceIndices.size > 0) {
   197→        ui.applyPreviewBtn?.click();
   198→      }
   199→    }
   200→
   201→    // Escape - Cancel
   202→    if (e.key === 'Escape') {
   203→      e.preventDefault();
   204→      ui.cancelPreviewBtn?.click();
   205→    }
   206→
   207→    // Cmd/Ctrl+A - Select all
   208→    if (e.key === 'a' && cmdKey) {
   209→      e.preventDefault();
   210→      if (ui.selectAllSilences) {
   211→        ui.selectAllSilences.checked = true;
   212→        selectAllSilences(true);
   213→      }
   214→    }
   215→
   216→    // Cmd/Ctrl+D - Deselect all
   217→    if (e.key === 'd' && cmdKey) {
   218→      e.preventDefault();
   219→      if (ui.selectAllSilences) {
   220→        ui.selectAllSilences.checked = false;
   221→        selectAllSilences(false);
   222→      }
   223→    }
   224→
   225→    // Cmd/Ctrl+I - Invert selection
   226→    if (e.key === 'i' && cmdKey) {
   227→      e.preventDefault();
   228→      invertSilenceSelection();
   229→    }
   230→  });
   231→}

   233→// ============================================================================
   234→// UNIFIED WORKFLOW (Silence Detection + Takes Detection)
   235→// ============================================================================
   236→function initUnifiedWorkflow() {
   237→  if (!ui.goBtn) return;
   238→
   239→  ui.goBtn.addEventListener('click', async () => {
   240→    // Prevent double-click
   241→    if (isOperationInProgress) {
   242→      setStatus('Operation in progress...');
   243→      return;
   244→    }
   245→
   246→    isOperationInProgress = true;
   247→    ui.goBtn.disabled = true;
   248→
   249→    try {
   250→      showProgress('Preparing audio...');
   251→
   252→      // Step 1: Export audio
   253→      const exportSuccess = await exportAudioInternal();
   254→      if (!exportSuccess) {
   255→        throw new Error('Audio export failed');
   256→      }
   257→
   258→      // Step 2: Get settings
   259→      const sensitivity = parseInt(ui.sensitivitySlider?.value || 50);
   260→      const useIsolated = ui.sourceIsolated?.checked ?? false;
   261→      const detectTakes = ui.enableTakesDetection?.checked ?? true;
   262→
   263→      // Step 3: Vocal isolation if needed
   264→      let audioPath = WAV_PATH;
   265→      if (useIsolated) {
   266→        // Check isolation access before proceeding
   267→        const credits = getCurrentCredits();
   268→        if (!credits || !credits.hasIsolationAccess) {
   269→          throw new Error('Vocal isolation requires Pro or Team tier. Upgrade to access this feature.');
   270→        }
   271→
   272→        // Show overage warning if applicable
   273→        const isolationMins = credits.isolationHoursRemaining * 60;
   274→        if (isolationMins <= 0) {
   275→          const overageRate = credits.isolationOverageRate || 0.08;
   276→          showProgress(`Isolating vocals (overage: $${overageRate}/min)...`);
   277→        } else {
   278→          showProgress(`Isolating vocals (${isolationMins.toFixed(0)} min remaining)...`);
   279→        }
   280→
   281→        const isolateResult = await isolateVocals();
   282→        if (isolateResult.success) {
   283→          audioPath = isolateResult.outputPath;
   284→          // Update credits display after isolation
   285→          await refreshCredits();
   286→        } else {
   287→          throw new Error(isolateResult.error || 'Vocal isolation failed');
   288→        }
   289→      }
   290→
   291→      // Step 4: Run detection (parallel if takes enabled)
   292→      const params = mapSensitivity(sensitivity);
   293→      let silences = [];
   294→      let takes = [];
   295→
   296→      if (detectTakes) {
   297→        // Parallel detection for performance
   298→        showProgress('Detecting silences and analyzing takes...');
   299→        const [silenceResult, transcriptResult] = await Promise.all([
   300→          detectSilences(audioPath, params),
   301→          transcribeAudio()
   302→        ]);
   303→
   304→        silences = silenceResult;
   305→
   306→        // Extract takes from transcript result
   307→        if (transcriptResult.success && transcriptResult.takes) {
   308→          if (transcriptResult.takes.takes) {
   309→            takes = transcriptResult.takes.takes;
   310→          } else if (Array.isArray(transcriptResult.takes)) {
   311→            takes = transcriptResult.takes;
   312→          }
   313→        }
   314→
   315→        // Store transcript for reference
   316→        window.currentTranscript = transcriptResult.transcript;
   317→      } else {
   318→        // Just detect silences
   319→        showProgress('Detecting silences...');
   320→        silences = await detectSilences(audioPath, params);
   321→      }
   322→
   323→      if (silences.length === 0) {
   324→        showEmptyState();
   325→        setStatus('No silences detected. Try increasing sensitivity.');
   326→        return;
   327→      }
   328→
   329→      // Step 5: Filter silences by takes (protect speech)
   330→      const filtered = filterSilencesByTakes(silences, takes);
   331→
   332→      // Step 6: Add timeline markers (clear old ones first)
   333→      showProgress('Adding timeline markers...');
   334→      await clearSpliceMarkers();
   335→      const markerResult = await addTimelineMarkers(filtered.safe, filtered.protected, takes);
   336→
   337→      // Step 7: Show combined preview
   338→      showCombinedPreview(filtered.safe, filtered.protected, takes);
   339→
   340→      const markerInfo = markerResult.added > 0 ? ` (${markerResult.added} markers added)` : '';
   341→      const statusMsg = takes.length > 0
   342→        ? `Found ${silences.length} silences, ${filtered.protected.length} protected by ${takes.length} takes${markerInfo}`
   343→        : `Found ${silences.length} silence(s) - review and apply${markerInfo}`;
   344→      setStatus(statusMsg);
   345→
   346→    } catch (err) {
   347→      showEmptyState();
   348→      setStatus('Error: ' + err.message);
   349→      console.error('[SPLICE] Unified workflow error:', err);
   350→    } finally {
   351→      isOperationInProgress = false;
   352→      ui.goBtn.disabled = false;
   353→    }
   354→  });
   355→}

   357→// ============================================================================
   358→// TAKE-AWARE SILENCE FILTERING
   359→// ============================================================================
   360→/**
   361→ * Filter silences to protect those that overlap with takes (speech segments).
   362→ * Returns { safe: [], protected: [] }
   363→ */
   364→function filterSilencesByTakes(silences, takes) {
   365→  if (!takes || takes.length === 0) {
   366→    // No takes = all silences are safe to remove
   367→    return { safe: silences, protected: [] };
   368→  }
   369→
   370→  const safe = [];
   371→  const protected_ = [];
   372→
   373→  silences.forEach(silence => {
   374→    // Check if this silence overlaps with any take
   375→    const overlaps = takes.some(take => {
   376→      const takeStart = take.startTime || take.start;
   377→      const takeEnd = take.endTime || take.end;
   378→      // Overlap exists if silence starts before take ends AND silence ends after take starts
   379→      return silence.start < takeEnd && silence.end > takeStart;
   380→    });
   381→
   382→    if (overlaps) {
   383→      protected_.push(silence);
   384→    } else {
   385→      safe.push(silence);
   386→    }
   387→  });
   388→
   389→  console.log(`[SPLICE] Filtered silences: ${safe.length} safe, ${protected_.length} protected`);
   390→  return { safe, protected: protected_ };
   391→}

   393→// ============================================================================
   394→// TIMELINE MARKERS (Visual indicators on Premiere Pro timeline)
   395→// ============================================================================
   396→/**
   397→ * Add colored markers to the timeline for silences and takes.
   398→ * - Red markers: silences safe to remove
   399→ * - Yellow markers: silences protected by takes
   400→ * - Blue markers: takes (speech segments)
   401→ *
   402→ * @param {Array} safeToRemove - Silences safe to remove
   403→ * @param {Array} protectedSilences - Silences overlapping speech
   404→ * @param {Array} takes - Detected speech segments
   405→ * @returns {Promise<{added: number, errors: number}>}
   406→ */
   407→async function addTimelineMarkers(safeToRemove, protectedSilences, takes) {
   408→  const results = { added: 0, errors: 0 };
   409→
   410→  try {
   411→    const context = await getActiveSequence();
   412→    if (!context) {
   413→      console.warn('[SPLICE] No sequence available for markers');
   414→      return results;
   415→    }
   416→
   417→    const { sequence, project } = context;
   418→
   419→    // Get markers object for the sequence
   420→    const markers = await ppro.Markers.getMarkers(sequence);
   421→    if (!markers) {
   422→      console.warn('[SPLICE] Markers API not available');
   423→      return results;
   424→    }
   425→
   426→    // Collect all marker actions
   427→    const markerActions = [];
   428→
   429→    // Red markers for safe-to-remove silences
   430→    for (const silence of safeToRemove) {
   431→      try {
   432→        const duration = silence.end - silence.start;
   433→
   434→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   435→        const startTime = ppro.TickTime.createWithSeconds(silence.start);
   436→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   437→
   438→        const action = markers.createAddMarkerAction(
   439→          `SPLICE: Remove (${duration.toFixed(2)}s)`,
   440→          ppro.Marker.MARKER_TYPE_COMMENT,
   441→          startTime,
   442→          durationTime,
   443→          `Silence to remove: ${formatTime(silence.start)} - ${formatTime(silence.end)}`
   444→        );
   445→        markerActions.push({ action, colorIndex: 1, type: 'silence' }); // Red = index 1
   446→      } catch (err) {
   447→        console.error('[SPLICE] Error creating silence marker:', err);
   448→        results.errors++;
   449→      }
   450→    }
   451→
   452→    // Yellow markers for protected silences
   453→    for (const silence of protectedSilences) {
   454→      try {
   455→        const duration = silence.end - silence.start;
   456→
   457→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   458→        const startTime = ppro.TickTime.createWithSeconds(silence.start);
   459→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   460→
   461→        const action = markers.createAddMarkerAction(
   462→          `SPLICE: Protected (${duration.toFixed(2)}s)`,
   463→          ppro.Marker.MARKER_TYPE_COMMENT,
   464→          startTime,
   465→          durationTime,
   466→          `Protected silence (overlaps speech): ${formatTime(silence.start)} - ${formatTime(silence.end)}`
   467→        );
   468→        markerActions.push({ action, colorIndex: 4, type: 'protected' }); // Yellow = index 4
   469→      } catch (err) {
   470→        console.error('[SPLICE] Error creating protected marker:', err);
   471→        results.errors++;
   472→      }
   473→    }
   474→
   475→    // Blue markers for takes (speech segments)
   476→    for (const take of takes) {
   477→      try {
   478→        const start = take.startTime || take.start;
   479→        const end = take.endTime || take.end;
   480→        const duration = end - start;
   481→        const textPreview = take.text?.substring(0, 30) || 'Speech';
   482→
   483→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   484→        const startTime = ppro.TickTime.createWithSeconds(start);
   485→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   486→
   487→        const action = markers.createAddMarkerAction(
   488→          `SPLICE: Take`,
   489→          ppro.Marker.MARKER_TYPE_COMMENT,
   490→          startTime,
   491→          durationTime,
   492→          `Speech segment: ${textPreview}${take.text?.length > 30 ? '...' : ''}`
   493→        );
   494→        markerActions.push({ action, colorIndex: 5, type: 'take' }); // Blue = index 5
   495→      } catch (err) {
   496→        console.error('[SPLICE] Error creating take marker:', err);
   497→        results.errors++;
   498→      }
   499→    }
   500→
   501→    // Execute all marker actions in a single transaction
   502→    if (markerActions.length > 0) {
   503→      await project.lockedAccess(async () => {
   504→        await project.executeTransaction((compoundAction) => {
   505→          for (const { action } of markerActions) {
   506→            compoundAction.addAction(action);
   507→          }
   508→        }, 'SPLICE: Add Timeline Markers');
   509→      });
   510→
   511→      results.added = markerActions.length;
   512→      console.log(`[SPLICE] Added ${results.added} timeline markers`);
   513→    }
   514→
   515→  } catch (err) {
   516→    console.error('[SPLICE] Timeline markers error:', err);
   517→    results.errors++;
   518→  }
   519→
   520→  return results;
   521→}

   523→/**
   524→ * Read current SPLICE marker positions from timeline.
   525→ * This allows users to manually adjust markers before applying.
   526→ * Returns silences based on current marker positions (not original detection).
   527→ */
   528→async function readMarkersAssilences() {
   529→  const silences = [];
   530→
   531→  try {
   532→    const context = await getActiveSequence();
   533→    if (!context) return silences;
   534→
   535→    const { sequence } = context;
   536→    const markers = await ppro.Markers.getMarkers(sequence);
   537→    if (!markers) return silences;
   538→
   539→    const allMarkers = await markers.getMarkers();
   540→    if (!allMarkers || allMarkers.length === 0) return silences;
   541→
   542→    for (const marker of allMarkers) {
   543→      const name = await marker.getName();
   544→
   545→      // Only read "Remove" markers (red ones that will be cut)
   546→      if (name && name.startsWith('SPLICE: Remove')) {
   547→        const startTickTime = await marker.getStart();
   548→        const durationTickTime = await marker.getDuration();
   549→
   550→        // Use TickTime.seconds property to get seconds directly (no TICKS_PER_SECOND needed)
   551→        const start = startTickTime.seconds;
   552→        const duration = durationTickTime.seconds;
   553→        const end = start + duration;
   554→
   555→        silences.push({
   556→          start,
   557→          end,
   558→          duration,
   559→          fromMarker: true  // Flag to indicate this came from marker adjustment
   560→        });
   561→      }
   562→    }
   563→
   564→    // Sort by start time
   565→    silences.sort((a, b) => a.start - b.start);
   566→    console.log(`[SPLICE] Read ${silences.length} silences from adjusted markers`);
   567→
   568→  } catch (err) {
   569→    console.error('[SPLICE] Error reading markers:', err);
   570→  }
   571→
   572→  return silences;
   573→}

   575→/**
   576→ * Remove all SPLICE markers from the timeline.
   577→ * Call this before adding new markers to avoid duplicates.
   578→ */
   579→async function clearSpliceMarkers() {
   580→  try {
   581→    const context = await getActiveSequence();
   582→    if (!context) return 0;
   583→
   584→    const { sequence, project } = context;
   585→    const markers = await ppro.Markers.getMarkers(sequence);
   586→    if (!markers) return 0;
   587→
   588→    // Get all markers
   589→    const allMarkers = await markers.getMarkers();
   590→    if (!allMarkers || allMarkers.length === 0) return 0;
   591→
   592→    // Filter SPLICE markers
   593→    const spliceMarkers = [];
   594→    for (const marker of allMarkers) {
   595→      const name = await marker.getName();
   596→      if (name && name.startsWith('SPLICE:')) {
   597→        spliceMarkers.push(marker);
   598→      }
   599→    }
   600→
   601→    if (spliceMarkers.length === 0) return 0;
   602→
   603→    // Remove all SPLICE markers in a single transaction
   604→    await project.lockedAccess(async () => {
   605→      await project.executeTransaction((compoundAction) => {
   606→        for (const marker of spliceMarkers) {
   607→          const removeAction = markers.createRemoveMarkerAction(marker);
   608→          compoundAction.addAction(removeAction);
   609→        }
   610→      }, 'SPLICE: Clear Markers');
   611→    });
   612→
   613→    console.log(`[SPLICE] Cleared ${spliceMarkers.length} previous markers`);
   614→    return spliceMarkers.length;
   615→
   616→  } catch (err) {

  1032→  safeSilences = [];

  1185→/**

  1629→/**
  1630→ * Poll batch job status
  1631→ */
  1632→async function pollBatchStatus(jobId, progressBar, progressText) {
  1633→  const pollInterval = 2000; // 2 seconds
  1634→  const maxPolls = 300; // 10 minutes max
  1635→  let polls = 0;
  1636→
  1637→  while (polls < maxPolls) {
  1638→    try {
  1639→      const response = await fetch(`${BACKEND_URL}/batch/status/${jobId}`, {
  1640→        headers: getAuthHeaders()
  1641→      });
  1642→
  1643→      if (!response.ok) {
  1644→        throw new Error('Failed to get batch status');
  1645→      }
  1646→
  1647→      const data = await response.json();
  1648→      const job = data.job;
  1649→
  1650→      // Update progress
  1651→      if (progressBar) progressBar.value = job.progress.percentage;
  1652→      if (progressText) progressText.textContent = `${job.progress.completed + job.progress.failed} / ${job.progress.total} files`;
  1653→
  1654→      // Check if complete
  1655→      if (job.status !== 'processing') {
  1656→        if (job.status === 'completed') {
  1657→          setStatus(`Batch complete: ${job.progress.completed} files processed`);
  1658→        } else if (job.status === 'completed_with_errors') {
  1659→          setStatus(`Batch complete with ${job.progress.failed} error(s)`);
  1660→        } else {
  1661→          setStatus('Batch failed');
  1662→        }
  1663→
  1664→        // Clear queue on completion
  1665→        batchQueue = [];
  1666→        updateBatchQueueDisplay();
  1667→        return;
  1668→      }
  1669→
  1670→      await new Promise(resolve => setTimeout(resolve, pollInterval));
  1671→      polls++;
  1672→    } catch (err) {
  1673→      setStatus('Polling error: ' + err.message);
  1674→      return;
  1675→    }
  1676→  }
  1677→
  1678→  setStatus('Batch timed out');
  1679→}
