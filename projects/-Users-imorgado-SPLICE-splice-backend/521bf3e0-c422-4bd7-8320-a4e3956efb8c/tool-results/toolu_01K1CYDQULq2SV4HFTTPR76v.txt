   113→let confirmModalCallback = null;
   114→
   115→/**
   116→ * Show a confirmation modal dialog (UXP-compatible replacement for confirm())
   117→ * @param {string} title - Modal title
   118→ * @param {string} message - Confirmation message
   119→ * @param {Function} onConfirm - Callback when confirmed
   120→ * @param {Function} [onCancel] - Optional callback when cancelled
   121→ * @param {Object} [options] - Optional customization { confirmText, cancelText, confirmStyle }
   122→ */
   123→function showConfirmModal(title, message, onConfirm, onCancel = null, options = {}) {
   124→  if (!ui.confirmModal) return;
   125→
   126→  // Set content
   127→  if (ui.confirmModalTitle) ui.confirmModalTitle.textContent = title;
   128→  if (ui.confirmModalMessage) ui.confirmModalMessage.textContent = message;
   129→
   130→  // Customize button text if provided
   131→  if (ui.confirmModalConfirmBtn) {
   132→    ui.confirmModalConfirmBtn.textContent = options.confirmText || 'Confirm';
   133→    ui.confirmModalConfirmBtn.style.background = options.confirmStyle || '#dc3545';
   134→  }
   135→  if (ui.confirmModalCancelBtn) {
   136→    ui.confirmModalCancelBtn.textContent = options.cancelText || 'Cancel';
   137→  }
   138→
   139→  // Store callbacks
   140→  confirmModalCallback = { onConfirm, onCancel };
   141→
   142→  // Show modal
   143→  ui.confirmModal.classList.remove('hidden');
   144→
   145→  // Focus cancel button for safety (user must explicitly confirm)
   146→  if (ui.confirmModalCancelBtn) ui.confirmModalCancelBtn.focus();
   147→}
   148→
   149→function hideConfirmModal(confirmed = false) {
   150→  if (!ui.confirmModal) return;
   151→
   152→  ui.confirmModal.classList.add('hidden');
   153→
   154→  // Execute callback
   155→  if (confirmModalCallback) {
   156→    if (confirmed && confirmModalCallback.onConfirm) {
   157→      confirmModalCallback.onConfirm();
   158→    } else if (!confirmed && confirmModalCallback.onCancel) {
   159→      confirmModalCallback.onCancel();
   160→    }
   161→    confirmModalCallback = null;
   162→  }
   163→}

   275→// DEBOUNCED UI UPDATES (Performance: batch DOM updates)
   276→// ============================================================================
   277→function scheduleUIUpdate(callback) {
   278→  if (pendingUIUpdate) {
   279→    cancelAnimationFrame(pendingUIUpdate);
   280→  }
   281→  pendingUIUpdate = requestAnimationFrame(() => {
   282→    callback();
   283→    pendingUIUpdate = null;
   284→  });
   285→}

   300→function initKeyboardShortcuts() {
   301→  document.addEventListener('keydown', (e) => {
   302→    // Only handle when combined preview is visible
   303→    const previewVisible = ui.combinedPreview?.style.display !== 'none' &&
   304→                           ui.combinedPreview?.style.display !== '';
   305→
   306→    if (!previewVisible) return;
   307→
   308→    const isMac = navigator.platform.toUpperCase().indexOf('MAC') >= 0;
   309→    const cmdKey = isMac ? e.metaKey : e.ctrlKey;
   310→
   311→    // Enter - Apply selected
   312→    if (e.key === 'Enter' && !e.shiftKey) {
   313→      e.preventDefault();
   314→      if (selectedSilenceIndices.size > 0) {
   315→        ui.applyPreviewBtn?.click();
   316→      }
   317→    }
   318→
   319→    // Escape - Cancel
   320→    if (e.key === 'Escape') {
   321→      e.preventDefault();
   322→      ui.cancelPreviewBtn?.click();
   323→    }
   324→
   325→    // Cmd/Ctrl+A - Select all
   326→    if (e.key === 'a' && cmdKey) {
   327→      e.preventDefault();
   328→      if (ui.selectAllSilences) {
   329→        ui.selectAllSilences.checked = true;
   330→        selectAllSilences(true);
   331→      }
   332→    }
   333→
   334→    // Cmd/Ctrl+D - Deselect all
   335→    if (e.key === 'd' && cmdKey) {
   336→      e.preventDefault();
   337→      if (ui.selectAllSilences) {
   338→        ui.selectAllSilences.checked = false;
   339→        selectAllSilences(false);
   340→      }
   341→    }
   342→
   343→    // Cmd/Ctrl+I - Invert selection
   344→    if (e.key === 'i' && cmdKey) {
   345→      e.preventDefault();
   346→      invertSilenceSelection();
   347→    }
   348→  });
   349→}

   351→// ============================================================================
   352→// UNIFIED WORKFLOW (Silence Detection + Takes Detection)
   353→// ============================================================================
   354→function initUnifiedWorkflow() {
   355→  if (!ui.goBtn) return;
   356→
   357→  ui.goBtn.addEventListener('click', async () => {
   358→    // Prevent double-click
   359→    if (isOperationInProgress) {
   360→      setStatus('Operation in progress...');
   361→      return;
   362→    }
   363→
   364→    isOperationInProgress = true;
   365→    ui.goBtn.disabled = true;
   366→
   367→    try {
   368→      showProgress('Preparing audio...');
   369→
   370→      // Step 1: Export audio
   371→      const exportSuccess = await exportAudioInternal();
   372→      if (!exportSuccess) {
   373→        throw new Error('Audio export failed');
   374→      }
   375→
   376→      // Step 2: Get settings
   377→      const sensitivity = parseInt(ui.sensitivitySlider?.value || 50);
   378→      const useIsolated = ui.sourceIsolated?.checked ?? false;
   379→      const detectTakes = ui.enableTakesDetection?.checked ?? true;
   380→
   381→      // Step 3: Vocal isolation if needed
   382→      let audioPath = WAV_PATH;
   383→      if (useIsolated) {
   384→        // Check isolation access before proceeding
   385→        const credits = getCurrentCredits();
   386→        if (!credits || !credits.hasIsolationAccess) {
   387→          throw new Error('Vocal isolation requires Pro or Team tier. Upgrade to access this feature.');
   388→        }
   389→
   390→        // Show overage warning if applicable
   391→        const isolationMins = credits.isolationHoursRemaining * 60;
   392→        if (isolationMins <= 0) {
   393→          const overageRate = credits.isolationOverageRate || 0.08;
   394→          showProgress(`Isolating vocals (overage: $${overageRate}/min)...`);
   395→        } else {
   396→          showProgress(`Isolating vocals (${isolationMins.toFixed(0)} min remaining)...`);
   397→        }
   398→
   399→        const isolateResult = await isolateVocals();
   400→        if (isolateResult.success) {
   401→          audioPath = isolateResult.outputPath;
   402→          // Update credits display after isolation
   403→          await refreshCredits();
   404→        } else {
   405→          throw new Error(isolateResult.error || 'Vocal isolation failed');
   406→        }
   407→      }
   408→
   409→      // Step 4: Run detection (parallel if takes enabled)
   410→      const params = mapSensitivity(sensitivity);
   411→      let silences = [];
   412→      let takes = [];
   413→
   414→      if (detectTakes) {
   415→        // Parallel detection for performance
   416→        showProgress('Detecting silences and analyzing takes...');
   417→        const [silenceResult, transcriptResult] = await Promise.all([
   418→          detectSilences(audioPath, params),
   419→          transcribeAudio()
   420→        ]);
   421→
   422→        silences = silenceResult;
   423→
   424→        // Extract takes from transcript result
   425→        if (transcriptResult.success && transcriptResult.takes) {
   426→          if (transcriptResult.takes.takes) {
   427→            takes = transcriptResult.takes.takes;
   428→          } else if (Array.isArray(transcriptResult.takes)) {
   429→            takes = transcriptResult.takes;
   430→          }
   431→        }
   432→
   433→        // Store transcript for reference
   434→        window.currentTranscript = transcriptResult.transcript;
   435→      } else {
   436→        // Just detect silences
   437→        showProgress('Detecting silences...');
   438→        silences = await detectSilences(audioPath, params);
   439→      }
   440→
   441→      if (silences.length === 0) {
   442→        showEmptyState();
   443→        setStatus('No silences detected. Try increasing sensitivity.');
   444→        return;
   445→      }
   446→
   447→      // Step 5: Filter silences by takes (protect speech)
   448→      const filtered = filterSilencesByTakes(silences, takes);
   449→
   450→      // Step 6: Add timeline markers (clear old ones first)
   451→      showProgress('Adding timeline markers...');
   452→      await clearSpliceMarkers();
   453→      const markerResult = await addTimelineMarkers(filtered.safe, filtered.protected, takes);
   454→
   455→      // Step 7: Show combined preview
   456→      showCombinedPreview(filtered.safe, filtered.protected, takes);
   457→
   458→      const markerInfo = markerResult.added > 0 ? ` (${markerResult.added} markers added)` : '';
   459→      const statusMsg = takes.length > 0
   460→        ? `Found ${silences.length} silences, ${filtered.protected.length} protected by ${takes.length} takes${markerInfo}`
   461→        : `Found ${silences.length} silence(s) - review and apply${markerInfo}`;
   462→      setStatus(statusMsg);
   463→
   464→    } catch (err) {
   465→      showEmptyState();
   466→      setStatus('Error: ' + err.message);
   467→      console.error('[SPLICE] Unified workflow error:', err);
   468→    } finally {
   469→      isOperationInProgress = false;
   470→      ui.goBtn.disabled = false;
   471→    }
   472→  });
   473→}

   475→// ============================================================================
   476→// TAKE-AWARE SILENCE FILTERING
   477→// ============================================================================
   478→/**
   479→ * Filter silences to protect those that overlap with takes (speech segments).
   480→ * Returns { safe: [], protected: [] }
   481→ */
   482→function filterSilencesByTakes(silences, takes) {
   483→  if (!takes || takes.length === 0) {
   484→    // No takes = all silences are safe to remove
   485→    return { safe: silences, protected: [] };
   486→  }
   487→
   488→  const safe = [];
   489→  const protected_ = [];
   490→
   491→  silences.forEach(silence => {
   492→    // Check if this silence overlaps with any take
   493→    const overlaps = takes.some(take => {
   494→      const takeStart = take.startTime || take.start;
   495→      const takeEnd = take.endTime || take.end;
   496→      // Overlap exists if silence starts before take ends AND silence ends after take starts
   497→      return silence.start < takeEnd && silence.end > takeStart;
   498→    });
   499→
   500→    if (overlaps) {
   501→      protected_.push(silence);
   502→    } else {
   503→      safe.push(silence);
   504→    }
   505→  });
   506→
   507→  console.log(`[SPLICE] Filtered silences: ${safe.length} safe, ${protected_.length} protected`);
   508→  return { safe, protected: protected_ };
   509→}

   511→// ============================================================================
   512→// TIMELINE MARKERS (Visual indicators on Premiere Pro timeline)
   513→// ============================================================================
   514→/**
   515→ * Add colored markers to the timeline for silences and takes.
   516→ * - Red markers: silences safe to remove
   517→ * - Yellow markers: silences protected by takes
   518→ * - Blue markers: takes (speech segments)
   519→ *
   520→ * @param {Array} safeToRemove - Silences safe to remove
   521→ * @param {Array} protectedSilences - Silences overlapping speech
   522→ * @param {Array} takes - Detected speech segments
   523→ * @returns {Promise<{added: number, errors: number}>}
   524→ */
   525→async function addTimelineMarkers(safeToRemove, protectedSilences, takes) {
   526→  const results = { added: 0, errors: 0 };
   527→
   528→  try {
   529→    const context = await getActiveSequence();
   530→    if (!context) {
   531→      console.warn('[SPLICE] No sequence available for markers');
   532→      return results;
   533→    }
   534→
   535→    const { sequence, project } = context;
   536→
   537→    // Get markers object for the sequence
   538→    const markers = await ppro.Markers.getMarkers(sequence);
   539→    if (!markers) {
   540→      console.warn('[SPLICE] Markers API not available');
   541→      return results;
   542→    }
   543→
   544→    // Collect all marker actions
   545→    const markerActions = [];
   546→
   547→    // Red markers for safe-to-remove silences
   548→    for (const silence of safeToRemove) {
   549→      try {
   550→        const duration = silence.end - silence.start;
   551→
   552→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   553→        const startTime = ppro.TickTime.createWithSeconds(silence.start);
   554→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   555→
   556→        const action = markers.createAddMarkerAction(
   557→          `SPLICE: Remove (${duration.toFixed(2)}s)`,
   558→          ppro.Marker.MARKER_TYPE_COMMENT,
   559→          startTime,
   560→          durationTime,
   561→          `Silence to remove: ${formatTime(silence.start)} - ${formatTime(silence.end)}`
   562→        );
   563→        markerActions.push({ action, colorIndex: 1, type: 'silence' }); // Red = index 1
   564→      } catch (err) {
   565→        console.error('[SPLICE] Error creating silence marker:', err);
   566→        results.errors++;
   567→      }
   568→    }
   569→
   570→    // Yellow markers for protected silences
   571→    for (const silence of protectedSilences) {
   572→      try {
   573→        const duration = silence.end - silence.start;
   574→
   575→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   576→        const startTime = ppro.TickTime.createWithSeconds(silence.start);
   577→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   578→
   579→        const action = markers.createAddMarkerAction(
   580→          `SPLICE: Protected (${duration.toFixed(2)}s)`,
   581→          ppro.Marker.MARKER_TYPE_COMMENT,
   582→          startTime,
   583→          durationTime,
   584→          `Protected silence (overlaps speech): ${formatTime(silence.start)} - ${formatTime(silence.end)}`
   585→        );
   586→        markerActions.push({ action, colorIndex: 4, type: 'protected' }); // Yellow = index 4
   587→      } catch (err) {
   588→        console.error('[SPLICE] Error creating protected marker:', err);
   589→        results.errors++;
   590→      }
   591→    }
   592→
   593→    // Blue markers for takes (speech segments)
   594→    for (const take of takes) {
   595→      try {
   596→        const start = take.startTime || take.start;
   597→        const end = take.endTime || take.end;
   598→        const duration = end - start;
   599→        const textPreview = take.text?.substring(0, 30) || 'Speech';
   600→
   601→        // Use TickTime API directly (no TICKS_PER_SECOND needed)
   602→        const startTime = ppro.TickTime.createWithSeconds(start);
   603→        const durationTime = ppro.TickTime.createWithSeconds(duration);
   604→
   605→        const action = markers.createAddMarkerAction(
   606→          `SPLICE: Take`,
   607→          ppro.Marker.MARKER_TYPE_COMMENT,
   608→          startTime,
   609→          durationTime,
   610→          `Speech segment: ${textPreview}${take.text?.length > 30 ? '...' : ''}`
   611→        );
   612→        markerActions.push({ action, colorIndex: 5, type: 'take' }); // Blue = index 5
   613→      } catch (err) {
   614→        console.error('[SPLICE] Error creating take marker:', err);
   615→        results.errors++;
   616→      }
   617→    }
   618→
   619→    // Execute all marker actions in a single transaction
   620→    if (markerActions.length > 0) {
   621→      await project.lockedAccess(async () => {
   622→        await project.executeTransaction((compoundAction) => {
   623→          for (const { action } of markerActions) {
   624→            compoundAction.addAction(action);
   625→          }
   626→        }, 'SPLICE: Add Timeline Markers');
   627→      });
   628→
   629→      results.added = markerActions.length;
   630→      console.log(`[SPLICE] Added ${results.added} timeline markers`);
   631→    }
   632→
   633→  } catch (err) {
   634→    console.error('[SPLICE] Timeline markers error:', err);
   635→    results.errors++;
   636→  }
   637→
   638→  return results;
   639→}

   641→/**
   642→ * Read current SPLICE marker positions from timeline.
   643→ * This allows users to manually adjust markers before applying.
   644→ * Returns silences based on current marker positions (not original detection).
   645→ */
   646→async function readMarkersAssilences() {
   647→  const silences = [];
   648→
   649→  try {
   650→    const context = await getActiveSequence();
   651→    if (!context) return silences;
   652→
   653→    const { sequence } = context;
   654→    const markers = await ppro.Markers.getMarkers(sequence);
   655→    if (!markers) return silences;
   656→
   657→    const allMarkers = await markers.getMarkers();
   658→    if (!allMarkers || allMarkers.length === 0) return silences;
   659→
   660→    for (const marker of allMarkers) {
   661→      const name = await marker.getName();
   662→
   663→      // Only read "Remove" markers (red ones that will be cut)
   664→      if (name && name.startsWith('SPLICE: Remove')) {
   665→        const startTickTime = await marker.getStart();
   666→        const durationTickTime = await marker.getDuration();
   667→
   668→        // Use TickTime.seconds property to get seconds directly (no TICKS_PER_SECOND needed)
   669→        const start = startTickTime.seconds;
   670→        const duration = durationTickTime.seconds;
   671→        const end = start + duration;
   672→
   673→        silences.push({
   674→          start,
   675→          end,
   676→          duration,
   677→          fromMarker: true  // Flag to indicate this came from marker adjustment
   678→        });
   679→      }
   680→    }
   681→
   682→    // Sort by start time
   683→    silences.sort((a, b) => a.start - b.start);
   684→    console.log(`[SPLICE] Read ${silences.length} silences from adjusted markers`);
   685→
   686→  } catch (err) {
   687→    console.error('[SPLICE] Error reading markers:', err);
   688→  }
   689→
   690→  return silences;
   691→}

   693→/**
   694→ * Remove all SPLICE markers from the timeline.
   695→ * Call this before adding new markers to avoid duplicates.
   696→ */
   697→async function clearSpliceMarkers() {
   698→  try {
   699→    const context = await getActiveSequence();
   700→    if (!context) return 0;
   701→
   702→    const { sequence, project } = context;
   703→    const markers = await ppro.Markers.getMarkers(sequence);
   704→    if (!markers) return 0;
   705→
   706→    // Get all markers
   707→    const allMarkers = await markers.getMarkers();
   708→    if (!allMarkers || allMarkers.length === 0) return 0;
   709→
   710→    // Filter SPLICE markers
   711→    const spliceMarkers = [];
   712→    for (const marker of allMarkers) {
   713→      const name = await marker.getName();
   714→      if (name && name.startsWith('SPLICE:')) {
   715→        spliceMarkers.push(marker);
   716→      }
   717→    }
   718→
   719→    if (spliceMarkers.length === 0) return 0;
   720→
   721→    // Remove all SPLICE markers in a single transaction
   722→    await project.lockedAccess(async () => {
   723→      await project.executeTransaction((compoundAction) => {
   724→        for (const marker of spliceMarkers) {
   725→          const removeAction = markers.createRemoveMarkerAction(marker);
   726→          compoundAction.addAction(removeAction);
   727→        }
   728→      }, 'SPLICE: Clear Markers');
   729→    });
   730→
   731→    console.log(`[SPLICE] Cleared ${spliceMarkers.length} previous markers`);
   732→    return spliceMarkers.length;
   733→
   734→  } catch (err) {
   735→    console.error('[SPLICE] Error clearing markers:', err);
   736→    return 0;
   737→  }
   738→}

  1339→/**

  1344→async function findFirstMediaItem(item) {

  1352→  // Search Children (Bin or Root)

  1356→      const child = item.children[i];

  1798→/**
  1799→ * Poll batch job status
  1800→ */
  1801→async function pollBatchStatus(jobId, progressBar, progressText) {
  1802→  const pollInterval = 2000; // 2 seconds
  1803→  const maxPolls = 300; // 10 minutes max
  1804→  let polls = 0;
  1805→
  1806→  while (polls < maxPolls) {
  1807→    try {
  1808→      const response = await fetch(`${BACKEND_URL}/batch/status/${jobId}`, {
  1809→        headers: getAuthHeaders()
  1810→      });
  1811→
  1812→      if (!response.ok) {
  1813→        throw new Error('Failed to get batch status');
  1814→      }
  1815→
  1816→      const data = await response.json();
  1817→      const job = data.job;
  1818→
  1819→      // Update progress
  1820→      if (progressBar) progressBar.value = job.progress.percentage;
  1821→      if (progressText) progressText.textContent = `${job.progress.completed + job.progress.failed} / ${job.progress.total} files`;
  1822→
  1823→      // Check if complete
  1824→      if (job.status !== 'processing') {
  1825→        if (job.status === 'completed') {
  1826→          setStatus(`Batch complete: ${job.progress.completed} files processed`);
  1827→        } else if (job.status === 'completed_with_errors') {
  1828→          setStatus(`Batch complete with ${job.progress.failed} error(s)`);
  1829→        } else {
  1830→          setStatus('Batch failed');
  1831→        }
  1832→
  1833→        // Clear queue on completion
  1834→        batchQueue = [];
  1835→        updateBatchQueueDisplay();
  1836→        return;
  1837→      }
  1838→
  1839→      await new Promise(resolve => setTimeout(resolve, pollInterval));
  1840→      polls++;
  1841→    } catch (err) {
  1842→      setStatus('Polling error: ' + err.message);
  1843→      return;
  1844→    }
  1845→  }
  1846→
  1847→  setStatus('Batch timed out');
  1848→}
  1849→
  1850→// Expose for inline onclick
  1851→if (typeof window !== 'undefined') {
  1852→  window.removeBatchItem = removeBatchItem;
  1853→}
