     7→/**
     8→ * Format seconds as MM:SS
     9→ * @param {number} seconds - Time in seconds
    10→ * @returns {string} Formatted time string (returns '0:00' for invalid input)
    11→ */
    12→function formatTime(seconds) {
    13→  if (typeof seconds !== 'number' || isNaN(seconds)) return '0:00';
    14→  // Handle negative values by using absolute value and prepending minus sign
    15→  const isNegative = seconds < 0;
    16→  const absSeconds = Math.abs(seconds);
    17→  const mins = Math.floor(absSeconds / 60);
    18→  const secs = Math.floor(absSeconds % 60);
    19→  const formatted = `${mins}:${secs.toString().padStart(2, '0')}`;
    20→  return isNegative ? `-${formatted}` : formatted;
    21→}
