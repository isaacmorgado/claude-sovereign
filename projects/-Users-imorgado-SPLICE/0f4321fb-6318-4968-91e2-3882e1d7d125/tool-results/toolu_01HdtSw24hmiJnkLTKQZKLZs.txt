   394â†’function applyPreset(presetId) {
   395â†’    const preset = PRESETS[presetId];
   396â†’    if (!preset) return;
   397â†’
   398â†’    // Custom preset - keep current settings
   399â†’    if (presetId === 'custom' || !preset.settings) {
   400â†’        saveSettings({ activePreset: 'custom' });
   401â†’        setStatus('Using custom settings');
   402â†’        return;
   403â†’    }
   404â†’
   405â†’    const settings = preset.settings;
   406â†’
   407â†’    // Apply to UI

   411â†’    }
   412â†’

   414â†’        ui.enableJCut.checked = settings.enableJCut || false;
   415â†’        if (ui.jcutSettings) {
   416â†’            ui.jcutSettings.classList.toggle('collapsed', !settings.enableJCut);
   417â†’        }
   418â†’    }
   419â†’
   420â†’    if (settings.jCutLeadIn !== undefined && ui.jcutLeadIn) {
   421â†’        ui.jcutLeadIn.value = Math.round(settings.jCutLeadIn * 100);
   422â†’        if (ui.jcutLeadInValue) ui.jcutLeadInValue.textContent = settings.jCutLeadIn + 's';
   423â†’    }
   424â†’
   425â†’    if (settings.jCutLeadOut !== undefined && ui.jcutLeadOut) {
   426â†’        ui.jcutLeadOut.value = Math.round(settings.jCutLeadOut * 100);
   427â†’        if (ui.jcutLeadOutValue) ui.jcutLeadOutValue.textContent = settings.jCutLeadOut + 's';
   428â†’    }
   429â†’
   430â†’    // Save merged settings
   431â†’    saveSettings({
   432â†’        activePreset: presetId,
   433â†’        sensitivity: settings.sensitivity,
   434â†’        threshold: settings.threshold,
   435â†’        minSilenceLength: settings.minSilenceLength,

   493â†’async function runDetection() {
   494â†’    if (isOperationInProgress) return;
   495â†’
   496â†’    // Check sequence
   497â†’    const hasSequence = await checkSequenceOpen();
   498â†’    if (!hasSequence) {
   499â†’        setStatus('Please open a sequence first', true);
   500â†’        return;
   501â†’    }
   502â†’
   503â†’    // Check online
   504â†’    if (!isOnline()) {
   505â†’        setStatus('Offline - Check your connection', true);
   506â†’        return;
   507â†’    }
   508â†’
   509â†’    isOperationInProgress = true;
   510â†’    ui.goBtn.disabled = true;
   511â†’    showProgress('Detecting silences...');
   512â†’
   513â†’    try {
   514â†’        const settings = getSettings();
   515â†’
   516â†’        // Get sequence info
   517â†’        const seqInfo = await jsx.call('getActiveSequence');
   518â†’        console.log('[SPLICE] Sequence info:', seqInfo);
   519â†’
   520â†’        // Store sequence info for later use
   521â†’        currentSequenceInfo = seqInfo;
   522â†’        window.spliceState.activeSequence = seqInfo;
   523â†’        if (seqInfo && seqInfo.videoPath) {
   524â†’            window.spliceState.currentVideoPath = seqInfo.videoPath;
   525â†’        }
   526â†’
   527â†’        // Export audio via JSX for silence detection
   528â†’        updateProgress(10, 'Exporting audio...');
   529â†’
   530â†’        let audioFilePath = null;
   531â†’        try {
   532â†’            // Get sequence info for audio export
   533â†’            const exportResult = await jsx.call('exportSequenceAudioForAnalysis');
   534â†’            if (exportResult && exportResult.success && exportResult.outputPath) {
   535â†’                audioFilePath = exportResult.outputPath;
   536â†’                console.log('[SPLICE] Audio exported to:', audioFilePath);
   537â†’            } else if (exportResult && exportResult.error) {
   538â†’                throw new Error(exportResult.error);
