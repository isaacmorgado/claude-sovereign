  1012→        const testLogEl = document.getElementById('test-log');
  1013→        const consoleLogEl = document.getElementById('console-output');
  1014→
  1015→        // Intercept console logs for display
  1016→        const originalLog = console.log;
  1017→        const originalWarn = console.warn;
  1018→        const originalError = console.error;
  1019→
  1020→        console.log = function (...args) {
  1021→            originalLog.apply(console, args);
  1022→            appendToConsoleOutput('log', args);
  1023→        };
  1024→
  1025→        console.warn = function (...args) {
  1026→            originalWarn.apply(console, args);
  1027→            appendToConsoleOutput('warn', args);
  1028→        };
  1029→
  1030→        console.error = function (...args) {
  1031→            originalError.apply(console, args);
  1032→            appendToConsoleOutput('error', args);
  1033→        };
  1034→
  1035→        function appendToConsoleOutput(type, args) {
  1036→            if (!consoleLogEl) return;
  1037→
  1038→            const msg = args.map(a => {
  1039→                if (typeof a === 'object') {
  1040→                    try { return JSON.stringify(a); } catch { return String(a); }
  1041→                }
  1042→                return String(a);
  1043→            }).join(' ');
  1044→
  1045→            // Skip style arguments
  1046→            if (msg.startsWith('%c')) return;
  1047→            if (msg.includes('color:')) return;
  1048→
  1049→            const line = document.createElement('div');
  1050→            line.className = type;
  1051→            line.textContent = msg.substring(0, 200) + (msg.length > 200 ? '...' : '');
  1052→            consoleLogEl.appendChild(line);
  1053→            consoleLogEl.scrollTop = consoleLogEl.scrollHeight;
  1054→        }
