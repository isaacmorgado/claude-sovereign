   340→/**
   341→ * Initialize event listeners for sequence state changes
   342→ * This provides real-time updates when sequences are activated/deactivated
   343→ */
   344→function initSequenceEventListeners() {
   345→    if (!jsx.cs) {
   346→        console.warn('[SPLICE] Cannot init sequence listeners - CSInterface not available');
   347→        return;
   348→    }
   349→
   350→    try {
   351→        // Listen for sequence activation events
   352→        jsx.cs.addEventListener('com.adobe.csxs.events.SequenceActivated', async (event) => {
   353→            debugLog('Sequence activated event received');
   354→            try {
   355→                const seqInfo = await jsx.call('getActiveSequence');
   356→                if (seqInfo && !seqInfo.error) {
   357→                    window.spliceState.activeSequence = seqInfo;
   358→                    setStatus('Sequence: ' + (seqInfo.name || 'Unknown'));
   359→                    debugLog('Active sequence updated:', seqInfo.name);
   360→                }
   361→            } catch (e) {
   362→                console.warn('[SPLICE] Failed to get sequence info on activation:', e.message);
   363→            }
   364→        });
   365→
   366→        // Listen for sequence close/deactivation
   367→        jsx.cs.addEventListener('com.adobe.csxs.events.SequenceDeactivated', () => {
   368→            debugLog('Sequence deactivated event');
   369→            window.spliceState.activeSequence = null;
   370→            setStatus('No sequence open', true);
   371→        });
   372→
   373→        // Listen for project item selection changes (includes sequences)
   374→        jsx.cs.addEventListener('com.adobe.csxs.events.SelectionChanged', async () => {
   375→            // Debounce rapid selection changes
   376→            if (window._selectionChangeTimeout) {
   377→                clearTimeout(window._selectionChangeTimeout);
   378→            }
   379→            // FIX: Increased debounce to 500ms to allow Premiere Pro state to settle
   380→            window._selectionChangeTimeout = setTimeout(async () => {
   381→                try {
   382→                    // FIX: Use retries (2 attempts, 300ms delay) to handle timing issues
   383→                    // Previous code used (0, 0) which caused race conditions
   384→                    const hasSequence = await checkSequenceOpen(2, 300);
   385→                    if (hasSequence) {
   386→                        const seqInfo = await jsx.call('getActiveSequence');
   387→                        if (seqInfo && !seqInfo.error) {
   388→                            // Check if sequence actually changed by comparing IDs
   389→                            const currentId = window.spliceState.activeSequence ? window.spliceState.activeSequence.id : null;
   390→
   391→                            // Always update if ID changed or we didn't have a sequence before
   392→                            if (currentId !== seqInfo.id) {
   393→                                window.spliceState.activeSequence = seqInfo;
   394→                                setStatus('Sequence: ' + (seqInfo.name || 'Unknown'));
   395→                                debugLog('Active sequence changed:', seqInfo.name, '(ID: ' + seqInfo.id + ')');
   396→                            }
   397→                        }
   398→                    } else {
   399→                        // If we had a sequence but now don't, likely closed or lost focus
   400→                        if (window.spliceState.activeSequence) {
   401→                            debugLog('Sequence check failed - potentially lost focus');
   402→                            // We don't verify strict nulling here to avoid flickering, 
   403→                            // as SequenceDeactivated event should handle the explicit close.
   404→                        }
   405→                    }
   406→                } catch (e) {
   407→                    // FIX: Log errors instead of silently ignoring
   408→                    console.warn('[SPLICE] Selection change handler error:', e.message);
   409→                    debugLog('Selection change failed:', e.message);
   410→                }
   411→            }, 500);
   412→        });
   413→
   414→        debugLog('Sequence event listeners initialized');
   415→    } catch (e) {
   416→        console.error('[SPLICE] Failed to init sequence event listeners:', e.message);
   417→    }
   418→}

   443→/**
   444→ * Wait for Premiere Pro API to be ready
   445→ * FIX: Ensures app.project and activeSequence APIs are available before operations
   446→ * @param {number} maxAttempts - Maximum number of attempts (default: 10)
   447→ * @param {number} delayMs - Delay between attempts in ms (default: 500)
   448→ * @returns {Promise<boolean>} true if Premiere Pro is ready
   449→ */
   450→async function waitForPremiereReady(maxAttempts = 10, delayMs = 500) {
   451→    console.log('[SPLICE] waitForPremiereReady: Starting with maxAttempts=' + maxAttempts);
   452→
   453→    for (let i = 0; i < maxAttempts; i++) {
   454→        try {
   455→            console.log('[SPLICE] waitForPremiereReady: Attempt ' + (i + 1) + ' of ' + maxAttempts);
   456→            const ready = await jsx.call('checkPremiereReady');
   457→
   458→            if (ready === true || ready === 'true') {
   459→                console.log('[SPLICE] waitForPremiereReady: ✓ Premiere Pro is ready');
   460→                return true;
   461→            }
   462→
   463→            console.log('[SPLICE] waitForPremiereReady: Not ready yet, waiting ' + delayMs + 'ms...');
   464→            await new Promise(r => setTimeout(r, delayMs));
   465→        } catch (e) {
   466→            console.warn('[SPLICE] waitForPremiereReady: Attempt ' + (i + 1) + ' failed:', e.message);
   467→            await new Promise(r => setTimeout(r, delayMs));
   468→        }
   469→    }
   470→
   471→    console.error('[SPLICE] waitForPremiereReady: Failed after ' + maxAttempts + ' attempts');
   472→    return false;
   473→}

   475→/**
   476→ * Check if a sequence is currently open in the Timeline
   477→ * Enhanced with logging and retry logic
   478→ * @param {number} retries - Number of retry attempts (default: 2)
   479→ * @param {number} delayMs - Delay between retries in ms (default: 300)
   480→ * @returns {Promise<boolean>} true if sequence is open
   481→ */
   482→async function checkSequenceOpen(retries = 2, delayMs = 300) {
   483→    console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC START ==========');
   484→    console.log('[SPLICE] checkSequenceOpen: retries=' + retries + ', delayMs=' + delayMs);
   485→    console.log('[SPLICE] checkSequenceOpen: window.spliceState.activeSequence =', window.spliceState.activeSequence);
   486→
   487→    for (let attempt = 0; attempt <= retries; attempt++) {
   488→        try {
   489→            console.log('[SPLICE] checkSequenceOpen: Attempt ' + (attempt + 1) + ' of ' + (retries + 1));
   490→            console.log('[SPLICE] checkSequenceOpen: Calling jsx.call("checkSequenceOpen")...');
   491→            const result = await jsx.call('checkSequenceOpen');
   492→            console.log('[SPLICE] checkSequenceOpen: JSX call returned');
   493→            console.log('[SPLICE] checkSequenceOpen: result =', result, '(type:', typeof result + ')');
   494→            debugLog('checkSequenceOpen attempt', attempt + 1, 'result:', result, 'type:', typeof result);
   495→
   496→            if (result === true || result === 'true') {
   497→                console.log('[SPLICE] checkSequenceOpen: Returning TRUE - sequence is open');
   498→                console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (TRUE) ==========');
   499→                return true;
   500→            }
   501→
   502→            // If false, wait and retry (sequence might be loading)
   503→            if (attempt < retries) {
   504→                console.log('[SPLICE] checkSequenceOpen: No sequence found, retrying in', delayMs, 'ms...');
   505→                debugLog('checkSequenceOpen: No sequence found, retrying in', delayMs, 'ms...');
   506→                await new Promise(r => setTimeout(r, delayMs));
   507→            }
   508→        } catch (error) {
   509→            console.error('[SPLICE] checkSequenceOpen: Attempt', attempt + 1, 'failed with exception:', error.message);
   510→            console.error('[SPLICE] checkSequenceOpen: Error stack:', error.stack);
   511→
   512→            // On last attempt, return false but log the actual error
   513→            if (attempt === retries) {
   514→                console.error('[SPLICE] checkSequenceOpen: All attempts failed. Last error:', error.message);
   515→                // Store error for diagnostics
   516→                window.spliceState.lastSequenceCheckError = error.message;
   517→                console.log('[SPLICE] checkSequenceOpen: Stored error in window.spliceState.lastSequenceCheckError:', error.message);
   518→                console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (ERROR) ==========');
   519→                return false;
   520→            }
   521→
   522→            // Wait before retry
   523→            console.log('[SPLICE] checkSequenceOpen: Waiting', delayMs, 'ms before retry...');
   524→            await new Promise(r => setTimeout(r, delayMs));
   525→        }
   526→    }
   527→    console.log('[SPLICE] checkSequenceOpen: All attempts completed without finding sequence');
   528→    console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (FALSE) ==========');
   529→    return false;
   530→}

   532→/**
   533→ * Get active sequence info with retry logic
   534→ * FIX: Enhanced version with automatic retries on failure
   535→ * @param {number} retries - Number of retry attempts (default: 3)
   536→ * @param {number} delayMs - Delay between retries in ms (default: 300)
   537→ * @returns {Promise<Object|null>} Sequence info or null on failure
   538→ */
   539→async function getActiveSequenceWithRetry(retries = 3, delayMs = 300) {
   540→    console.log('[SPLICE] getActiveSequenceWithRetry: Starting with retries=' + retries);
   541→
   542→    for (let attempt = 0; attempt <= retries; attempt++) {
   543→        try {
   544→            console.log('[SPLICE] getActiveSequenceWithRetry: Attempt ' + (attempt + 1) + ' of ' + (retries + 1));
   545→            const seqInfo = await jsx.call('getActiveSequence');
   546→
   547→            if (seqInfo && !seqInfo.error) {
   548→                console.log('[SPLICE] getActiveSequenceWithRetry: ✓ Got sequence:', seqInfo.name);
   549→                return seqInfo;
   550→            }
   551→
   552→            if (seqInfo && seqInfo.error) {
   553→                console.warn('[SPLICE] getActiveSequenceWithRetry: Error from JSX:', seqInfo.error);
   554→            }
   555→
   556→            // Retry if not last attempt
   557→            if (attempt < retries) {
   558→                console.log('[SPLICE] getActiveSequenceWithRetry: No sequence found, retrying in ' + delayMs + 'ms...');
   559→                await new Promise(r => setTimeout(r, delayMs));
   560→            }
   561→        } catch (error) {
   562→            console.error('[SPLICE] getActiveSequenceWithRetry: Attempt failed:', error.message);
   563→            if (attempt === retries) {
   564→                console.error('[SPLICE] getActiveSequenceWithRetry: All attempts failed');
   565→                return null;
   566→            }
   567→            await new Promise(r => setTimeout(r, delayMs));
   568→        }
   569→    }
   570→    console.log('[SPLICE] getActiveSequenceWithRetry: Returning null after all attempts');
   571→    return null;
   572→}

   783→function applyPreset(presetId) {
   784→    const preset = PRESETS[presetId];
   785→    if (!preset) return;
   786→
   787→    // Custom preset - keep current settings
   788→    if (presetId === 'custom' || !preset.settings) {
   789→        saveSettings({ activePreset: 'custom' });
   790→        setStatus('Using custom settings');
   791→        return;
   792→    }
   793→
   794→    const settings = preset.settings;
   795→
   796→    // Apply to UI
   797→    if (ui.sensitivitySlider && settings.sensitivity !== undefined) {
   798→        ui.sensitivitySlider.value = settings.sensitivity;
   799→        if (ui.sensitivityValue) ui.sensitivityValue.textContent = settings.sensitivity;
   800→    }
   801→
   802→    if (ui.enableJCut) {
   803→        ui.enableJCut.checked = settings.enableJCut || false;
   804→        if (ui.jcutSettings) {
   805→            ui.jcutSettings.classList.toggle('collapsed', !settings.enableJCut);
   806→        }
   807→    }
   808→
   809→    if (settings.jCutLeadIn !== undefined && ui.jcutLeadIn) {
   810→        ui.jcutLeadIn.value = Math.round(settings.jCutLeadIn * 100);
   811→        if (ui.jcutLeadInValue) ui.jcutLeadInValue.textContent = settings.jCutLeadIn + 's';
   812→    }
   813→
   814→    if (settings.jCutLeadOut !== undefined && ui.jcutLeadOut) {
   815→        ui.jcutLeadOut.value = Math.round(settings.jCutLeadOut * 100);
   816→        if (ui.jcutLeadOutValue) ui.jcutLeadOutValue.textContent = settings.jCutLeadOut + 's';
   817→    }
   818→
   819→    // Save merged settings
   820→    saveSettings({
   821→        activePreset: presetId,
   822→        sensitivity: settings.sensitivity,
   823→        threshold: settings.threshold,
   824→        minSilenceLength: settings.minSilenceLength,
   825→        paddingStart: settings.paddingStart,
   826→        paddingEnd: settings.paddingEnd,
   827→        enableTakesDetection: settings.enableTakesDetection,
   828→        enableJCut: settings.enableJCut || false,
   829→        jcutLeadIn: settings.jCutLeadIn || 0.3,
   830→        jcutLeadOut: settings.jCutLeadOut || 0.2
   831→    });
   832→
   833→    setStatus(`Applied ${preset.name} preset`);
   834→}

   892→async function runDetection() {
   893→    if (isOperationInProgress) return;
   894→
   895→    // Check sequence with enhanced error messaging
   896→    const hasSequence = await checkSequenceOpen();
   897→    if (!hasSequence) {
   898→        // Check if there was a communication error vs just no sequence
   899→        const lastError = window.spliceState.lastSequenceCheckError;
   900→        if (lastError && (lastError.includes('CEP') || lastError.includes('communication') || lastError.includes('evalScript'))) {
   901→            setStatus('Premiere Pro connection issue - try reopening panel', true);
   902→            console.error('[SPLICE] runDetection: JSX communication failure:', lastError);
   903→        } else {
   904→            setStatus('Please open a sequence in the Timeline first', true);
   905→            debugLog('runDetection: No active sequence in Timeline');
   906→        }
   907→        // Clear the error for next check
   908→        window.spliceState.lastSequenceCheckError = null;
   909→        return;
   910→    }
   911→
   912→    // Check online
   913→    if (!isOnline()) {
   914→        setStatus('Offline - Check your connection', true);
   915→        return;
   916→    }
   917→
   918→    isOperationInProgress = true;
   919→    ui.goBtn.disabled = true;
   920→    showProgress('Detecting silences...');
   921→
   922→    try {
   923→        const settings = getSettings();
   924→
   925→        // FIX: Get sequence info with retry logic
   926→        const seqInfo = await getActiveSequenceWithRetry(3, 300);
   927→        debugLog('Sequence info:', seqInfo);
   928→
   929→        if (!seqInfo) {
   930→            console.warn('[SPLICE] runDetection: getActiveSequenceWithRetry failed after retries');
   931→            setStatus('Could not access active sequence info', true);
   932→            ui.goBtn.disabled = false;
   933→            isOperationInProgress = false;
   934→            hideProgress();
   935→            return;
   936→        }
   937→
   938→        // Store sequence info for later use
   939→        currentSequenceInfo = seqInfo;
   940→        window.spliceState.activeSequence = seqInfo;
   941→        if (seqInfo && seqInfo.videoPath) {
   942→            window.spliceState.currentVideoPath = seqInfo.videoPath;
   943→        }
   944→
   945→        // Export audio via JSX for silence detection
   946→        updateProgress(10, 'Exporting audio...');
   947→
   948→        let audioFilePath = null;
   949→        try {
   950→            // FIX: CEP-CRIT-010 - Main.js Export Timeout
   951→            // Use LONG timeout (120s) for export operations
   952→            const exportResult = await jsx.callWithTimeout(
   953→                'exportSequenceAudioForAnalysis',
   954→                [],
   955→                JSX_TIMEOUTS.LONG
   956→            );
   957→            if (exportResult && exportResult.success && exportResult.outputPath) {
   958→                audioFilePath = exportResult.outputPath;
   959→                debugLog('Audio exported to:', audioFilePath);
   960→            } else if (exportResult && exportResult.error) {
   961→                throw new Error(exportResult.error);
   962→            }
   963→        } catch (exportError) {
   964→            console.warn('[SPLICE] Audio export failed, trying fallback:', exportError.message);
   965→            // Try to get audio from first clip in timeline
   966→            try {
   967→                const clipInfo = await jsx.call('getFirstClipAudioPath');
   968→                if (clipInfo && clipInfo.path) {
   969→                    audioFilePath = clipInfo.path;
   970→                    debugLog('Using clip audio path:', audioFilePath);
   971→                }
   972→            } catch (clipError) {
   973→                console.warn('[SPLICE] Fallback audio path failed:', clipError.message);
   974→            }
   975→        }
   976→
   977→        if (!audioFilePath) {
   978→            throw new Error('Could not export or locate audio file. Please ensure sequence has audio tracks.');
   979→        }
   980→
   981→        // Store audio path globally for other modules
   982→        currentAudioPath = audioFilePath;
   983→        window.spliceState.audioPath = audioFilePath;
   984→
   985→        // Call backend for silence detection with audio file path
   986→        updateProgress(30, 'Detecting silences...');
   987→        const silenceResponse = await fetchWithTimeout(
   988→            `${getBackendUrl()}/silences-rms`,
   989→            {
   990→                method: 'POST',
   991→                headers: getAuthHeaders(),
   992→                body: JSON.stringify({
   993→                    wavPath: audioFilePath,
   994→                    sensitivity: settings.sensitivity,
   995→                    minSilenceLength: settings.minSilenceLength || 0.5
   996→                })
   997→            }
   998→        );
   999→
  1000→        if (!silenceResponse.ok) {
  1001→            throw new Error(await parseErrorResponse(silenceResponse));
  1002→        }
  1003→
  1004→        const silenceData = await silenceResponse.json();
  1005→        previewSilences = silenceData.silences || [];
  1006→
  1007→        // Detect takes if enabled
  1008→        if (settings.enableTakesDetection) {
  1009→            updateProgress(60, 'Detecting takes...');
  1010→            try {
  1011→                const analyzeResponse = await fetchWithTimeout(
  1012→                    `${getBackendUrl()}/analyze`,
  1013→                    {
  1014→                        method: 'POST',
  1015→                        headers: getAuthHeaders(),
  1016→                        body: JSON.stringify({
  1017→                            wavPath: audioFilePath,
  1018→                            detectTakes: true
  1019→                        })
  1020→                    }
  1021→                );
  1022→
  1023→                if (analyzeResponse.ok) {
  1024→                    const analyzeData = await analyzeResponse.json();
  1025→                    previewTakes = analyzeData.takes || [];
  1026→
  1027→                    // Store transcript for captions/music modules
  1028→                    if (analyzeData.transcript) {
  1029→                        currentTranscript = analyzeData.transcript;
  1030→                        window.currentTranscript = analyzeData.transcript;
  1031→                        window.spliceState.lastTranscript = analyzeData.transcript;
  1032→                    }
  1033→                } else {
  1034→                    // Handle specific error codes
  1035→                    const errorMsg = await parseErrorResponse(analyzeResponse);
  1036→                    if (analyzeResponse.status === 402) {
  1037→                        console.warn('[SPLICE] Takes detection skipped: insufficient credits');
  1038→                        setStatus('Takes detection skipped: insufficient credits', true);
  1039→                    } else if (analyzeResponse.status === 401) {
  1040→                        console.warn('[SPLICE] Takes detection skipped: authentication required');
  1041→                    } else {
  1042→                        console.warn('[SPLICE] Takes detection failed:', errorMsg);
  1043→                    }
  1044→                }
  1045→            } catch (e) {
  1046→                console.warn('[SPLICE] Takes detection failed:', e.message);
  1047→            }
  1048→        }
  1049→
  1050→        // Detect profanity if enabled
  1051→        if (settings.enableProfanity) {
  1052→            updateProgress(65, 'Detecting profanity...');
  1053→            try {
  1054→                const profanityResponse = await fetchWithTimeout(
  1055→                    `${getBackendUrl()}/profanity`,
  1056→                    {
  1057→                        method: 'POST',
  1058→                        headers: getAuthHeaders(),
  1059→                        body: JSON.stringify({
  1060→                            wavPath: audioFilePath,
  1061→                            language: settings.profanityLanguage || 'en',
  1062→                            bleepType: settings.bleepType || 'standard'
  1063→                        })
  1064→                    }
  1065→                );
  1066→
  1067→                if (profanityResponse.ok) {
  1068→                    const profanityData = await profanityResponse.json();
  1069→                    // Backend returns 'words' array, not 'instances'
  1070→                    currentProfanity = profanityData.words || profanityData.instances || [];
  1071→                    if (currentProfanity.length > 0) {
  1072→                        debugLog(`Found ${currentProfanity.length} profanity instances`);
  1073→                    }
  1074→                } else {
  1075→                    console.warn('[SPLICE] Profanity detection failed:', await parseErrorResponse(profanityResponse));
  1076→                }
  1077→            } catch (e) {
  1078→                console.warn('[SPLICE] Profanity detection failed:', e.message);
  1079→            }
  1080→        }
  1081→
  1082→        // Detect chapters if enabled
  1083→        if (settings.enableChapters) {
  1084→            updateProgress(80, 'Detecting chapters...');
  1085→            try {
  1086→                const chapterResponse = await fetchWithTimeout(
  1087→                    `${getBackendUrl()}/chapters`,
  1088→                    {
  1089→                        method: 'POST',
  1090→                        headers: getAuthHeaders(),
  1091→                        body: JSON.stringify({
  1092→                            wavPath: audioFilePath,
  1093→                            transcript: currentTranscript,
  1094→                            maxChapters: settings.maxChapters || 10,
  1095→                            minChapterLength: settings.minChapterLength || 60
  1096→                        })
  1097→                    }
  1098→                );
  1099→
  1100→                if (chapterResponse.ok) {
  1101→                    const chapterData = await chapterResponse.json();
  1102→                    currentChapters = chapterData.chapters || [];
  1103→                } else {
  1104→                    const errorMsg = await parseErrorResponse(chapterResponse);
  1105→                    console.warn('[SPLICE] Chapter detection failed:', errorMsg);
  1106→                    // Don't block workflow, just notify user
  1107→                    setStatus('Chapter detection skipped: ' + errorMsg, true);
  1108→                }
  1109→            } catch (e) {
  1110→                console.warn('[SPLICE] Chapter detection failed:', e);
  1111→                setStatus('Chapter detection failed - continuing without chapters', true);
  1112→            }
  1113→        }
  1114→
  1115→        // Detect zoom points if enabled
  1116→        if (settings.enableZoom) {
  1117→            updateProgress(90, 'Detecting zoom points...');
  1118→            try {
  1119→                const zoomResponse = await fetchWithTimeout(
  1120→                    `${getBackendUrl()}/zoom`,
  1121→                    {
  1122→                        method: 'POST',
  1123→                        headers: getAuthHeaders(),
  1124→                        body: JSON.stringify({
  1125→                            wavPath: audioFilePath,

  1454→    try {

  1476→            `${getBackendUrl()}/cut-list`,

  1482→                    sourcePath: currentSequenceInfo?.treePath || currentSequenceInfo?.name || 'Untitled',
  1483→                    duration: duration,

  1485→                    takes: previewTakes,
  1486→                    enableTakes: settings.enableTakesDetection,
  1487→                    jCutOffset: settings.enableJCut ? settings.jcutLeadIn : 0,
  1488→                    lCutOffset: settings.enableJCut ? settings.jcutLeadOut : 0
  1489→                })
  1490→            }
  1491→        );
  1492→
  1493→        if (!cutListResponse.ok) {
  1494→            throw new Error(await parseErrorResponse(cutListResponse));

  1497→        const cutList = await cutListResponse.json();
  1498→
  1499→        // Build sequence via JSX
