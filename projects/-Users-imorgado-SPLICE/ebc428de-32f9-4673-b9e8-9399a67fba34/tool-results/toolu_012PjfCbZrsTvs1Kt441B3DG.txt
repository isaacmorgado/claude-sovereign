    87→// Countdown timer component
    88→function UrgencyCountdown() {
    89→  const [timeLeft, setTimeLeft] = useState({ days: 0, hours: 0, minutes: 0, seconds: 0 });
    90→
    91→  useEffect(() => {
    92→    // Set end date to end of current month
    93→    const getEndDate = () => {
    94→      const now = new Date();
    95→      return new Date(now.getFullYear(), now.getMonth() + 1, 0, 23, 59, 59);
    96→    };
    97→
    98→    const calculateTimeLeft = () => {
    99→      const difference = getEndDate().getTime() - new Date().getTime();
   100→      if (difference > 0) {
   101→        return {
   102→          days: Math.floor(difference / (1000 * 60 * 60 * 24)),
   103→          hours: Math.floor((difference / (1000 * 60 * 60)) % 24),
   104→          minutes: Math.floor((difference / 1000 / 60) % 60),
   105→          seconds: Math.floor((difference / 1000) % 60),
   106→        };
   107→      }
   108→      return { days: 0, hours: 0, minutes: 0, seconds: 0 };
   109→    };
   110→
   111→    setTimeLeft(calculateTimeLeft());
   112→    const timer = setInterval(() => setTimeLeft(calculateTimeLeft()), 1000);
   113→    return () => clearInterval(timer);
   114→  }, []);
   115→
   116→  return (
   117→    <div className="flex items-center justify-center gap-3 text-sm">
   118→      {[
   119→        { value: timeLeft.days, label: "Days" },
   120→        { value: timeLeft.hours, label: "Hrs" },
   121→        { value: timeLeft.minutes, label: "Min" },
   122→        { value: timeLeft.seconds, label: "Sec" },
   123→      ].map((item, i) => (
   124→        <div key={item.label} className="flex items-center gap-1">
   125→          <div className="bg-[#00D4FF]/20 text-[#00D4FF] font-mono font-bold px-2 py-1 rounded">
   126→            {String(item.value).padStart(2, "0")}
   127→          </div>
   128→          <span className="text-gray-400 text-xs">{item.label}</span>
   129→          {i < 3 && <span className="text-gray-600 ml-1">:</span>}
   130→        </div>
   131→      ))}
   132→    </div>
   133→  );
   134→}
