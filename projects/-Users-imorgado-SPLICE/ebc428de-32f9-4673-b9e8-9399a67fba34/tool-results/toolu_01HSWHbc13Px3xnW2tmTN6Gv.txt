     1→"use client";
     2→
     3→import { useState, useEffect } from "react";
     4→import Link from "next/link";
     5→import Image from "next/image";
     6→import { usePathname } from "next/navigation";
     7→import { Button } from "@/components/ui/button";
     8→import { Menu, X } from "lucide-react";
     9→import { motion, AnimatePresence } from "framer-motion";
    10→
    11→export function Header() {
    12→  const [isScrolled, setIsScrolled] = useState(false);
    13→  const [isMobileMenuOpen, setIsMobileMenuOpen] = useState(false);
    14→  const pathname = usePathname();
    15→
    16→  useEffect(() => {
    17→    const handleScroll = () => {
    18→      setIsScrolled(window.scrollY > 10);
    19→    };
    20→    window.addEventListener("scroll", handleScroll);
    21→    return () => window.removeEventListener("scroll", handleScroll);
    22→  }, []);
    23→
    24→  const scrollToSection = (e: React.MouseEvent<HTMLAnchorElement>, href: string) => {
    25→    // Only handle hash links on the home page
    26→    if (href.startsWith("#")) {
    27→      e.preventDefault();
    28→
    29→      // If not on home page, navigate to home first
    30→      if (pathname !== "/") {
    31→        window.location.href = "/" + href;
    32→        return;
    33→      }
    34→
    35→      const element = document.querySelector(href);
    36→      if (element) {
    37→        const headerOffset = 80;
    38→        const elementPosition = element.getBoundingClientRect().top;
    39→        const offsetPosition = elementPosition + window.pageYOffset - headerOffset;
    40→
    41→        window.scrollTo({
    42→          top: offsetPosition,
    43→          behavior: "smooth"
    44→        });
    45→      }
    46→      setIsMobileMenuOpen(false);
    47→    }
    48→  };
    49→
    50→  const navLinks = [
    51→    { href: "#features", label: "Features" },
    52→    { href: "#comparison", label: "Compare" },
    53→    { href: "#pricing", label: "Pricing" },
    54→    { href: "/download", label: "Download" },
    55→    { href: "/docs", label: "Docs" },
    56→  ];
    57→
    58→  return (
    59→    <header
    60→      className={`fixed top-0 left-0 right-0 z-50 transition-all duration-300 ${isScrolled
    61→        ? "bg-black/80 backdrop-blur-md border-b border-white/10 py-3 shadow-lg"
    62→        : "bg-transparent py-4"
    63→        }`}
    64→    >
    65→      <div className="container mx-auto px-4 flex items-center justify-between">
    66→        {/* Logo */}
    67→        <Link href="/" className="flex items-center gap-2">
    68→          <div className="w-8 h-8 neon-glow">
    69→            <Image
    70→              src="/images/logo.svg"
    71→              alt="SPLICE Logo"
    72→              width={32}
    73→              height={32}
    74→              className="w-full h-full object-contain"
    75→            />
    76→          </div>
    77→          <span className="font-bold text-xl text-white">SPLICE</span>
    78→        </Link>
    79→
    80→        {/* Desktop Navigation */}
    81→        <nav className="hidden md:flex items-center gap-8">
    82→          {navLinks.map((link) => (
    83→            <a
    84→              key={link.href}
    85→              href={link.href}
    86→              onClick={(e) => scrollToSection(e, link.href)}
    87→              className="text-sm font-medium text-muted-foreground hover:text-foreground transition-colors cursor-pointer"
    88→            >
    89→              {link.label}
    90→            </a>
    91→          ))}
    92→        </nav>
    93→
    94→        {/* Desktop CTAs */}
    95→        <div className="hidden md:flex items-center gap-4">
    96→          <Button
    97→            variant="outline"
    98→            size="sm"
    99→            className="text-white border-white/20 hover:bg-white/10 hover:text-white"
   100→            asChild
   101→          >
   102→            <Link href="/login">Log In</Link>
   103→          </Button>
   104→          <Button
   105→            size="sm"
   106→            className="bg-[#00D4FF] hover:bg-[#00B8E6] text-black font-semibold btn-glow"
   107→            asChild
   108→          >
   109→            <a
   110→              href="#pricing"
   111→              onClick={(e) => scrollToSection(e, "#pricing")}
   112→            >
   113→              Start Free Trial
   114→            </a>
   115→          </Button>
   116→        </div>
   117→
   118→        {/* Mobile Menu Button */}
   119→        <button
   120→          className="md:hidden p-2"
   121→          onClick={() => setIsMobileMenuOpen(!isMobileMenuOpen)}
   122→          aria-label="Toggle menu"
   123→        >
   124→          {isMobileMenuOpen ? <X size={24} /> : <Menu size={24} />}
   125→        </button>
   126→      </div>
   127→
   128→      {/* Mobile Menu */}
   129→      <AnimatePresence>
   130→        {isMobileMenuOpen && (
   131→          <motion.div
   132→            initial={{ opacity: 0, height: 0 }}
   133→            animate={{ opacity: 1, height: "auto" }}
   134→            exit={{ opacity: 0, height: 0 }}
   135→            className="md:hidden glass border-t border-border/50 bg-black/95 backdrop-blur-xl"
   136→          >
   137→            <nav className="container mx-auto px-4 py-4 flex flex-col gap-4">
   138→              {navLinks.map((link) => (
   139→                <a
   140→                  key={link.href}
   141→                  href={link.href}
   142→                  onClick={(e) => scrollToSection(e, link.href)}
   143→                  className="text-sm font-medium py-2 cursor-pointer"
   144→                >
   145→                  {link.label}
   146→                </a>
   147→              ))}
   148→              <div className="flex flex-col gap-2 pt-4 border-t border-border/50">
   149→                <Button variant="outline" size="sm" asChild>
   150→                  <Link href="/login">Log In</Link>
   151→                </Button>
   152→                <Button
   153→                  size="sm"
   154→                  className="bg-[#00D4FF] hover:bg-[#00B8E6] text-black font-semibold"
   155→                  onClick={(e) => {
   156→                    e.preventDefault();
   157→                    const element = document.querySelector("#pricing");
   158→                    if (element) {
   159→                      element.scrollIntoView({ behavior: "smooth" });
   160→                    }
   161→                    setIsMobileMenuOpen(false);
   162→                  }}
   163→                >
   164→                  Start Free Trial
   165→                </Button>
   166→              </div>
   167→            </nav>
   168→          </motion.div>
   169→        )}
   170→      </AnimatePresence>
   171→    </header>
   172→  );
   173→}
