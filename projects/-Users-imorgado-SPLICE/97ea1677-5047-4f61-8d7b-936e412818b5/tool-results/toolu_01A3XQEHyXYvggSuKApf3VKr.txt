   368→/**
   369→ * Initialize event listeners for sequence state changes
   370→ * This provides real-time updates when sequences are activated/deactivated
   371→ */
   372→function initSequenceEventListeners() {
   373→    if (!jsx.cs) {
   374→        console.warn('[SPLICE] Cannot init sequence listeners - CSInterface not available');
   375→        return;
   376→    }
   377→
   378→    try {
   379→        // Listen for sequence activation events
   380→        jsx.cs.addEventListener('com.adobe.csxs.events.SequenceActivated', async (event) => {
   381→            debugLog('Sequence activated event received');
   382→            try {
   383→                const seqInfo = await jsx.call('getActiveSequence');
   384→                if (seqInfo && !seqInfo.error) {
   385→                    window.spliceState.activeSequence = seqInfo;
   386→                    setStatus('Sequence: ' + (seqInfo.name || 'Unknown'));
   387→                    debugLog('Active sequence updated:', seqInfo.name);
   388→                }
   389→            } catch (e) {
   390→                console.warn('[SPLICE] Failed to get sequence info on activation:', e.message);
   391→            }
   392→        });
   393→
   394→        // Listen for sequence close/deactivation
   395→        jsx.cs.addEventListener('com.adobe.csxs.events.SequenceDeactivated', () => {
   396→            debugLog('Sequence deactivated event');
   397→            window.spliceState.activeSequence = null;
   398→            setStatus('No sequence open', true);
   399→        });
   400→
   401→        // Listen for project item selection changes (includes sequences)
   402→        jsx.cs.addEventListener('com.adobe.csxs.events.SelectionChanged', async () => {
   403→            // Debounce rapid selection changes
   404→            if (window._selectionChangeTimeout) {
   405→                clearTimeout(window._selectionChangeTimeout);
   406→            }
   407→            // FIX: Increased debounce to 500ms to allow Premiere Pro state to settle
   408→            window._selectionChangeTimeout = setTimeout(async () => {
   409→                try {
   410→                    // FIX: Use retries (2 attempts, 300ms delay) to handle timing issues
   411→                    // Previous code used (0, 0) which caused race conditions
   412→                    const hasSequence = await checkSequenceOpen(2, 300);
   413→                    if (hasSequence) {
   414→                        const seqInfo = await jsx.call('getActiveSequence');
   415→                        if (seqInfo && !seqInfo.error) {
   416→                            // Check if sequence actually changed by comparing IDs
   417→                            const currentId = window.spliceState.activeSequence ? window.spliceState.activeSequence.id : null;
   418→
   419→                            // Always update if ID changed or we didn't have a sequence before
   420→                            if (currentId !== seqInfo.id) {
   421→                                window.spliceState.activeSequence = seqInfo;
   422→                                setStatus('Sequence: ' + (seqInfo.name || 'Unknown'));
   423→                                debugLog('Active sequence changed:', seqInfo.name, '(ID: ' + seqInfo.id + ')');
   424→                            }
   425→                        }
   426→                    } else {
   427→                        // If we had a sequence but now don't, likely closed or lost focus
   428→                        if (window.spliceState.activeSequence) {
   429→                            debugLog('Sequence check failed - potentially lost focus');
   430→                            // We don't verify strict nulling here to avoid flickering, 
   431→                            // as SequenceDeactivated event should handle the explicit close.
   432→                        }
   433→                    }
   434→                } catch (e) {
   435→                    // FIX: Log errors instead of silently ignoring
   436→                    console.warn('[SPLICE] Selection change handler error:', e.message);
   437→                    debugLog('Selection change failed:', e.message);
   438→                }
   439→            }, 500);
   440→        });
   441→
   442→        debugLog('Sequence event listeners initialized');
   443→    } catch (e) {
   444→        console.error('[SPLICE] Failed to init sequence event listeners:', e.message);
   445→    }
   446→}

   471→/**
   472→ * Wait for Premiere Pro API to be ready
   473→ * FIX: Ensures app.project and activeSequence APIs are available before operations
   474→ * @param {number} maxAttempts - Maximum number of attempts (default: 10)
   475→ * @param {number} delayMs - Delay between attempts in ms (default: 500)
   476→ * @returns {Promise<boolean>} true if Premiere Pro is ready
   477→ */
   478→async function waitForPremiereReady(maxAttempts = 10, delayMs = 500) {
   479→    console.log('[SPLICE] waitForPremiereReady: Starting with maxAttempts=' + maxAttempts);
   480→
   481→    for (let i = 0; i < maxAttempts; i++) {
   482→        try {
   483→            console.log('[SPLICE] waitForPremiereReady: Attempt ' + (i + 1) + ' of ' + maxAttempts);
   484→            const ready = await jsx.call('checkPremiereReady');
   485→
   486→            if (ready === true || ready === 'true') {
   487→                console.log('[SPLICE] waitForPremiereReady: ✓ Premiere Pro is ready');
   488→                return true;
   489→            }
   490→
   491→            console.log('[SPLICE] waitForPremiereReady: Not ready yet, waiting ' + delayMs + 'ms...');
   492→            await new Promise(r => setTimeout(r, delayMs));
   493→        } catch (e) {
   494→            console.warn('[SPLICE] waitForPremiereReady: Attempt ' + (i + 1) + ' failed:', e.message);
   495→            await new Promise(r => setTimeout(r, delayMs));
   496→        }
   497→    }
   498→
   499→    console.error('[SPLICE] waitForPremiereReady: Failed after ' + maxAttempts + ' attempts');
   500→    return false;
   501→}

   503→/**
   504→ * Check if a sequence is currently open in the Timeline
   505→ * Enhanced with logging and retry logic
   506→ * @param {number} retries - Number of retry attempts (default: 2)
   507→ * @param {number} delayMs - Delay between retries in ms (default: 300)
   508→ * @returns {Promise<boolean>} true if sequence is open
   509→ */
   510→async function checkSequenceOpen(retries = 2, delayMs = 300) {
   511→    console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC START ==========');
   512→    console.log('[SPLICE] checkSequenceOpen: retries=' + retries + ', delayMs=' + delayMs);
   513→    console.log('[SPLICE] checkSequenceOpen: window.spliceState.activeSequence =', window.spliceState.activeSequence);
   514→
   515→    for (let attempt = 0; attempt <= retries; attempt++) {
   516→        try {
   517→            console.log('[SPLICE] checkSequenceOpen: Attempt ' + (attempt + 1) + ' of ' + (retries + 1));
   518→            console.log('[SPLICE] checkSequenceOpen: Calling jsx.call("checkSequenceOpen")...');
   519→            const result = await jsx.call('checkSequenceOpen');
   520→            console.log('[SPLICE] checkSequenceOpen: JSX call returned');
   521→            console.log('[SPLICE] checkSequenceOpen: result =', result, '(type:', typeof result + ')');
   522→            debugLog('checkSequenceOpen attempt', attempt + 1, 'result:', result, 'type:', typeof result);
   523→
   524→            if (result === true || result === 'true') {
   525→                console.log('[SPLICE] checkSequenceOpen: Returning TRUE - sequence is open');
   526→                console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (TRUE) ==========');
   527→                return true;
   528→            }
   529→
   530→            // If false, wait and retry (sequence might be loading)
   531→            if (attempt < retries) {
   532→                console.log('[SPLICE] checkSequenceOpen: No sequence found, retrying in', delayMs, 'ms...');
   533→                debugLog('checkSequenceOpen: No sequence found, retrying in', delayMs, 'ms...');
   534→                await new Promise(r => setTimeout(r, delayMs));
   535→            }
   536→        } catch (error) {
   537→            console.error('[SPLICE] checkSequenceOpen: Attempt', attempt + 1, 'failed with exception:', error.message);
   538→            console.error('[SPLICE] checkSequenceOpen: Error stack:', error.stack);
   539→
   540→            // On last attempt, return false but log the actual error
   541→            if (attempt === retries) {
   542→                console.error('[SPLICE] checkSequenceOpen: All attempts failed. Last error:', error.message);
   543→                // Store error for diagnostics
   544→                window.spliceState.lastSequenceCheckError = error.message;
   545→                console.log('[SPLICE] checkSequenceOpen: Stored error in window.spliceState.lastSequenceCheckError:', error.message);
   546→                console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (ERROR) ==========');
   547→                return false;
   548→            }
   549→
   550→            // Wait before retry
   551→            console.log('[SPLICE] checkSequenceOpen: Waiting', delayMs, 'ms before retry...');
   552→            await new Promise(r => setTimeout(r, delayMs));
   553→        }
   554→    }
   555→    console.log('[SPLICE] checkSequenceOpen: All attempts completed without finding sequence');
   556→    console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (FALSE) ==========');
   557→    return false;
   558→}

   561→ * Get active sequence info with retry logic
   562→ * FIX: Enhanced version with automatic retries on failure
   563→ * @param {number} retries - Number of retry attempts (default: 3)
   564→ * @param {number} delayMs - Delay between retries in ms (default: 300)
   565→ * @returns {Promise<Object|null>} Sequence info or null on failure
   566→ */
   567→async function getActiveSequenceWithRetry(retries = 3, delayMs = 300) {
   568→    console.log('[SPLICE] getActiveSequenceWithRetry: Starting with retries=' + retries);
   569→
   570→    for (let attempt = 0; attempt <= retries; attempt++) {
   571→        try {
   572→            console.log('[SPLICE] getActiveSequenceWithRetry: Attempt ' + (attempt + 1) + ' of ' + (retries + 1));
   573→            const seqInfo = await jsx.call('getActiveSequence');
   574→
   575→            if (seqInfo && !seqInfo.error) {
   576→                console.log('[SPLICE] getActiveSequenceWithRetry: ✓ Got sequence:', seqInfo.name);
   577→                return seqInfo;
   578→            }
   579→
   580→            if (seqInfo && seqInfo.error) {
   581→                console.warn('[SPLICE] getActiveSequenceWithRetry: Error from JSX:', seqInfo.error);
   582→            }
   583→
   584→            // Retry if not last attempt
   585→            if (attempt < retries) {
   586→                console.log('[SPLICE] getActiveSequenceWithRetry: No sequence found, retrying in ' + delayMs + 'ms...');
   587→                await new Promise(r => setTimeout(r, delayMs));
   588→            }
   589→        } catch (error) {
   590→            console.error('[SPLICE] getActiveSequenceWithRetry: Attempt failed:', error.message);
   591→            if (attempt === retries) {
   592→                console.error('[SPLICE] getActiveSequenceWithRetry: All attempts failed');
   593→                return null;
   594→            }
   595→            await new Promise(r => setTimeout(r, delayMs));
   596→        }
   597→    }
   598→    console.log('[SPLICE] getActiveSequenceWithRetry: Returning null after all attempts');
   599→    return null;
   600→}

   811→function applyPreset(presetId) {
   812→    const preset = PRESETS[presetId];
   813→    if (!preset) return;
   814→
   815→    // Custom preset - keep current settings
   816→    if (presetId === 'custom' || !preset.settings) {
   817→        saveSettings({ activePreset: 'custom' });
   818→        setStatus('Using custom settings');
   819→        return;
   820→    }
   821→
   822→    const settings = preset.settings;
   823→
   824→    // Apply to UI
   825→    if (ui.sensitivitySlider && settings.sensitivity !== undefined) {
   826→        ui.sensitivitySlider.value = settings.sensitivity;
   827→        if (ui.sensitivityValue) ui.sensitivityValue.textContent = settings.sensitivity;
   828→    }
   829→
   830→    if (ui.enableJCut) {
   831→        ui.enableJCut.checked = settings.enableJCut || false;
   832→        if (ui.jcutSettings) {
   833→            ui.jcutSettings.classList.toggle('collapsed', !settings.enableJCut);
   834→        }
   835→    }
   836→
   837→    if (settings.jCutLeadIn !== undefined && ui.jcutLeadIn) {
   838→        ui.jcutLeadIn.value = Math.round(settings.jCutLeadIn * 100);
   839→        if (ui.jcutLeadInValue) ui.jcutLeadInValue.textContent = settings.jCutLeadIn + 's';
   840→    }
   841→
   842→    if (settings.jCutLeadOut !== undefined && ui.jcutLeadOut) {
   843→        ui.jcutLeadOut.value = Math.round(settings.jCutLeadOut * 100);
   844→        if (ui.jcutLeadOutValue) ui.jcutLeadOutValue.textContent = settings.jCutLeadOut + 's';
   845→    }
   846→
   847→    // Save merged settings
   848→    saveSettings({
   849→        activePreset: presetId,
   850→        sensitivity: settings.sensitivity,
   851→        threshold: settings.threshold,
   852→        minSilenceLength: settings.minSilenceLength,
   853→        paddingStart: settings.paddingStart,
   854→        paddingEnd: settings.paddingEnd,
   855→        enableTakesDetection: settings.enableTakesDetection,
   856→        enableJCut: settings.enableJCut || false,
   857→        jcutLeadIn: settings.jCutLeadIn || 0.3,
   858→        jcutLeadOut: settings.jCutLeadOut || 0.2
   859→    });
   860→
   861→    setStatus(`Applied ${preset.name} preset`);
   862→}

   920→async function runDetection() {
   921→    if (isOperationInProgress) return;
   922→
   923→    // Check sequence with enhanced error messaging
   924→    const hasSequence = await checkSequenceOpen();
   925→    if (!hasSequence) {
   926→        // Check if there was a communication error vs just no sequence
   927→        const lastError = window.spliceState.lastSequenceCheckError;
   928→        if (lastError && (lastError.includes('CEP') || lastError.includes('communication') || lastError.includes('evalScript'))) {
   929→            setStatus('Premiere Pro connection issue - try reopening panel', true);
   930→            console.error('[SPLICE] runDetection: JSX communication failure:', lastError);
   931→        } else {
   932→            setStatus('Please open a sequence in the Timeline first', true);
   933→            debugLog('runDetection: No active sequence in Timeline');
   934→        }
   935→        // Clear the error for next check
   936→        window.spliceState.lastSequenceCheckError = null;
   937→        return;
   938→    }
   939→
   940→    // Check online
   941→    if (!isOnline()) {
   942→        setStatus('Offline - Check your connection', true);
   943→        return;
   944→    }
   945→
   946→    isOperationInProgress = true;
   947→    ui.goBtn.disabled = true;
   948→    showProgress('Detecting silences...');
   949→
   950→    try {
   951→        const settings = getSettings();
   952→
   953→        // FIX: Get sequence info with retry logic
   954→        const seqInfo = await getActiveSequenceWithRetry(3, 300);
   955→        debugLog('Sequence info:', seqInfo);
   956→
   957→        if (!seqInfo) {
   958→            console.warn('[SPLICE] runDetection: getActiveSequenceWithRetry failed after retries');
   959→            setStatus('Could not access active sequence info', true);
   960→            ui.goBtn.disabled = false;
   961→            isOperationInProgress = false;
   962→            hideProgress();
   963→            return;
   964→        }
   965→
   966→        // Store sequence info for later use
   967→        currentSequenceInfo = seqInfo;
   968→        window.spliceState.activeSequence = seqInfo;
   969→        if (seqInfo && seqInfo.videoPath) {
   970→            window.spliceState.currentVideoPath = seqInfo.videoPath;
   971→        }
   972→
   973→        // Export audio via JSX for silence detection
   974→        updateProgress(10, 'Exporting audio...');
   975→
   976→        let audioFilePath = null;
   977→        try {
   978→            // FIX: CEP-CRIT-010 - Main.js Export Timeout
   979→            // Use LONG timeout (120s) for export operations
   980→            const exportResult = await jsx.callWithTimeout(
   981→                'exportSequenceAudioForAnalysis',
   982→                [],
   983→                JSX_TIMEOUTS.LONG
   984→            );
   985→            if (exportResult && exportResult.success && exportResult.outputPath) {
   986→                audioFilePath = exportResult.outputPath;
   987→                debugLog('Audio exported to:', audioFilePath);
   988→            } else if (exportResult && exportResult.error) {
   989→                throw new Error(exportResult.error);
   990→            }
   991→        } catch (exportError) {
   992→            console.warn('[SPLICE] Audio export failed, trying fallback:', exportError.message);
   993→            // Try to get audio from first clip in timeline
   994→            try {
   995→                const clipInfo = await jsx.call('getFirstClipAudioPath');
   996→                if (clipInfo && clipInfo.path) {
   997→                    audioFilePath = clipInfo.path;
   998→                    debugLog('Using clip audio path:', audioFilePath);
   999→                }
  1000→            } catch (clipError) {
  1001→                console.warn('[SPLICE] Fallback audio path failed:', clipError.message);
  1002→            }
  1003→        }
  1004→
  1005→        if (!audioFilePath) {
  1006→            throw new Error('Could not export or locate audio file. Please ensure sequence has audio tracks.');
  1007→        }
  1008→
  1009→        // Store audio path globally for other modules
  1010→        currentAudioPath = audioFilePath;
  1011→        window.spliceState.audioPath = audioFilePath;
  1012→
  1013→        // Call backend for silence detection with audio file path
  1014→        updateProgress(30, 'Detecting silences...');
  1015→        const silenceResponse = await fetchWithTimeout(
  1016→            `${getBackendUrl()}/silences-rms`,
  1017→            {
  1018→                method: 'POST',
  1019→                headers: getAuthHeaders(),

  1021→                    wavPath: audioFilePath,
  1022→                    sensitivity: settings.sensitivity,
  1023→                    minSilenceLength: settings.minSilenceLength || 0.5
  1024→                })
  1025→            }
  1026→        );
  1027→
  1028→        if (!silenceResponse.ok) {
  1029→            throw new Error(await parseErrorResponse(silenceResponse));
  1030→        }
  1031→
  1032→        const silenceData = await silenceResponse.json();
  1033→        previewSilences = silenceData.silences || [];
  1034→
  1035→        // Detect takes if enabled
  1036→        if (settings.enableTakesDetection) {
  1037→            updateProgress(60, 'Detecting takes...');

  1040→                    `${getBackendUrl()}/analyze`,

  1042→                        method: 'POST',
  1043→                        headers: getAuthHeaders(),
  1044→                        body: JSON.stringify({
  1045→                            wavPath: audioFilePath,
  1046→                            detectTakes: true
  1047→                        })
  1048→                    }
  1049→                );

  1051→                if (analyzeResponse.ok) {
  1052→                    const analyzeData = await analyzeResponse.json();
  1053→                    previewTakes = analyzeData.takes || [];

  1090→                            bleepType: settings.bleepType || 'standard'

  1092→                    }
  1093→                );
  1094→
  1095→                if (profanityResponse.ok) {
  1096→                    const profanityData = await profanityResponse.json();
  1097→                    // Backend returns 'words' array, not 'instances'
  1098→                    currentProfanity = profanityData.words || profanityData.instances || [];
  1099→                    if (currentProfanity.length > 0) {

  1102→                } else {
  1103→                    console.warn('[SPLICE] Profanity detection failed:', await parseErrorResponse(profanityResponse));

  1106→                console.warn('[SPLICE] Profanity detection failed:', e.message);

  1110→        // Detect chapters if enabled
  1111→        if (settings.enableChapters) {
  1112→            updateProgress(80, 'Detecting chapters...');
  1113→            try {
