   417â†’async function runDetection() {
   418â†’    if (isOperationInProgress) return;
   419â†’
   420â†’    // Check sequence
   421â†’    const hasSequence = await checkSequenceOpen();
   422â†’    if (!hasSequence) {
   423â†’        setStatus('Please open a sequence first', true);
   424â†’        return;
   425â†’    }
   426â†’
   427â†’    // Check online
   428â†’    if (!isOnline()) {
   429â†’        setStatus('Offline - Check your connection', true);
   430â†’        return;
   431â†’    }
   432â†’
   433â†’    isOperationInProgress = true;
   434â†’    ui.goBtn.disabled = true;
   435â†’    showProgress('Detecting silences...');
   436â†’
   437â†’    try {
   438â†’        const settings = getSettings();
   439â†’
   440â†’        // Get sequence info
   441â†’        const seqInfo = await jsx.call('getActiveSequence');
   442â†’        console.log('[SPLICE] Sequence info:', seqInfo);
   443â†’
   444â†’        // Export audio via JSX for silence detection
   445â†’        updateProgress(10, 'Exporting audio...');
   446â†’
   447â†’        let audioFilePath = null;
   448â†’        try {
   449â†’            // Get sequence info for audio export
   450â†’            const exportResult = await jsx.call('exportSequenceAudioForAnalysis');
   451â†’            if (exportResult && exportResult.success && exportResult.outputPath) {
   452â†’                audioFilePath = exportResult.outputPath;
   453â†’                console.log('[SPLICE] Audio exported to:', audioFilePath);
   454â†’            } else if (exportResult && exportResult.error) {
   455â†’                throw new Error(exportResult.error);
   456â†’            }
   457â†’        } catch (exportError) {
   458â†’            console.warn('[SPLICE] Audio export failed, trying fallback:', exportError.message);
   459â†’            // Try to get audio from first clip in timeline
   460â†’            try {
   461â†’                const clipInfo = await jsx.call('getFirstClipAudioPath');
   462â†’                if (clipInfo && clipInfo.path) {
   463â†’                    audioFilePath = clipInfo.path;
   464â†’                    console.log('[SPLICE] Using clip audio path:', audioFilePath);
   465â†’                }
   466â†’            } catch (clipError) {
   467â†’                console.warn('[SPLICE] Fallback audio path failed:', clipError.message);
   468â†’            }
   469â†’        }
   470â†’
   471â†’        if (!audioFilePath) {
   472â†’            throw new Error('Could not export or locate audio file. Please ensure sequence has audio tracks.');
   473â†’        }
   474â†’
   475â†’        // Call backend for silence detection with audio file path
   476â†’        updateProgress(30, 'Detecting silences...');
   477â†’        const silenceResponse = await fetchWithTimeout(
   478â†’            `${getBackendUrl()}/silences-rms`,
   479â†’            {
   480â†’                method: 'POST',
   481â†’                headers: getAuthHeaders(),
   482â†’                body: JSON.stringify({
   483â†’                    audioPath: audioFilePath,
   484â†’                    sensitivity: settings.sensitivity,
   485â†’                    minSilenceLength: settings.minSilenceLength || 0.5
   486â†’                })
   487â†’            }
   488â†’        );
   489â†’
   490â†’        if (!silenceResponse.ok) {
   491â†’            throw new Error(await parseErrorResponse(silenceResponse));
   492â†’        }
   493â†’
   494â†’        const silenceData = await silenceResponse.json();
   495â†’        previewSilences = silenceData.silences || [];
   496â†’
   497â†’        // Detect takes if enabled
   498â†’        if (settings.enableTakesDetection) {
   499â†’            updateProgress(60, 'Detecting takes...');
   500â†’            const analyzeResponse = await fetchWithTimeout(
   501â†’                `${getBackendUrl()}/analyze`,
   502â†’                {
   503â†’                    method: 'POST',
   504â†’                    headers: getAuthHeaders(),
   505â†’                    body: JSON.stringify({
   506â†’                        detectTakes: true
   507â†’                    })
   508â†’                }
   509â†’            );
   510â†’
   511â†’            if (analyzeResponse.ok) {
   512â†’                const analyzeData = await analyzeResponse.json();
   513â†’                previewTakes = analyzeData.takes || [];
   514â†’            }
   515â†’        }
   516â†’
   517â†’        // Detect chapters if enabled
   518â†’        if (settings.enableChapters) {
   519â†’            updateProgress(80, 'Detecting chapters...');
   520â†’            try {
   521â†’                const chapterResponse = await fetchWithTimeout(
   522â†’                    `${getBackendUrl()}/chapters`,
   523â†’                    {
   524â†’                        method: 'POST',
   525â†’                        headers: getAuthHeaders(),
   526â†’                        body: JSON.stringify({
   527â†’                            maxChapters: settings.maxChapters || 10,
   528â†’                            minChapterLength: settings.minChapterLength || 60
   529â†’                        })
   530â†’                    }
