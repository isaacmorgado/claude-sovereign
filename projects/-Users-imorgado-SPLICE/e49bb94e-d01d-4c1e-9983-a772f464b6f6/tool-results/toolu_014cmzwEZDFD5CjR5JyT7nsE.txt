    99→// Countdown timer component
   100→function UrgencyCountdown() {
   101→  const [timeLeft, setTimeLeft] = useState({ days: 0, hours: 0, minutes: 0, seconds: 0 });
   102→
   103→  useEffect(() => {
   104→    // Set end date to end of current month
   105→    const getEndDate = () => {
   106→      const now = new Date();
   107→      return new Date(now.getFullYear(), now.getMonth() + 1, 0, 23, 59, 59);
   108→    };
   109→
   110→    const calculateTimeLeft = () => {
   111→      const difference = getEndDate().getTime() - new Date().getTime();
   112→      if (difference > 0) {
   113→        return {
   114→          days: Math.floor(difference / (1000 * 60 * 60 * 24)),
   115→          hours: Math.floor((difference / (1000 * 60 * 60)) % 24),
   116→          minutes: Math.floor((difference / 1000 / 60) % 60),
   117→          seconds: Math.floor((difference / 1000) % 60),
   118→        };
   119→      }
   120→      return { days: 0, hours: 0, minutes: 0, seconds: 0 };
   121→    };
   122→
   123→    setTimeLeft(calculateTimeLeft());
   124→    const timer = setInterval(() => setTimeLeft(calculateTimeLeft()), 1000);
   125→    return () => clearInterval(timer);
   126→  }, []);
   127→
   128→  return (
   129→    <div className="flex items-center justify-center gap-3 text-sm">
   130→      {[
   131→        { value: timeLeft.days, label: "Days" },
   132→        { value: timeLeft.hours, label: "Hrs" },
   133→        { value: timeLeft.minutes, label: "Min" },
   134→        { value: timeLeft.seconds, label: "Sec" },
   135→      ].map((item, i) => (
   136→        <div key={item.label} className="flex items-center gap-1">
   137→          <div className="bg-[#00D4FF]/20 text-[#00D4FF] font-mono font-bold px-2 py-1 rounded">
   138→            {String(item.value).padStart(2, "0")}
   139→          </div>
   140→          <span className="text-gray-400 text-xs">{item.label}</span>
   141→          {i < 3 && <span className="text-gray-600 ml-1">:</span>}
   142→        </div>
   143→      ))}
   144→    </div>
   145→  );
   146→}
