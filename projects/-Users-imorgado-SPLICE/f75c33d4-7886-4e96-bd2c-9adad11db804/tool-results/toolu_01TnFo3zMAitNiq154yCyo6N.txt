   430â†’function applyPreset(presetId) {
   431â†’    const preset = PRESETS[presetId];
   432â†’    if (!preset) return;
   433â†’
   434â†’    // Custom preset - keep current settings
   435â†’    if (presetId === 'custom' || !preset.settings) {
   436â†’        saveSettings({ activePreset: 'custom' });
   437â†’        setStatus('Using custom settings');
   438â†’        return;
   439â†’    }
   440â†’
   441â†’    const settings = preset.settings;
   442â†’
   443â†’    // Apply to UI
   444â†’    if (ui.sensitivitySlider && settings.sensitivity !== undefined) {
   445â†’        ui.sensitivitySlider.value = settings.sensitivity;
   446â†’        if (ui.sensitivityValue) ui.sensitivityValue.textContent = settings.sensitivity;
   447â†’    }
   448â†’
   449â†’    if (ui.enableJCut) {
   450â†’        ui.enableJCut.checked = settings.enableJCut || false;
   451â†’        if (ui.jcutSettings) {
   452â†’            ui.jcutSettings.classList.toggle('collapsed', !settings.enableJCut);
   453â†’        }
   454â†’    }
   455â†’
   456â†’    if (settings.jCutLeadIn !== undefined && ui.jcutLeadIn) {
   457â†’        ui.jcutLeadIn.value = Math.round(settings.jCutLeadIn * 100);
   458â†’        if (ui.jcutLeadInValue) ui.jcutLeadInValue.textContent = settings.jCutLeadIn + 's';
   459â†’    }
   460â†’
   461â†’    if (settings.jCutLeadOut !== undefined && ui.jcutLeadOut) {
   462â†’        ui.jcutLeadOut.value = Math.round(settings.jCutLeadOut * 100);
   463â†’        if (ui.jcutLeadOutValue) ui.jcutLeadOutValue.textContent = settings.jCutLeadOut + 's';
   464â†’    }
   465â†’
   466â†’    // Save merged settings
   467â†’    saveSettings({
   468â†’        activePreset: presetId,
   469â†’        sensitivity: settings.sensitivity,
   470â†’        threshold: settings.threshold,
   471â†’        minSilenceLength: settings.minSilenceLength,
   472â†’        paddingStart: settings.paddingStart,
   473â†’        paddingEnd: settings.paddingEnd,
   474â†’        enableTakesDetection: settings.enableTakesDetection,
   475â†’        enableJCut: settings.enableJCut || false,
   476â†’        jcutLeadIn: settings.jCutLeadIn || 0.3,
   477â†’        jcutLeadOut: settings.jCutLeadOut || 0.2
   478â†’    });
   479â†’
   480â†’    setStatus(`Applied ${preset.name} preset`);
   481â†’}
