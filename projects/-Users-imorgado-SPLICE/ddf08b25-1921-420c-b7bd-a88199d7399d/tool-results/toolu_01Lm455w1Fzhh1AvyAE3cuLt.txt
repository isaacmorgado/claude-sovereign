     4→ * Analyzes Whisper transcription segments to find silent gaps.
     5→ * Returns silence regions for timeline removal.

     8→/**
     9→ * Detect silences (gaps between speech segments)
    10→ * @param {Array} segments - Whisper segments with start/end timestamps
    11→ * @param {number} threshold - Minimum gap duration to count as silence (seconds)
    12→ * @returns {Array<{start: number, end: number, duration: number}>}
    13→ */
    14→function detectSilences(segments, threshold = 0.5) {
    15→  if (!segments || segments.length < 2) {
    16→    return [];
    17→  }
    18→
    19→  const silences = [];
    20→
    21→  // Check gap before first segment (leading silence)
    22→  if (segments[0].start >= threshold) {
    23→    silences.push({
    24→      start: 0,
    25→      end: segments[0].start,
    26→      duration: segments[0].start
    27→    });
    28→  }
    29→
    30→  // Check gaps between segments
    31→  for (let i = 0; i < segments.length - 1; i++) {
    32→    const gap = segments[i + 1].start - segments[i].end;
    33→    if (gap >= threshold) {
    34→      silences.push({
    35→        start: segments[i].end,
    36→        end: segments[i + 1].start,
    37→        duration: gap
    38→      });
    39→    }
    40→  }
    41→
    42→  console.log(`[SPLICE] Detected ${silences.length} silence(s) with threshold ${threshold}s`);
    43→  return silences;
    44→}
    45→
    46→module.exports = { detectSilences };
