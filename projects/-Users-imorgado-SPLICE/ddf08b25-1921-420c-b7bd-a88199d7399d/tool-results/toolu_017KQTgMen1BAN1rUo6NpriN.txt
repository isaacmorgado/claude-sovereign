    89→    89→    /**

    94→    94→    init: function() {

   101→   101→    /**
   102→   102→     * FireCut-style retry logic for CSInterface initialization
   103→   103→     * Retries up to 5 times with exponential backoff
   104→   104→     */
   105→   105→    _initWithRetry: async function() {
   106→   106→        const maxRetries = 5;
   107→   107→        const baseDelay = 200; // Start with 200ms, will double each retry
   108→   108→
   109→   109→        for (let attempt = 1; attempt <= maxRetries; attempt++) {
   110→   110→            const success = await this._tryInit(attempt);
   111→   111→            if (success) {
   112→   112→                // After CSInterface init, verify JSX is ready
   113→   113→                const jsxReady = await this._verifyJSXReady();
   114→   114→                if (jsxReady) {
   115→   115→                    this._jsxReady = true;
   116→   116→                    debugLog(`[JSX] Full initialization complete on attempt ${attempt}`);
   117→   117→                    return true;
   118→   118→                } else {
   119→   119→                    debugLog(`[JSX] CSInterface ready but JSX not responding, attempt ${attempt}/${maxRetries}`);
   120→   120→                }
   121→   121→            }
   122→   122→
   123→   123→            if (attempt < maxRetries) {
   124→   124→                const delay = baseDelay * Math.pow(2, attempt - 1);
   125→   125→                debugLog(`[JSX] Init attempt ${attempt} failed, retrying in ${delay}ms...`);
   126→   126→                await this._delay(delay);
   127→   127→            }
   128→   128→        }
   129→   129→
   130→   130→        console.warn('[JSX] Initialization failed after all retries - panel may have limited functionality');
   131→   131→        return this._initialized;
   132→   132→    },

   134→   134→    /**
   135→   135→     * Single initialization attempt
   136→   136→     */
   137→   137→    _tryInit: function(attempt) {
   138→   138→        return new Promise((resolve) => {
   139→   139→            try {
   140→   140→                if (typeof CSInterface !== 'undefined') {
   141→   141→                    this.cs = new CSInterface();
   142→   142→                    // Verify we're in a real CEP environment by checking for __adobe_cep__
   143→   143→                    if (window.__adobe_cep__) {
   144→   144→                        this._initialized = true;
   145→   145→                        debugLog(`[JSX] CSInterface initialized successfully (CEP environment detected) - attempt ${attempt}`);
   146→   146→                        resolve(true);
   147→   147→                    } else {
   148→   148→                        // CSInterface exists but no CEP runtime - browser/dev mode
   149→   149→                        this._initialized = true;
   150→   150→                        debugLog(`[JSX] CSInterface initialized (development/browser mode) - attempt ${attempt}`);
   151→   151→                        resolve(true);
   152→   152→                    }
   153→   153→                } else {
   154→   154→                    debugLog(`[JSX] CSInterface not yet available - attempt ${attempt}`);
   155→   155→                    resolve(false);
   156→   156→                }
   157→   157→            } catch (err) {
   158→   158→                console.error(`[JSX] Init attempt ${attempt} error:`, err);
   159→   159→                resolve(false);
   160→   160→            }
   161→   161→        });
   162→   162→    },

   168→   168→    _verifyJSXReady: async function() {

   170→   170→
   171→   171→        return new Promise((resolve) => {
   172→   172→            const timeout = setTimeout(() => {
   173→   173→                debugLog('[JSX] JSX readiness check timed out');
   174→   174→                resolve(false);
   175→   175→            }, 2000);

   180→   180→                    clearTimeout(timeout);
   181→   181→                    if (result === 'ready') {
   182→   182→                        debugLog('[JSX] JSX bridge verified ready - hostScript.jsx loaded');
   183→   183→                        resolve(true);
   184→   184→                    } else {
   185→   185→                        debugLog('[JSX] JSX functions not yet available:', result);

   187→   187→                    }

   192→   192→                resolve(false);
   193→   193→            }
   194→   194→        });

   216→   216→    /**
   217→   217→     * Execute ExtendScript in Premiere Pro with timeout protection
   218→   218→     * FIX: CEP-CRIT-001 - JSX Bridge Timeout Bug
   219→   219→     * @param {string} script - ExtendScript to execute
   220→   220→     * @param {number} timeout - Timeout in ms (default: JSX_TIMEOUTS.DEFAULT)
   221→   221→     * @returns {Promise<any>} - Parsed result
   222→   222→     */
   223→   223→    evalScript: async function(script, timeout = JSX_TIMEOUTS.DEFAULT) {
   224→   224→        await this.ensureInitialized();
   225→   225→
   226→   226→        return new Promise((resolve, reject) => {
   227→   227→            let completed = false;
   228→   228→
   229→   229→            // Timeout handler - rejects if ExtendScript hangs
   230→   230→            const timeoutId = setTimeout(() => {
   231→   231→                if (!completed) {
   232→   232→                    completed = true;
   233→   233→                    const scriptPreview = script.length > 100 ? script.substring(0, 100) + '...' : script;
   234→   234→                    const error = new Error(`ExtendScript timeout after ${timeout}ms: ${scriptPreview}`);
   235→   235→                    error.code = 'JSX_TIMEOUT';
   236→   236→                    console.error('[JSX] Timeout:', scriptPreview);
   237→   237→                    reject(error);
   238→   238→                }
   239→   239→            }, timeout);
   240→   240→
   241→   241→            this.cs.evalScript(script, (result) => {
   242→   242→                // Ignore late callbacks after timeout
   243→   243→                if (completed) {
   244→   244→                    console.warn('[JSX] Late evalScript result received after timeout - ignoring');
   245→   245→                    return;
   246→   246→                }
   247→   247→
   248→   248→                completed = true;
   249→   249→                clearTimeout(timeoutId);
   250→   250→
   251→   251→                if (result === 'undefined' || result === undefined) {
   252→   252→                    resolve(null);
   253→   253→                    return;
   254→   254→                }
   255→   255→
   256→   256→                // Check for EvalScript error or ExtendScript runtime errors
   257→   257→                if (typeof result === 'string' && (
   258→   258→                    result.indexOf('EvalScript error') !== -1 ||
   259→   259→                    result.indexOf('Error:') !== -1 ||
   260→   260→                    result.indexOf('TypeError:') !== -1 ||
   261→   261→                    result.indexOf('ReferenceError:') !== -1
   262→   262→                )) {
   263→   263→                    reject(new Error(result));
   264→   264→                    return;
   265→   265→                }
   266→   266→
   267→   267→                // Try to parse JSON
   268→   268→                try {
   269→   269→                    const parsed = JSON.parse(result);
   270→   270→                    if (parsed.error) {
   271→   271→                        reject(new Error(parsed.error));
   272→   272→                    } else {
   273→   273→                        resolve(parsed);
   274→   274→                    }
   275→   275→                } catch (e) {
   276→   276→                    // Return as-is if not JSON
   277→   277→                    resolve(result);
   278→   278→                }
   279→   279→            });
   280→   280→        });
   281→   281→    },

   283→   283→    /**
   284→   284→     * Call a JSX function with arguments
   285→   285→     * @param {string} funcName - Function name
   286→   286→     * @param {...any} args - Arguments to pass (last arg can be {timeout: number})
   287→   287→     * @returns {Promise<any>}
   288→   288→     */
   289→   289→    call: async function(funcName, ...args) {
   290→   290→        await this.ensureInitialized();
   291→   291→
   292→   292→        // Check if last argument is an options object with timeout
   293→   293→        let timeout = JSX_TIMEOUTS.DEFAULT;
   294→   294→        let funcArgs = args;
   295→   295→
   296→   296→        if (args.length > 0 && typeof args[args.length - 1] === 'object' && args[args.length - 1] !== null) {
   297→   297→            const lastArg = args[args.length - 1];
   298→   298→            if ('_jsxTimeout' in lastArg) {
   299→   299→                timeout = lastArg._jsxTimeout;
   300→   300→                funcArgs = args.slice(0, -1);
   301→   301→            }
   302→   302→        }
   303→   303→
   304→   304→        const escapedArgs = funcArgs.map(arg => {
   305→   305→            if (arg === null || arg === undefined) return 'null';
   306→   306→            if (typeof arg === 'string') {
   307→   307→                // Escape backslashes first, then other special chars
   308→   308→                return `'${arg
   309→   309→                    .replace(/\\/g, '\\\\')
   310→   310→                    .replace(/'/g, "\\'")
   311→   311→                    .replace(/\r/g, '\\r')
   312→   312→                    .replace(/\n/g, '\\n')
   313→   313→                    .replace(/\t/g, '\\t')}'`;
   314→   314→            }
   315→   315→            if (typeof arg === 'object') {
   316→   316→                return `'${JSON.stringify(arg)
   317→   317→                    .replace(/\\/g, '\\\\')
   318→   318→                    .replace(/'/g, "\\'")}'`;
   319→   319→            }
   320→   320→            return String(arg);
   321→   321→        });
   322→   322→
   323→   323→        const script = `${funcName}(${escapedArgs.join(', ')})`;
   324→   324→        return this.evalScript(script, timeout);
   325→   325→    },
   326→   326→
   327→   327→    /**
   328→   328→     * Call a JSX function with explicit timeout
   329→   329→     * Convenience wrapper for operations that need non-default timeouts
   330→   330→     * @param {string} funcName - Function name
   331→   331→     * @param {Array} args - Arguments array
   332→   332→     * @param {number} timeout - Timeout in ms
   333→   333→     * @returns {Promise<any>}
   334→   334→     */
   335→   335→    callWithTimeout: async function(funcName, args = [], timeout = JSX_TIMEOUTS.DEFAULT) {
   336→   336→        return this.call(funcName, ...args, { _jsxTimeout: timeout });
   337→   337→    }

   348→   348→ * @param {number} timeout - Timeout in ms

   350→   350→async function fetchWithTimeout(url, options = {}, timeout = SPLICE_CONFIG.FETCH_TIMEOUT) {
   351→   351→    // DIAGNOSTIC: Log fetch request details (debug mode only)
   352→   352→    debugFetch('=== FETCH REQUEST START ===');

   354→   354→    debugFetch('Method:', options?.method || 'GET');
   355→   355→    debugFetch('Headers:', options?.headers);
   356→   356→    debugFetch('Timeout:', timeout);

   358→   358→    debugFetch('=================================');
   359→   359→
   360→   360→    const controller = new AbortController();
   361→   361→    const timeoutId = setTimeout(() => controller.abort(), timeout);

   365→   365→            ...options,
   366→   366→            signal: controller.signal

   368→   368→        clearTimeout(timeoutId);
   369→   369→
   370→   370→        // DIAGNOSTIC: Log successful response (debug mode only)
   371→   371→        debugFetch('=== FETCH RESPONSE SUCCESS ===');
   372→   372→        debugFetch('Status:', response.status);

   374→   374→        debugFetch('=================================');
   375→   375→
   376→   376→        return response;
   377→   377→    } catch (error) {

   380→   380→        // DIAGNOSTIC: Log raw fetch error (debug mode only)

   383→   383→        debugFetch('Error message:', error?.message);

   387→   387→        debugFetch('Is TypeError:', error.name === 'TypeError');
   388→   388→        debugFetch('Full error object:', JSON.stringify(error, Object.getOwnPropertyNames(error)));
   389→   389→        debugFetch('=================================');
   390→   390→
   391→   391→        if (error.name === 'AbortError') {
   392→   392→            throw new Error(`Request timeout after ${timeout}ms`);
   393→   393→        }
   394→   394→        throw error;
   395→   395→    }

   436→   436→ * Check if the access token is expired or about to expire
   437→   437→ * @returns {boolean} True if token needs refresh

   439→   439→function isTokenExpired() {
   440→   440→    const settings = getSettings();
   441→   441→    if (!settings.tokenExpiry) return true;
   442→   442→
   443→   443→    // Refresh if less than 5 minutes remaining
   444→   444→    const bufferMs = 5 * 60 * 1000;
   445→   445→    return Date.now() >= (settings.tokenExpiry - bufferMs);
   446→   446→}

   448→   448→/**
   449→   449→ * Refresh the access token using the refresh token
   450→   450→ * @returns {Promise<boolean>} True if refresh succeeded
   451→   451→ */
   452→   452→async function refreshAccessToken() {
   453→   453→    const settings = getSettings();
   454→   454→
   455→   455→    if (!settings.refreshToken) {
   456→   456→        debugLog('[Auth] No refresh token available');
   457→   457→        return false;
   458→   458→    }
   459→   459→
   460→   460→    try {
   461→   461→        debugLog('[Auth] Refreshing access token...');
   462→   462→
   463→   463→        const response = await fetchWithTimeout(`${getBackendUrl()}/auth/refresh`, {
   464→   464→            method: 'POST',
   465→   465→            headers: { 'Content-Type': 'application/json' },
   466→   466→            body: JSON.stringify({ refreshToken: settings.refreshToken })
   467→   467→        }, 30000);
   468→   468→
   469→   469→        if (!response.ok) {
   470→   470→            const errorMsg = await parseErrorResponse(response);
   471→   471→            console.error('[Auth] Token refresh failed:', errorMsg);
   472→   472→
   473→   473→            // If refresh token is invalid, clear auth and require re-login
   474→   474→            if (response.status === 401) {
   475→   475→                clearAuthTokens();
   476→   476→            }
   477→   477→            return false;
   478→   478→        }
   479→   479→
   480→   480→        const data = await response.json();
   481→   481→
   482→   482→        // Save new access token
   483→   483→        const expiresInMs = (data.expiresIn || 86400) * 1000; // Default 24h
   484→   484→        saveSettings({
   485→   485→            accessToken: data.token,
   486→   486→            tokenExpiry: Date.now() + expiresInMs
   487→   487→        });
   488→   488→
   489→   489→        debugLog('[Auth] Access token refreshed successfully');
   490→   490→        return true;
   491→   491→    } catch (err) {
   492→   492→        console.error('[Auth] Token refresh error:', err);
   493→   493→        return false;
   494→   494→    }
   495→   495→}

   510→   510→ * Ensure we have a valid access token, refreshing if needed
   511→   511→ * @returns {Promise<boolean>} True if we have a valid token
   512→   512→ */
   513→   513→async function ensureValidToken() {
   514→   514→    const settings = getSettings();

   575→   575→// Encryption constants
   576→   576→const SPLICE_ENCRYPTION = {
   577→   577→    ALGORITHM: 'AES-GCM',
   578→   578→    KEY_LENGTH: 256,
   579→   579→    IV_LENGTH: 12,
   580→   580→    TAG_LENGTH: 128,
   581→   581→    SALT: 'SPLICE_PANEL_SALT_v1', // Static salt for key derivation
   582→   582→    ITERATIONS: 100000,
   583→   583→    PREFIX: 'ENC:v1:'  // Prefix to identify encrypted values
   584→   584→};

   586→   586→// Cached encryption key (derived once per session)
   587→   587→let _encryptionKey = null;
   588→   588→
   589→   589→/**
   590→   590→ * Get or derive the encryption key from a machine-specific identifier
   591→   591→ * Uses PBKDF2 to derive a strong key from available identifiers
   592→   592→ * @returns {Promise<CryptoKey>} The derived encryption key
   593→   593→ */
   594→   594→async function getEncryptionKey() {
   595→   595→    if (_encryptionKey) {
   596→   596→        return _encryptionKey;
   597→   597→    }
   598→   598→
   599→   599→    try {
   600→   600→        // Use a combination of identifiers for key derivation
   601→   601→        // In CEP, we use localStorage ID + app version as the key material
   602→   602→        let keyMaterial = localStorage.getItem('splice_device_id');
   603→   603→        if (!keyMaterial) {
   604→   604→            // Generate a random device ID on first use
   605→   605→            keyMaterial = crypto.randomUUID ? crypto.randomUUID() :
   606→   606→                'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
   607→   607→                    const r = Math.random() * 16 | 0;
   608→   608→                    const v = c === 'x' ? r : (r & 0x3 | 0x8);
   609→   609→                    return v.toString(16);
   610→   610→                });
   611→   611→            localStorage.setItem('splice_device_id', keyMaterial);
   612→   612→        }
   613→   613→
   614→   614→        const encoder = new TextEncoder();
   615→   615→        const keyData = encoder.encode(keyMaterial + SPLICE_ENCRYPTION.SALT);
   616→   616→
   617→   617→        // Import the key material for PBKDF2
   618→   618→        const baseKey = await crypto.subtle.importKey(
   619→   619→            'raw',
   620→   620→            keyData,
   621→   621→            'PBKDF2',
   622→   622→            false,
   623→   623→            ['deriveKey']
   624→   624→        );
   625→   625→
   626→   626→        // Derive the actual AES key
   627→   627→        _encryptionKey = await crypto.subtle.deriveKey(
   628→   628→            {
   629→   629→                name: 'PBKDF2',
   630→   630→                salt: encoder.encode(SPLICE_ENCRYPTION.SALT),
   631→   631→                iterations: SPLICE_ENCRYPTION.ITERATIONS,
   632→   632→                hash: 'SHA-256'
   633→   633→            },
   634→   634→            baseKey,
   635→   635→            {
   636→   636→                name: SPLICE_ENCRYPTION.ALGORITHM,
   637→   637→                length: SPLICE_ENCRYPTION.KEY_LENGTH
   638→   638→            },
   639→   639→            false,
   640→   640→            ['encrypt', 'decrypt']
   641→   641→        );
   642→   642→
   643→   643→        return _encryptionKey;
   644→   644→    } catch (e) {
   645→   645→        debugLog('[Config] Key derivation error:', e);
   646→   646→        return null;
   647→   647→    }
   648→   648→}

   650→   650→/**
   651→   651→ * Encrypt data for secure localStorage storage using AES-GCM
   652→   652→ *
   653→   653→ * @param {any} data - Data to encrypt
   654→   654→ * @returns {Promise<string|null>} Encrypted string with prefix or null on error
   655→   655→ */
   656→   656→async function encryptForStorageAsync(data) {
   657→   657→    try {
   658→   658→        if (data === null || data === undefined) {
   659→   659→            return null;
   660→   660→        }
   661→   661→
   662→   662→        const key = await getEncryptionKey();
   663→   663→        if (!key) {
   664→   664→            // Fallback to legacy encoding if crypto not available
   665→   665→            debugLog('[Config] Crypto not available, using legacy encoding');
   666→   666→            return legacyEncode(data);
   667→   667→        }
   668→   668→
   669→   669→        const encoder = new TextEncoder();
   670→   670→        const jsonStr = JSON.stringify(data);
   671→   671→        const plaintext = encoder.encode(jsonStr);
   672→   672→
   673→   673→        // Generate random IV
   674→   674→        const iv = crypto.getRandomValues(new Uint8Array(SPLICE_ENCRYPTION.IV_LENGTH));
   675→   675→
   676→   676→        // Encrypt
   677→   677→        const ciphertext = await crypto.subtle.encrypt(
   678→   678→            {
   679→   679→                name: SPLICE_ENCRYPTION.ALGORITHM,
   680→   680→                iv: iv,
   681→   681→                tagLength: SPLICE_ENCRYPTION.TAG_LENGTH
   682→   682→            },
   683→   683→            key,
   684→   684→            plaintext
   685→   685→        );
   686→   686→
   687→   687→        // Combine IV + ciphertext and encode as base64
   688→   688→        const combined = new Uint8Array(iv.length + ciphertext.byteLength);
   689→   689→        combined.set(iv);
   690→   690→        combined.set(new Uint8Array(ciphertext), iv.length);
   691→   691→
   692→   692→        // Convert to base64 with prefix
   693→   693→        const base64 = btoa(String.fromCharCode(...combined));
   694→   694→        return SPLICE_ENCRYPTION.PREFIX + base64;
   695→   695→    } catch (e) {
   696→   696→        debugLog('[Config] Encryption error:', e);
   697→   697→        return legacyEncode(data);
   698→   698→    }
   699→   699→}

   701→   701→/**
   702→   702→ * Decrypt data from localStorage using AES-GCM
   703→   703→ *
   704→   704→ * @param {string} data - Encrypted string
   705→   705→ * @returns {Promise<any|null>} Decrypted data or null on error
   706→   706→ */
   707→   707→async function decryptFromStorageAsync(data) {
   708→   708→    try {
   709→   709→        if (!data || typeof data !== 'string') {
   710→   710→            return null;
   711→   711→        }
   712→   712→
   713→   713→        // Check if this is the new encrypted format
   714→   714→        if (!data.startsWith(SPLICE_ENCRYPTION.PREFIX)) {
   715→   715→            // Try legacy decoding for backward compatibility
   716→   716→            return legacyDecode(data);
   717→   717→        }
   718→   718→
   719→   719→        const key = await getEncryptionKey();
   720→   720→        if (!key) {
   721→   721→            debugLog('[Config] Crypto not available for decryption');
   722→   722→            return null;
   723→   723→        }
   724→   724→
   725→   725→        // Remove prefix and decode base64
   726→   726→        const base64Data = data.slice(SPLICE_ENCRYPTION.PREFIX.length);
   727→   727→        const combined = Uint8Array.from(atob(base64Data), c => c.charCodeAt(0));
   728→   728→
   729→   729→        // Extract IV and ciphertext
   730→   730→        const iv = combined.slice(0, SPLICE_ENCRYPTION.IV_LENGTH);
   731→   731→        const ciphertext = combined.slice(SPLICE_ENCRYPTION.IV_LENGTH);
   732→   732→
   733→   733→        // Decrypt
   734→   734→        const plaintext = await crypto.subtle.decrypt(
   735→   735→            {
   736→   736→                name: SPLICE_ENCRYPTION.ALGORITHM,
   737→   737→                iv: iv,
   738→   738→                tagLength: SPLICE_ENCRYPTION.TAG_LENGTH
   739→   739→            },
   740→   740→            key,
   741→   741→            ciphertext
   742→   742→        );
   743→   743→
   744→   744→        const decoder = new TextDecoder();
   745→   745→        const jsonStr = decoder.decode(plaintext);
   746→   746→        return JSON.parse(jsonStr);
   747→   747→    } catch (e) {
   748→   748→        // Try legacy decoding as fallback
   749→   749→        return legacyDecode(data);
   750→   750→    }
   751→   751→}
   752→   752→
   753→   753→/**
   754→   754→ * Legacy encoding (Base64) for backward compatibility
   755→   755→ * @param {any} data - Data to encode
   756→   756→ * @returns {string|null} Encoded string

   877→   877→function decryptSensitiveFields(settings) {

   881→   881→        if (decrypted[field] !== null && decrypted[field] !== undefined) {
   882→   882→            // Check if it looks encrypted before attempting to decrypt
   883→   883→            if (isEncryptedValue(decrypted[field])) {

   885→   885→                if (decryptedValue !== null) {

   890→   890→            // If not encrypted (legacy data), keep the original value

  1048→  1048→/**
  1049→  1049→ * SECURITY: Escape HTML to prevent XSS attacks
  1050→  1050→ * Use this when inserting untrusted content into innerHTML
  1051→  1051→ * @param {string} str - String to escape
  1052→  1052→ * @returns {string} Escaped string safe for innerHTML
  1053→  1053→ */
  1054→  1054→function escapeHtml(str) {
  1055→  1055→    if (str === null || str === undefined) return '';
  1056→  1056→    if (typeof str !== 'string') str = String(str);
  1057→  1057→
  1058→  1058→    const escapeMap = {
  1059→  1059→        '&': '&amp;',
  1060→  1060→        '<': '&lt;',
  1061→  1061→        '>': '&gt;',
  1062→  1062→        '"': '&quot;',
  1063→  1063→        "'": '&#x27;',
  1064→  1064→        '/': '&#x2F;',
  1065→  1065→        '`': '&#x60;'
  1066→  1066→    };
  1067→  1067→
  1068→  1068→    return str.replace(/[&<>"'\/`]/g, char => escapeMap[char]);
  1069→  1069→}
  1070→  1070→
  1071→  1071→/**
  1072→  1072→ * SECURITY: Create safe HTML from template with escaped values
  1073→  1073→ * @param {TemplateStringsArray} strings - Template strings
  1074→  1074→ * @param {...any} values - Values to escape
  1075→  1075→ * @returns {string} HTML with escaped values
  1076→  1076→ */
  1077→  1077→function safeHtml(strings, ...values) {
  1078→  1078→    return strings.reduce((result, str, i) => {
  1079→  1079→        const value = values[i - 1];
  1080→  1080→        const escaped = escapeHtml(value);
  1081→  1081→        return result + escaped + str;
  1082→  1082→    });
  1083→  1083→}

  1090→  1090→
  1091→  1091→    const hours = Math.floor(seconds / 3600);
