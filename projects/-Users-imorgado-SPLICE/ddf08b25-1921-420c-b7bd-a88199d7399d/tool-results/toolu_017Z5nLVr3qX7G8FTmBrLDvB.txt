   113→   113→function safeQEOperation(operation, operationName) {

   161→   161→function quantise_time(time_seconds) {
   162→   162→    // Get seconds per frame (e.g., 0.0333 for 30fps)
   163→   163→    var secsPerFrame = app.project.activeSequence.getSettings().videoFrameRate.seconds;
   164→   164→    // Round to nearest frame boundary
   165→   165→    return secsPerFrame * Math.round(time_seconds / secsPerFrame);

   275→   275→function cloneSequence(newName) {
   276→   276→    var seq = app.project.activeSequence;
   277→   277→    if (!seq) return JSON.stringify({ error: "No active sequence" });
   278→   278→
   279→   279→    try {
   280→   280→        // Clone the sequence
   281→   281→        seq.clone();
   282→   282→
   283→   283→        // Find the cloned sequence (last one with matching name pattern)
   284→   284→        var clonedSeq = null;
   285→   285→        for (var i = app.project.sequences.numSequences - 1; i >= 0; i--) {
   286→   286→            var s = app.project.sequences[i];
   287→   287→            if (s.name.indexOf(seq.name) === 0 && s.sequenceID !== seq.sequenceID) {
   288→   288→                clonedSeq = s;
   289→   289→                break;
   290→   290→            }
   291→   291→        }
   292→   292→
   293→   293→        if (clonedSeq && newName) {
   294→   294→            clonedSeq.name = newName;
   295→   295→        }
   296→   296→
   297→   297→        return JSON.stringify({
   298→   298→            success: true,
   299→   299→            sequenceId: clonedSeq ? clonedSeq.sequenceID : null,
   300→   300→            name: clonedSeq ? clonedSeq.name : newName
   301→   301→        });
   302→   302→    } catch (e) {
   303→   303→        return JSON.stringify({ error: e.message });
   304→   304→    }
