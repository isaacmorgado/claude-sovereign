   248→    const criticalElements = {
   249→        'goBtn': { el: ui.goBtn, desc: 'GO button (main action)' },
   250→        'settingsBtn': { el: ui.settingsBtn, desc: 'Settings button' },
   251→        'debugBtn': { el: ui.debugBtn, desc: 'Debug button' },
   252→        'creditBadge': { el: ui.creditBadge, desc: 'Credit badge' },

   255→        'optionsPanel': { el: ui.optionsPanel, desc: 'Options panel' }
   256→    };
   257→
   258→    const missingElements = [];
   259→    const presentElements = [];
   260→
   261→    Object.entries(criticalElements).forEach(([id, { el, desc }]) => {
   262→        if (!el) {
   263→            missingElements.push(`#${id} (${desc})`);
   264→        } else {
   265→            presentElements.push(id);
   266→        }
   267→    });
   268→
   269→    if (missingElements.length > 0) {
   270→        console.error('[SPLICE] ⚠️ UI INITIALIZATION INCOMPLETE');
   271→        console.error('[SPLICE] Missing critical elements:', missingElements);
   272→        console.error('[SPLICE] Panel may not function correctly. Check index.html structure.');
   273→
   274→        // Show error in status bar if available
   275→        const statusEl = document.getElementById('status');
   276→        if (statusEl) {
   277→            statusEl.textContent = 'Init error: ' + missingElements.length + ' UI elements missing';
   278→            statusEl.classList.add('error');
   279→        }
   280→    } else {
   281→        console.log('[SPLICE] ✓ All critical UI elements initialized:', presentElements.join(', '));
   282→    }
   283→
   284→    // Verify GO button has click listener
   285→    if (ui.goBtn) {
   286→        const listeners = getEventListeners ? getEventListeners(ui.goBtn) : null;

   297→/**
   298→ * Initialize event listeners for sequence state changes
   299→ * This provides real-time updates when sequences are activated/deactivated
   300→ */
   301→function initSequenceEventListeners() {
   302→    if (!jsx.cs) {
   303→        console.warn('[SPLICE] Cannot init sequence listeners - CSInterface not available');
   304→        return;
   305→    }
   306→
   307→    try {
   308→        // Listen for sequence activation events
   309→        jsx.cs.addEventListener('com.adobe.csxs.events.SequenceActivated', async (event) => {
   310→            debugLog('Sequence activated event received');
   311→            try {
   312→                const seqInfo = await jsx.call('getActiveSequence');
   313→                if (seqInfo && !seqInfo.error) {
   314→                    window.spliceState.activeSequence = seqInfo;
   315→                    setStatus('Sequence: ' + (seqInfo.name || 'Unknown'));
   316→                    debugLog('Active sequence updated:', seqInfo.name);
   317→                }
   318→            } catch (e) {
   319→                console.warn('[SPLICE] Failed to get sequence info on activation:', e.message);
   320→            }
   321→        });
   322→
   323→        // Listen for sequence close/deactivation
   324→        jsx.cs.addEventListener('com.adobe.csxs.events.SequenceDeactivated', () => {
   325→            debugLog('Sequence deactivated event');
   326→            window.spliceState.activeSequence = null;
   327→            setStatus('No sequence open', true);
   328→        });
   329→
   330→        // Listen for project item selection changes (includes sequences)
   331→        jsx.cs.addEventListener('com.adobe.csxs.events.SelectionChanged', async () => {
   332→            // Debounce rapid selection changes
   333→            if (window._selectionChangeTimeout) {
   334→                clearTimeout(window._selectionChangeTimeout);
   335→            }
   336→            // FIX: Increased debounce to 500ms to allow Premiere Pro state to settle
   337→            window._selectionChangeTimeout = setTimeout(async () => {
   338→                try {
   339→                    // FIX: Use retries (2 attempts, 300ms delay) to handle timing issues
   340→                    // Previous code used (0, 0) which caused race conditions
   341→                    const hasSequence = await checkSequenceOpen(2, 300);
   342→                    if (hasSequence && !window.spliceState.activeSequence) {
   343→                        const seqInfo = await jsx.call('getActiveSequence');
   344→                        if (seqInfo && !seqInfo.error) {
   345→                            window.spliceState.activeSequence = seqInfo;
   346→                            setStatus('Sequence: ' + (seqInfo.name || 'Unknown'));
   347→                            debugLog('Sequence detected via selection change:', seqInfo.name);
   348→                        }
   349→                    }
   350→                } catch (e) {
   351→                    // FIX: Log errors instead of silently ignoring
   352→                    console.warn('[SPLICE] Selection change handler error:', e.message);
   353→                    debugLog('Selection change failed:', e.message);
   354→                }
   355→            }, 500);
   356→        });
   357→
   358→        debugLog('Sequence event listeners initialized');
   359→    } catch (e) {
   360→        console.error('[SPLICE] Failed to init sequence event listeners:', e.message);
   361→    }
   362→}

   379→/**
   380→ * Check if a sequence is currently open in the Timeline
   381→ * Enhanced with logging and retry logic
   382→ * @param {number} retries - Number of retry attempts (default: 2)
   383→ * @param {number} delayMs - Delay between retries in ms (default: 300)
   384→ * @returns {Promise<boolean>} true if sequence is open
   385→ */
   386→async function checkSequenceOpen(retries = 2, delayMs = 300) {
   387→    console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC START ==========');
   388→    console.log('[SPLICE] checkSequenceOpen: retries=' + retries + ', delayMs=' + delayMs);
   389→    console.log('[SPLICE] checkSequenceOpen: window.spliceState.activeSequence =', window.spliceState.activeSequence);
   390→    
   391→    for (let attempt = 0; attempt <= retries; attempt++) {
   392→        try {
   393→            console.log('[SPLICE] checkSequenceOpen: Attempt ' + (attempt + 1) + ' of ' + (retries + 1));
   394→            console.log('[SPLICE] checkSequenceOpen: Calling jsx.call("checkSequenceOpen")...');
   395→            const result = await jsx.call('checkSequenceOpen');
   396→            console.log('[SPLICE] checkSequenceOpen: JSX call returned');
   397→            console.log('[SPLICE] checkSequenceOpen: result =', result, '(type:', typeof result + ')');
   398→            debugLog('checkSequenceOpen attempt', attempt + 1, 'result:', result, 'type:', typeof result);
   399→
   400→            if (result === true || result === 'true') {
   401→                console.log('[SPLICE] checkSequenceOpen: Returning TRUE - sequence is open');
   402→                console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (TRUE) ==========');
   403→                return true;
   404→            }
   405→
   406→            // If false, wait and retry (sequence might be loading)
   407→            if (attempt < retries) {
   408→                console.log('[SPLICE] checkSequenceOpen: No sequence found, retrying in', delayMs, 'ms...');
   409→                debugLog('checkSequenceOpen: No sequence found, retrying in', delayMs, 'ms...');
   410→                await new Promise(r => setTimeout(r, delayMs));
   411→            }
   412→        } catch (error) {
   413→            console.error('[SPLICE] checkSequenceOpen: Attempt', attempt + 1, 'failed with exception:', error.message);
   414→            console.error('[SPLICE] checkSequenceOpen: Error stack:', error.stack);
   415→
   416→            // On last attempt, return false but log the actual error
   417→            if (attempt === retries) {
   418→                console.error('[SPLICE] checkSequenceOpen: All attempts failed. Last error:', error.message);
   419→                // Store error for diagnostics
   420→                window.spliceState.lastSequenceCheckError = error.message;
   421→                console.log('[SPLICE] checkSequenceOpen: Stored error in window.spliceState.lastSequenceCheckError:', error.message);
   422→                console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (ERROR) ==========');
   423→                return false;
   424→            }
   425→
   426→            // Wait before retry
   427→            console.log('[SPLICE] checkSequenceOpen: Waiting', delayMs, 'ms before retry...');
   428→            await new Promise(r => setTimeout(r, delayMs));
   429→        }
   430→    }
   431→    console.log('[SPLICE] checkSequenceOpen: All attempts completed without finding sequence');
   432→    console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (FALSE) ==========');
   433→    return false;
   434→}

   439→function initOptionsToggle() {
   440→    debugLog('initOptionsToggle - toggle:', !!ui.optionsToggle, 'panel:', !!ui.optionsPanel);
   441→    if (ui.optionsToggle && ui.optionsPanel) {
   442→        ui.optionsToggle.addEventListener('click', () => {
   443→            debugLog('Options toggle clicked');
   444→            // Check current state BEFORE toggling
   445→            const wasCollapsed = ui.optionsPanel.classList.contains('collapsed');
   446→            ui.optionsPanel.classList.toggle('collapsed');
   447→            // If it was collapsed, it's now expanded (show 'expanded' class)
   448→            ui.optionsToggle.classList.toggle('expanded', wasCollapsed);
   449→            saveSettings({ expandedOptions: wasCollapsed });
   450→        });
   451→
   452→        // Restore state
   453→        const settings = getSettings();
   454→        if (!settings.expandedOptions) {
   455→            ui.optionsPanel.classList.add('collapsed');
   456→            ui.optionsToggle.classList.remove('expanded');
   457→        } else {
   458→            ui.optionsToggle.classList.add('expanded');
   459→            ui.optionsPanel.classList.remove('collapsed');
   460→        }
   461→    }
   462→}

   645→function applyPreset(presetId) {
   646→    const preset = PRESETS[presetId];
   647→    if (!preset) return;
   648→
   649→    // Custom preset - keep current settings
   650→    if (presetId === 'custom' || !preset.settings) {
   651→        saveSettings({ activePreset: 'custom' });
   652→        setStatus('Using custom settings');
   653→        return;
   654→    }
   655→
   656→    const settings = preset.settings;
   657→
   658→    // Apply to UI
   659→    if (ui.sensitivitySlider && settings.sensitivity !== undefined) {
   660→        ui.sensitivitySlider.value = settings.sensitivity;
   661→        if (ui.sensitivityValue) ui.sensitivityValue.textContent = settings.sensitivity;
   662→    }
   663→
   664→    if (ui.enableJCut) {
   665→        ui.enableJCut.checked = settings.enableJCut || false;
   666→        if (ui.jcutSettings) {
   667→            ui.jcutSettings.classList.toggle('collapsed', !settings.enableJCut);
   668→        }
   669→    }
   670→
   671→    if (settings.jCutLeadIn !== undefined && ui.jcutLeadIn) {
   672→        ui.jcutLeadIn.value = Math.round(settings.jCutLeadIn * 100);
   673→        if (ui.jcutLeadInValue) ui.jcutLeadInValue.textContent = settings.jCutLeadIn + 's';
   674→    }
   675→
   676→    if (settings.jCutLeadOut !== undefined && ui.jcutLeadOut) {
   677→        ui.jcutLeadOut.value = Math.round(settings.jCutLeadOut * 100);
   678→        if (ui.jcutLeadOutValue) ui.jcutLeadOutValue.textContent = settings.jCutLeadOut + 's';
   679→    }
   680→
   681→    // Save merged settings
   682→    saveSettings({
   683→        activePreset: presetId,
   684→        sensitivity: settings.sensitivity,
   685→        threshold: settings.threshold,
   686→        minSilenceLength: settings.minSilenceLength,
   687→        paddingStart: settings.paddingStart,
   688→        paddingEnd: settings.paddingEnd,
   689→        enableTakesDetection: settings.enableTakesDetection,
   690→        enableJCut: settings.enableJCut || false,
   691→        jcutLeadIn: settings.jCutLeadIn || 0.3,
   692→        jcutLeadOut: settings.jCutLeadOut || 0.2
   693→    });
   694→
   695→    setStatus(`Applied ${preset.name} preset`);
   696→}

   749→async function runDetection() {
   750→    if (isOperationInProgress) return;
   751→
   752→    // Check sequence with enhanced error messaging
   753→    const hasSequence = await checkSequenceOpen();
   754→    if (!hasSequence) {
   755→        // Check if there was a communication error vs just no sequence
   756→        const lastError = window.spliceState.lastSequenceCheckError;
   757→        if (lastError && (lastError.includes('CEP') || lastError.includes('communication') || lastError.includes('evalScript'))) {
   758→            setStatus('Premiere Pro connection issue - try reopening panel', true);
   759→            console.error('[SPLICE] runDetection: JSX communication failure:', lastError);
   760→        } else {
   761→            setStatus('Please open a sequence in the Timeline first', true);
   762→            debugLog('runDetection: No active sequence in Timeline');
   763→        }
   764→        // Clear the error for next check
   765→        window.spliceState.lastSequenceCheckError = null;
   766→        return;
   767→    }
   768→
   769→    // Check online
   770→    if (!isOnline()) {
   771→        setStatus('Offline - Check your connection', true);
   772→        return;
   773→    }
   774→
   775→    isOperationInProgress = true;
   776→    ui.goBtn.disabled = true;
   777→    showProgress('Detecting silences...');
   778→
   779→    try {
   780→        const settings = getSettings();
   781→
   782→        // Get sequence info
   783→        const seqInfo = await jsx.call('getActiveSequence');
   784→        debugLog('Sequence info:', seqInfo);
   785→
   786→        // Store sequence info for later use
   787→        currentSequenceInfo = seqInfo;
   788→        window.spliceState.activeSequence = seqInfo;
   789→        if (seqInfo && seqInfo.videoPath) {
   790→            window.spliceState.currentVideoPath = seqInfo.videoPath;
   791→        }
   792→
   793→        // Export audio via JSX for silence detection
   794→        updateProgress(10, 'Exporting audio...');
   795→
   796→        let audioFilePath = null;
   797→        try {
   798→            // FIX: CEP-CRIT-010 - Main.js Export Timeout
   799→            // Use LONG timeout (120s) for export operations
   800→            const exportResult = await jsx.callWithTimeout(
   801→                'exportSequenceAudioForAnalysis',
   802→                [],
   803→                JSX_TIMEOUTS.LONG
   804→            );
   805→            if (exportResult && exportResult.success && exportResult.outputPath) {
   806→                audioFilePath = exportResult.outputPath;
   807→                debugLog('Audio exported to:', audioFilePath);
   808→            } else if (exportResult && exportResult.error) {
   809→                throw new Error(exportResult.error);
   810→            }
   811→        } catch (exportError) {
   812→            console.warn('[SPLICE] Audio export failed, trying fallback:', exportError.message);
   813→            // Try to get audio from first clip in timeline
   814→            try {
   815→                const clipInfo = await jsx.call('getFirstClipAudioPath');
   816→                if (clipInfo && clipInfo.path) {
   817→                    audioFilePath = clipInfo.path;
   818→                    debugLog('Using clip audio path:', audioFilePath);
   819→                }
   820→            } catch (clipError) {
   821→                console.warn('[SPLICE] Fallback audio path failed:', clipError.message);
   822→            }
   823→        }
   824→
   825→        if (!audioFilePath) {
   826→            throw new Error('Could not export or locate audio file. Please ensure sequence has audio tracks.');
   827→        }
   828→
   829→        // Store audio path globally for other modules
   830→        currentAudioPath = audioFilePath;
   831→        window.spliceState.audioPath = audioFilePath;
   832→
   833→        // Call backend for silence detection with audio file path
   834→        updateProgress(30, 'Detecting silences...');
   835→        const silenceResponse = await fetchWithTimeout(
   836→            `${getBackendUrl()}/silences-rms`,
   837→            {
   838→                method: 'POST',
   839→                headers: getAuthHeaders(),
   840→                body: JSON.stringify({
   841→                    wavPath: audioFilePath,
   842→                    sensitivity: settings.sensitivity,
   843→                    minSilenceLength: settings.minSilenceLength || 0.5
   844→                })
   845→            }
   846→        );
   847→
   848→        if (!silenceResponse.ok) {
   849→            throw new Error(await parseErrorResponse(silenceResponse));
   850→        }
   851→
   852→        const silenceData = await silenceResponse.json();
   853→        previewSilences = silenceData.silences || [];
   854→
   855→        // Detect takes if enabled
   856→        if (settings.enableTakesDetection) {
   857→            updateProgress(60, 'Detecting takes...');
   858→            try {
   859→                const analyzeResponse = await fetchWithTimeout(
   860→                    `${getBackendUrl()}/analyze`,
   861→                    {
   862→                        method: 'POST',
   863→                        headers: getAuthHeaders(),
   864→                        body: JSON.stringify({
   865→                            wavPath: audioFilePath,
   866→                            detectTakes: true
   867→                        })
   868→                    }
   869→                );
   870→
   871→                if (analyzeResponse.ok) {
   872→                    const analyzeData = await analyzeResponse.json();
   873→                    previewTakes = analyzeData.takes || [];
   874→
   875→                    // Store transcript for captions/music modules
   876→                    if (analyzeData.transcript) {
   877→                        currentTranscript = analyzeData.transcript;
   878→                        window.currentTranscript = analyzeData.transcript;
   879→                        window.spliceState.lastTranscript = analyzeData.transcript;
   880→                    }
   881→                } else {
   882→                    // Handle specific error codes
   883→                    const errorMsg = await parseErrorResponse(analyzeResponse);
   884→                    if (analyzeResponse.status === 402) {
   885→                        console.warn('[SPLICE] Takes detection skipped: insufficient credits');
   886→                        setStatus('Takes detection skipped: insufficient credits', true);
   887→                    } else if (analyzeResponse.status === 401) {
   888→                        console.warn('[SPLICE] Takes detection skipped: authentication required');
   889→                    } else {
   890→                        console.warn('[SPLICE] Takes detection failed:', errorMsg);
   891→                    }
   892→                }
   893→            } catch (e) {
   894→                console.warn('[SPLICE] Takes detection failed:', e.message);
   895→            }
   896→        }
   897→
   898→        // Detect profanity if enabled
   899→        if (settings.enableProfanity) {
   900→            updateProgress(65, 'Detecting profanity...');
   901→            try {
   902→                const profanityResponse = await fetchWithTimeout(
   903→                    `${getBackendUrl()}/profanity`,
   904→                    {
   905→                        method: 'POST',
   906→                        headers: getAuthHeaders(),
   907→                        body: JSON.stringify({
   908→                            wavPath: audioFilePath,
   909→                            language: settings.profanityLanguage || 'en',
   910→                            bleepType: settings.bleepType || 'standard'
   911→                        })
   912→                    }
   913→                );
   914→
   915→                if (profanityResponse.ok) {
   916→                    const profanityData = await profanityResponse.json();
   917→                    // Backend returns 'words' array, not 'instances'
   918→                    currentProfanity = profanityData.words || profanityData.instances || [];
   919→                    if (currentProfanity.length > 0) {
   920→                        debugLog(`Found ${currentProfanity.length} profanity instances`);
   921→                    }
   922→                } else {
   923→                    console.warn('[SPLICE] Profanity detection failed:', await parseErrorResponse(profanityResponse));
   924→                }
   925→            } catch (e) {
   926→                console.warn('[SPLICE] Profanity detection failed:', e.message);
   927→            }
   928→        }
   929→
   930→        // Detect chapters if enabled
   931→        if (settings.enableChapters) {
   932→            updateProgress(80, 'Detecting chapters...');
   933→            try {
   934→                const chapterResponse = await fetchWithTimeout(
   935→                    `${getBackendUrl()}/chapters`,
   936→                    {
   937→                        method: 'POST',
   938→                        headers: getAuthHeaders(),
   939→                        body: JSON.stringify({
   940→                            wavPath: audioFilePath,
   941→                            transcript: currentTranscript,
   942→                            maxChapters: settings.maxChapters || 10,
   943→                            minChapterLength: settings.minChapterLength || 60
   944→                        })
   945→                    }
   946→                );
   947→
   948→                if (chapterResponse.ok) {
   949→                    const chapterData = await chapterResponse.json();
   950→                    currentChapters = chapterData.chapters || [];
   951→                } else {
   952→                    const errorMsg = await parseErrorResponse(chapterResponse);
   953→                    console.warn('[SPLICE] Chapter detection failed:', errorMsg);
   954→                    // Don't block workflow, just notify user
   955→                    setStatus('Chapter detection skipped: ' + errorMsg, true);
   956→                }
   957→            } catch (e) {
   958→                console.warn('[SPLICE] Chapter detection failed:', e);
   959→                setStatus('Chapter detection failed - continuing without chapters', true);
   960→            }
   961→        }
   962→
   963→        // Detect zoom points if enabled
   964→        if (settings.enableZoom) {
   965→            updateProgress(90, 'Detecting zoom points...');
   966→            try {
   967→                const zoomResponse = await fetchWithTimeout(
   968→                    `${getBackendUrl()}/zoom`,
   969→                    {
   970→                        method: 'POST',
   971→                        headers: getAuthHeaders(),
   972→                        body: JSON.stringify({
   973→                            wavPath: audioFilePath,
   974→                            transcript: currentTranscript,
   975→                            frequency: settings.zoomFrequency || 'medium',
   976→                            preset: settings.zoomPreset || 'medium',
   977→                            placement: settings.zoomPlacement || 'sentence_start'
   978→                        })
   979→                    }
   980→                );
   981→
   982→                if (zoomResponse.ok) {
   983→                    const zoomData = await zoomResponse.json();
   984→                    currentZooms = zoomData.zoomPoints || [];
   985→                } else {
   986→                    const errorMsg = await parseErrorResponse(zoomResponse);
   987→                    console.warn('[SPLICE] Zoom detection failed:', errorMsg);
   988→                    setStatus('Zoom detection skipped: ' + errorMsg, true);
   989→                }
   990→            } catch (e) {
   991→                console.warn('[SPLICE] Zoom detection failed:', e);
   992→                setStatus('Zoom detection failed - continuing without zoom points', true);
   993→            }
   994→        }
   995→
   996→        // Store results
   997→        lastDetectionResult = {
   998→            silences: previewSilences,
   999→            takes: previewTakes,
  1000→            chapters: currentChapters,
  1001→            zooms: currentZooms
  1002→        };
  1003→
  1004→        // Show preview
  1005→        hideProgress();
  1006→        showCombinedPreview();
  1007→        setStatus(`Found ${previewSilences.length} silences, ${previewTakes.length} takes`);
  1008→
  1009→    } catch (error) {
  1010→        console.error('[SPLICE] Detection error:', error);
  1011→        setStatus('Detection failed: ' + error.message, true);
  1012→        hideProgress();
  1013→    } finally {
  1014→        isOperationInProgress = false;
  1015→        ui.goBtn.disabled = false;
  1016→    }
  1017→}

  1145→async function seekToTime(seconds) {
  1146→    try {
  1147→        // FIX: Removed invalid 'app' check - 'app' is ExtendScript DOM, not available in panel JS
  1148→        // The JSX evalScript will handle the error case if no sequence is open
  1149→

  1296→async function buildSequence() {
  1297→    if (isOperationInProgress) return;
  1298→
  1299→    isOperationInProgress = true;
  1300→    setStatus('Building sequence...');
  1301→
  1302→    try {
  1303→        const settings = getSettings();
  1304→
  1305→        // Ensure we have sequence info
  1306→        if (!currentSequenceInfo) {
  1307→            currentSequenceInfo = await jsx.call('getActiveSequence');
  1308→        }
  1309→
  1310→        // Create cut list from selected silences
  1311→        const selectedSilences = previewSilences.filter((_, i) => selectedSilenceIndices.has(i));
  1312→
  1313→        // Calculate total duration from sequence info or last silence
  1314→        let duration = 0;
  1315→        if (currentSequenceInfo && currentSequenceInfo.duration) {
  1316→            duration = currentSequenceInfo.duration;
  1317→        } else if (previewSilences.length > 0) {
  1318→            const lastSilence = previewSilences[previewSilences.length - 1];
  1319→            duration = lastSilence.end + 10; // Add buffer
  1320→        }
  1321→
  1322→        // Call backend to generate cut list
  1323→        const cutListResponse = await fetchWithTimeout(
  1324→            `${getBackendUrl()}/cut-list`,
  1325→            {
  1326→                method: 'POST',
  1327→                headers: getAuthHeaders(),
  1328→                body: JSON.stringify({
  1329→                    sourceName: currentSequenceInfo?.name || 'Untitled Sequence',
  1330→                    sourcePath: currentSequenceInfo?.treePath || currentSequenceInfo?.name || 'Untitled',
  1331→                    duration: duration,
  1332→                    silences: selectedSilences,
  1333→                    takes: previewTakes,
  1334→                    enableTakes: settings.enableTakesDetection,
  1335→                    jCutOffset: settings.enableJCut ? settings.jcutLeadIn : 0,
  1336→                    lCutOffset: settings.enableJCut ? settings.jcutLeadOut : 0
  1337→                })
  1338→            }
  1339→        );
  1340→
  1341→        if (!cutListResponse.ok) {
  1342→            throw new Error(await parseErrorResponse(cutListResponse));
  1343→        }
  1344→
  1345→        const cutList = await cutListResponse.json();
  1346→
  1347→        // Build sequence via JSX
  1348→        const result = await jsx.call('buildSequenceFromCutList', JSON.stringify(cutList));
  1349→
  1350→        if (result && result.success) {
  1351→            setStatus(`Built sequence: ${result.sequenceName}`);
  1352→            hidePreview();
  1353→        } else {
  1354→            throw new Error(result?.error || 'Build failed');
  1355→        }
  1356→    } catch (error) {
  1357→        setStatus('Build failed: ' + error.message, true);
  1358→    } finally {
  1359→        isOperationInProgress = false;
  1360→    }
  1361→}
