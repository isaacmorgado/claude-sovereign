   239→        'goBtn': { el: ui.goBtn, desc: 'GO button (main action)' },
   240→        'settingsBtn': { el: ui.settingsBtn, desc: 'Settings button' },
   241→        'debugBtn': { el: ui.debugBtn, desc: 'Debug button' },
   242→        'creditBadge': { el: ui.creditBadge, desc: 'Credit badge' },
   243→        'status': { el: ui.status, desc: 'Status bar' },
   244→        'optionsToggle': { el: ui.optionsToggle, desc: 'Options toggle' },
   245→        'optionsPanel': { el: ui.optionsPanel, desc: 'Options panel' }
   246→    };
   247→
   248→    const missingElements = [];
   249→    const presentElements = [];
   250→
   251→    Object.entries(criticalElements).forEach(([id, { el, desc }]) => {
   252→        if (!el) {
   253→            missingElements.push(`#${id} (${desc})`);
   254→        } else {
   255→            presentElements.push(id);
   256→        }
   257→    });
   258→
   259→    if (missingElements.length > 0) {
   260→        console.error('[SPLICE] ⚠️ UI INITIALIZATION INCOMPLETE');
   261→        console.error('[SPLICE] Missing critical elements:', missingElements);
   262→        console.error('[SPLICE] Panel may not function correctly. Check index.html structure.');
   263→
   264→        // Show error in status bar if available
   265→        const statusEl = document.getElementById('status');
   266→        if (statusEl) {
   267→            statusEl.textContent = 'Init error: ' + missingElements.length + ' UI elements missing';
   268→            statusEl.classList.add('error');
   269→        }
   270→    } else {
   271→        console.log('[SPLICE] ✓ All critical UI elements initialized:', presentElements.join(', '));
   272→    }
   273→
   274→    // Verify GO button has click listener
   275→    if (ui.goBtn) {
   276→        const listeners = getEventListeners ? getEventListeners(ui.goBtn) : null;
   277→        if (!listeners || !listeners.click || listeners.click.length === 0) {
   278→            // Can't check in production, but log that we tried
   279→            console.log('[SPLICE] GO button present - click handler should be attached');

   287→/**
   288→ * Initialize event listeners for sequence state changes
   289→ * This provides real-time updates when sequences are activated/deactivated
   290→ */
   291→function initSequenceEventListeners() {
   292→    if (!jsx.cs) {
   293→        console.warn('[SPLICE] Cannot init sequence listeners - CSInterface not available');
   294→        return;
   295→    }
   296→
   297→    try {
   298→        // Listen for sequence activation events
   299→        jsx.cs.addEventListener('com.adobe.csxs.events.SequenceActivated', async (event) => {
   300→            debugLog('Sequence activated event received');
   301→            try {
   302→                const seqInfo = await jsx.call('getActiveSequence');
   303→                if (seqInfo && !seqInfo.error) {
   304→                    window.spliceState.activeSequence = seqInfo;
   305→                    setStatus('Sequence: ' + (seqInfo.name || 'Unknown'));
   306→                    debugLog('Active sequence updated:', seqInfo.name);
   307→                }
   308→            } catch (e) {
   309→                console.warn('[SPLICE] Failed to get sequence info on activation:', e.message);
   310→            }
   311→        });
   312→
   313→        // Listen for sequence close/deactivation
   314→        jsx.cs.addEventListener('com.adobe.csxs.events.SequenceDeactivated', () => {
   315→            debugLog('Sequence deactivated event');
   316→            window.spliceState.activeSequence = null;
   317→            setStatus('No sequence open', true);
   318→        });
   319→
   320→        // Listen for project item selection changes (includes sequences)
   321→        jsx.cs.addEventListener('com.adobe.csxs.events.SelectionChanged', async () => {
   322→            // Debounce rapid selection changes
   323→            if (window._selectionChangeTimeout) {
   324→                clearTimeout(window._selectionChangeTimeout);
   325→            }
   326→            // FIX: Increased debounce to 500ms to allow Premiere Pro state to settle
   327→            window._selectionChangeTimeout = setTimeout(async () => {
   328→                try {
   329→                    // FIX: Use retries (2 attempts, 300ms delay) to handle timing issues
   330→                    // Previous code used (0, 0) which caused race conditions
   331→                    const hasSequence = await checkSequenceOpen(2, 300);
   332→                    if (hasSequence && !window.spliceState.activeSequence) {
   333→                        const seqInfo = await jsx.call('getActiveSequence');
   334→                        if (seqInfo && !seqInfo.error) {
   335→                            window.spliceState.activeSequence = seqInfo;
   336→                            setStatus('Sequence: ' + (seqInfo.name || 'Unknown'));
   337→                            debugLog('Sequence detected via selection change:', seqInfo.name);
   338→                        }
   339→                    }
   340→                } catch (e) {
   341→                    // FIX: Log errors instead of silently ignoring
   342→                    console.warn('[SPLICE] Selection change handler error:', e.message);
   343→                    debugLog('Selection change failed:', e.message);
   344→                }
   345→            }, 500);
   346→        });
   347→
   348→        debugLog('Sequence event listeners initialized');
   349→    } catch (e) {
   350→        console.error('[SPLICE] Failed to init sequence event listeners:', e.message);
   351→    }
   352→}

   369→/**
   370→ * Check if a sequence is currently open in the Timeline
   371→ * Enhanced with logging and retry logic
   372→ * @param {number} retries - Number of retry attempts (default: 2)
   373→ * @param {number} delayMs - Delay between retries in ms (default: 300)
   374→ * @returns {Promise<boolean>} true if sequence is open
   375→ */
   376→async function checkSequenceOpen(retries = 2, delayMs = 300) {
   377→    console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC START ==========');
   378→    console.log('[SPLICE] checkSequenceOpen: retries=' + retries + ', delayMs=' + delayMs);
   379→    console.log('[SPLICE] checkSequenceOpen: window.spliceState.activeSequence =', window.spliceState.activeSequence);
   380→    
   381→    for (let attempt = 0; attempt <= retries; attempt++) {
   382→        try {
   383→            console.log('[SPLICE] checkSequenceOpen: Attempt ' + (attempt + 1) + ' of ' + (retries + 1));
   384→            console.log('[SPLICE] checkSequenceOpen: Calling jsx.call("checkSequenceOpen")...');
   385→            const result = await jsx.call('checkSequenceOpen');
   386→            console.log('[SPLICE] checkSequenceOpen: JSX call returned');
   387→            console.log('[SPLICE] checkSequenceOpen: result =', result, '(type:', typeof result + ')');
   388→            debugLog('checkSequenceOpen attempt', attempt + 1, 'result:', result, 'type:', typeof result);
   389→
   390→            if (result === true || result === 'true') {
   391→                console.log('[SPLICE] checkSequenceOpen: Returning TRUE - sequence is open');
   392→                console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (TRUE) ==========');
   393→                return true;
   394→            }
   395→
   396→            // If false, wait and retry (sequence might be loading)
   397→            if (attempt < retries) {
   398→                console.log('[SPLICE] checkSequenceOpen: No sequence found, retrying in', delayMs, 'ms...');
   399→                debugLog('checkSequenceOpen: No sequence found, retrying in', delayMs, 'ms...');
   400→                await new Promise(r => setTimeout(r, delayMs));
   401→            }
   402→        } catch (error) {
   403→            console.error('[SPLICE] checkSequenceOpen: Attempt', attempt + 1, 'failed with exception:', error.message);
   404→            console.error('[SPLICE] checkSequenceOpen: Error stack:', error.stack);
   405→
   406→            // On last attempt, return false but log the actual error
   407→            if (attempt === retries) {
   408→                console.error('[SPLICE] checkSequenceOpen: All attempts failed. Last error:', error.message);
   409→                // Store error for diagnostics
   410→                window.spliceState.lastSequenceCheckError = error.message;
   411→                console.log('[SPLICE] checkSequenceOpen: Stored error in window.spliceState.lastSequenceCheckError:', error.message);
   412→                console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (ERROR) ==========');
   413→                return false;
   414→            }
   415→
   416→            // Wait before retry
   417→            console.log('[SPLICE] checkSequenceOpen: Waiting', delayMs, 'ms before retry...');
   418→            await new Promise(r => setTimeout(r, delayMs));
   419→        }
   420→    }
   421→    console.log('[SPLICE] checkSequenceOpen: All attempts completed without finding sequence');
   422→    console.log('[SPLICE] ========== checkSequenceOpen DIAGNOSTIC END (FALSE) ==========');
   423→    return false;
   424→}

   429→function initOptionsToggle() {
   430→    debugLog('initOptionsToggle - toggle:', !!ui.optionsToggle, 'panel:', !!ui.optionsPanel);
   431→    if (ui.optionsToggle && ui.optionsPanel) {
   432→        ui.optionsToggle.addEventListener('click', () => {
   433→            debugLog('Options toggle clicked');
   434→            // Check current state BEFORE toggling
   435→            const wasCollapsed = ui.optionsPanel.classList.contains('collapsed');
   436→            ui.optionsPanel.classList.toggle('collapsed');
   437→            // If it was collapsed, it's now expanded (show 'expanded' class)
   438→            ui.optionsToggle.classList.toggle('expanded', wasCollapsed);
   439→            saveSettings({ expandedOptions: wasCollapsed });
   440→        });
   441→
   442→        // Restore state
   443→        const settings = getSettings();
   444→        if (!settings.expandedOptions) {
   445→            ui.optionsPanel.classList.add('collapsed');
   446→            ui.optionsToggle.classList.remove('expanded');
   447→        } else {
   448→            ui.optionsToggle.classList.add('expanded');
   449→            ui.optionsPanel.classList.remove('collapsed');
   450→        }
   451→    }
   452→}

   663→        if (ui.jcutLeadInValue) ui.jcutLeadInValue.textContent = settings.jCutLeadIn + 's';

   668→        if (ui.jcutLeadOutValue) ui.jcutLeadOutValue.textContent = settings.jCutLeadOut + 's';
   669→    }
   670→
   671→    // Save merged settings
   672→    saveSettings({
   673→        activePreset: presetId,
   674→        sensitivity: settings.sensitivity,
   675→        threshold: settings.threshold,
   676→        minSilenceLength: settings.minSilenceLength,
   677→        paddingStart: settings.paddingStart,
   678→        paddingEnd: settings.paddingEnd,
   679→        enableTakesDetection: settings.enableTakesDetection,

   682→        jcutLeadOut: settings.jCutLeadOut || 0.2

   685→    setStatus(`Applied ${preset.name} preset`);
   686→}

   739→async function runDetection() {
   740→    if (isOperationInProgress) return;
   741→
   742→    // Check sequence with enhanced error messaging
   743→    const hasSequence = await checkSequenceOpen();
   744→    if (!hasSequence) {
   745→        // Check if there was a communication error vs just no sequence
   746→        const lastError = window.spliceState.lastSequenceCheckError;
   747→        if (lastError && (lastError.includes('CEP') || lastError.includes('communication') || lastError.includes('evalScript'))) {
   748→            setStatus('Premiere Pro connection issue - try reopening panel', true);
   749→            console.error('[SPLICE] runDetection: JSX communication failure:', lastError);
   750→        } else {
   751→            setStatus('Please open a sequence in the Timeline first', true);
   752→            debugLog('runDetection: No active sequence in Timeline');
   753→        }
   754→        // Clear the error for next check
   755→        window.spliceState.lastSequenceCheckError = null;
   756→        return;
   757→    }
   758→
   759→    // Check online
   760→    if (!isOnline()) {
   761→        setStatus('Offline - Check your connection', true);
   762→        return;
   763→    }
   764→
   765→    isOperationInProgress = true;
   766→    ui.goBtn.disabled = true;
   767→    showProgress('Detecting silences...');
   768→
   769→    try {
   770→        const settings = getSettings();
   771→
   772→        // Get sequence info
   773→        const seqInfo = await jsx.call('getActiveSequence');
   774→        debugLog('Sequence info:', seqInfo);
   775→
   776→        // Store sequence info for later use
   777→        currentSequenceInfo = seqInfo;
   778→        window.spliceState.activeSequence = seqInfo;
   779→        if (seqInfo && seqInfo.videoPath) {
   780→            window.spliceState.currentVideoPath = seqInfo.videoPath;
   781→        }
   782→
   783→        // Export audio via JSX for silence detection
   784→        updateProgress(10, 'Exporting audio...');
   785→
   786→        let audioFilePath = null;
   787→        try {
   788→            // FIX: CEP-CRIT-010 - Main.js Export Timeout
   789→            // Use LONG timeout (120s) for export operations
   790→            const exportResult = await jsx.callWithTimeout(
   791→                'exportSequenceAudioForAnalysis',
   792→                [],
   793→                JSX_TIMEOUTS.LONG
   794→            );
   795→            if (exportResult && exportResult.success && exportResult.outputPath) {
   796→                audioFilePath = exportResult.outputPath;
   797→                debugLog('Audio exported to:', audioFilePath);
   798→            } else if (exportResult && exportResult.error) {
   799→                throw new Error(exportResult.error);
   800→            }
   801→        } catch (exportError) {
   802→            console.warn('[SPLICE] Audio export failed, trying fallback:', exportError.message);
   803→            // Try to get audio from first clip in timeline
   804→            try {
   805→                const clipInfo = await jsx.call('getFirstClipAudioPath');
   806→                if (clipInfo && clipInfo.path) {
   807→                    audioFilePath = clipInfo.path;
   808→                    debugLog('Using clip audio path:', audioFilePath);
   809→                }
   810→            } catch (clipError) {
   811→                console.warn('[SPLICE] Fallback audio path failed:', clipError.message);
   812→            }
   813→        }
   814→
   815→        if (!audioFilePath) {
   816→            throw new Error('Could not export or locate audio file. Please ensure sequence has audio tracks.');
   817→        }
   818→
   819→        // Store audio path globally for other modules
   820→        currentAudioPath = audioFilePath;
   821→        window.spliceState.audioPath = audioFilePath;
   822→
   823→        // Call backend for silence detection with audio file path
   824→        updateProgress(30, 'Detecting silences...');
   825→        const silenceResponse = await fetchWithTimeout(
   826→            `${getBackendUrl()}/silences-rms`,
   827→            {
   828→                method: 'POST',
   829→                headers: getAuthHeaders(),
   830→                body: JSON.stringify({
   831→                    wavPath: audioFilePath,
   832→                    sensitivity: settings.sensitivity,
   833→                    minSilenceLength: settings.minSilenceLength || 0.5
   834→                })
   835→            }
   836→        );
   837→
   838→        if (!silenceResponse.ok) {
   839→            throw new Error(await parseErrorResponse(silenceResponse));
   840→        }
   841→
   842→        const silenceData = await silenceResponse.json();
   843→        previewSilences = silenceData.silences || [];
   844→
   845→        // Detect takes if enabled
   846→        if (settings.enableTakesDetection) {
   847→            updateProgress(60, 'Detecting takes...');
   848→            try {
   849→                const analyzeResponse = await fetchWithTimeout(
   850→                    `${getBackendUrl()}/analyze`,
   851→                    {
   852→                        method: 'POST',
   853→                        headers: getAuthHeaders(),
   854→                        body: JSON.stringify({
   855→                            wavPath: audioFilePath,
   856→                            detectTakes: true
   857→                        })
   858→                    }
   859→                );
   860→
   861→                if (analyzeResponse.ok) {
   862→                    const analyzeData = await analyzeResponse.json();
   863→                    previewTakes = analyzeData.takes || [];
   864→
   865→                    // Store transcript for captions/music modules
   866→                    if (analyzeData.transcript) {
   867→                        currentTranscript = analyzeData.transcript;
   868→                        window.currentTranscript = analyzeData.transcript;
   869→                        window.spliceState.lastTranscript = analyzeData.transcript;
   870→                    }
   871→                } else {
   872→                    // Handle specific error codes
   873→                    const errorMsg = await parseErrorResponse(analyzeResponse);
   874→                    if (analyzeResponse.status === 402) {
   875→                        console.warn('[SPLICE] Takes detection skipped: insufficient credits');
   876→                        setStatus('Takes detection skipped: insufficient credits', true);
   877→                    } else if (analyzeResponse.status === 401) {
   878→                        console.warn('[SPLICE] Takes detection skipped: authentication required');
   879→                    } else {
   880→                        console.warn('[SPLICE] Takes detection failed:', errorMsg);
   881→                    }
   882→                }
   883→            } catch (e) {
   884→                console.warn('[SPLICE] Takes detection failed:', e.message);
   885→            }
   886→        }
   887→
   888→        // Detect profanity if enabled
   889→        if (settings.enableProfanity) {
   890→            updateProgress(65, 'Detecting profanity...');
   891→            try {
   892→                const profanityResponse = await fetchWithTimeout(
   893→                    `${getBackendUrl()}/profanity`,
   894→                    {
   895→                        method: 'POST',
   896→                        headers: getAuthHeaders(),
   897→                        body: JSON.stringify({
   898→                            wavPath: audioFilePath,
   899→                            language: settings.profanityLanguage || 'en',
   900→                            bleepType: settings.bleepType || 'standard'
   901→                        })
   902→                    }
   903→                );
   904→
   905→                if (profanityResponse.ok) {
   906→                    const profanityData = await profanityResponse.json();
   907→                    // Backend returns 'words' array, not 'instances'
   908→                    currentProfanity = profanityData.words || profanityData.instances || [];
   909→                    if (currentProfanity.length > 0) {
   910→                        debugLog(`Found ${currentProfanity.length} profanity instances`);
   911→                    }
   912→                } else {
   913→                    console.warn('[SPLICE] Profanity detection failed:', await parseErrorResponse(profanityResponse));
   914→                }
   915→            } catch (e) {
   916→                console.warn('[SPLICE] Profanity detection failed:', e.message);
   917→            }
   918→        }
   919→
   920→        // Detect chapters if enabled
   921→        if (settings.enableChapters) {
   922→            updateProgress(80, 'Detecting chapters...');
   923→            try {
   924→                const chapterResponse = await fetchWithTimeout(
   925→                    `${getBackendUrl()}/chapters`,
   926→                    {
   927→                        method: 'POST',
   928→                        headers: getAuthHeaders(),
   929→                        body: JSON.stringify({
   930→                            wavPath: audioFilePath,
   931→                            transcript: currentTranscript,
   932→                            maxChapters: settings.maxChapters || 10,
   933→                            minChapterLength: settings.minChapterLength || 60
   934→                        })
   935→                    }
   936→                );
   937→
   938→                if (chapterResponse.ok) {
   939→                    const chapterData = await chapterResponse.json();
   940→                    currentChapters = chapterData.chapters || [];
   941→                } else {
   942→                    const errorMsg = await parseErrorResponse(chapterResponse);
   943→                    console.warn('[SPLICE] Chapter detection failed:', errorMsg);
   944→                    // Don't block workflow, just notify user
   945→                    setStatus('Chapter detection skipped: ' + errorMsg, true);
   946→                }
   947→            } catch (e) {
   948→                console.warn('[SPLICE] Chapter detection failed:', e);
   949→                setStatus('Chapter detection failed - continuing without chapters', true);
   950→            }
   951→        }
   952→
   953→        // Detect zoom points if enabled
   954→        if (settings.enableZoom) {
   955→            updateProgress(90, 'Detecting zoom points...');
   956→            try {
   957→                const zoomResponse = await fetchWithTimeout(
   958→                    `${getBackendUrl()}/zoom`,
   959→                    {
   960→                        method: 'POST',
   961→                        headers: getAuthHeaders(),
   962→                        body: JSON.stringify({
   963→                            wavPath: audioFilePath,
   964→                            transcript: currentTranscript,
   965→                            frequency: settings.zoomFrequency || 'medium',
   966→                            preset: settings.zoomPreset || 'medium',
   967→                            placement: settings.zoomPlacement || 'sentence_start'
   968→                        })
   969→                    }
   970→                );
   971→
   972→                if (zoomResponse.ok) {
   973→                    const zoomData = await zoomResponse.json();
   974→                    currentZooms = zoomData.zoomPoints || [];
   975→                } else {
   976→                    const errorMsg = await parseErrorResponse(zoomResponse);
   977→                    console.warn('[SPLICE] Zoom detection failed:', errorMsg);
   978→                    setStatus('Zoom detection skipped: ' + errorMsg, true);
   979→                }
   980→            } catch (e) {
   981→                console.warn('[SPLICE] Zoom detection failed:', e);
   982→                setStatus('Zoom detection failed - continuing without zoom points', true);
   983→            }
   984→        }
   985→
   986→        // Store results
   987→        lastDetectionResult = {
   988→            silences: previewSilences,
   989→            takes: previewTakes,
   990→            chapters: currentChapters,
   991→            zooms: currentZooms
   992→        };
   993→
   994→        // Show preview
   995→        hideProgress();
   996→        showCombinedPreview();
   997→        setStatus(`Found ${previewSilences.length} silences, ${previewTakes.length} takes`);
   998→
   999→    } catch (error) {
  1000→        console.error('[SPLICE] Detection error:', error);
  1001→        setStatus('Detection failed: ' + error.message, true);
  1002→        hideProgress();
  1003→    } finally {
  1004→        isOperationInProgress = false;
  1005→        ui.goBtn.disabled = false;
  1006→    }
  1007→}

  1135→async function seekToTime(seconds) {
  1136→    try {
  1137→        // FIX: Removed invalid 'app' check - 'app' is ExtendScript DOM, not available in panel JS
  1138→        // The JSX evalScript will handle the error case if no sequence is open
  1139→
  1140→        // Use JSX to set playhead position
  1141→        // Note: setPlayerPosition expects ticks as a string without quotes around the number value
  1142→        const ticks = Math.round(seconds * 254016000000);
  1143→        const result = await jsx.evalScript(`
  1144→            (function() {
  1145→                try {
  1146→                    if (!app.project.activeSequence) {
  1147→                        return JSON.stringify({ error: 'No active sequence' });
  1148→                    }
  1149→                    app.project.activeSequence.setPlayerPosition('${ticks}');
  1150→                    return JSON.stringify({ success: true });

  1286→async function buildSequence() {
  1287→    if (isOperationInProgress) return;
  1288→
  1289→    isOperationInProgress = true;
  1290→    setStatus('Building sequence...');
  1291→
  1292→    try {
  1293→        const settings = getSettings();
  1294→
  1295→        // Ensure we have sequence info
  1296→        if (!currentSequenceInfo) {
  1297→            currentSequenceInfo = await jsx.call('getActiveSequence');
  1298→        }
  1299→
  1300→        // Create cut list from selected silences
  1301→        const selectedSilences = previewSilences.filter((_, i) => selectedSilenceIndices.has(i));
  1302→
  1303→        // Calculate total duration from sequence info or last silence
  1304→        let duration = 0;
  1305→        if (currentSequenceInfo && currentSequenceInfo.duration) {
  1306→            duration = currentSequenceInfo.duration;
  1307→        } else if (previewSilences.length > 0) {
  1308→            const lastSilence = previewSilences[previewSilences.length - 1];
  1309→            duration = lastSilence.end + 10; // Add buffer
  1310→        }
  1311→
  1312→        // Call backend to generate cut list
  1313→        const cutListResponse = await fetchWithTimeout(
  1314→            `${getBackendUrl()}/cut-list`,
  1315→            {
  1316→                method: 'POST',
  1317→                headers: getAuthHeaders(),
  1318→                body: JSON.stringify({
  1319→                    sourceName: currentSequenceInfo?.name || 'Untitled Sequence',
  1320→                    sourcePath: currentSequenceInfo?.treePath || currentSequenceInfo?.name || 'Untitled',
  1321→                    duration: duration,
  1322→                    silences: selectedSilences,
  1323→                    takes: previewTakes,
  1324→                    enableTakes: settings.enableTakesDetection,
  1325→                    jCutOffset: settings.enableJCut ? settings.jcutLeadIn : 0,
  1326→                    lCutOffset: settings.enableJCut ? settings.jcutLeadOut : 0
  1327→                })
  1328→            }
  1329→        );
  1330→
  1331→        if (!cutListResponse.ok) {
  1332→            throw new Error(await parseErrorResponse(cutListResponse));
  1333→        }
  1334→
  1335→        const cutList = await cutListResponse.json();
  1336→
  1337→        // Build sequence via JSX
  1338→        const result = await jsx.call('buildSequenceFromCutList', JSON.stringify(cutList));
  1339→
  1340→        if (result && result.success) {
  1341→            setStatus(`Built sequence: ${result.sequenceName}`);
  1342→            hidePreview();
  1343→        } else {
  1344→            throw new Error(result?.error || 'Build failed');
  1345→        }
  1346→    } catch (error) {
  1347→        setStatus('Build failed: ' + error.message, true);
  1348→    } finally {
  1349→        isOperationInProgress = false;
  1350→    }
  1351→}
