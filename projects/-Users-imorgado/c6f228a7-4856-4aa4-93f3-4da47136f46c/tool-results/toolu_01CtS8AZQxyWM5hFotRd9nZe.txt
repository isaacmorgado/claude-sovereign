     1→import { NextRequest, NextResponse } from "next/server";
     2→import { getBeautifyPrompt } from "@/lib/prompts";
     3→import { runClaudeCommand, MAX_CODE_SIZE, TIMEOUT_MS } from "@/lib/claude";
     4→import { processCodeWithSourceMap } from "@/lib/source-map";
     5→import { stripMarkdownFences } from "@/lib/utils";
     6→
     7→export async function POST(request: NextRequest) {
     8→  let body;
     9→  try {
    10→    body = await request.json();
    11→  } catch {
    12→    return NextResponse.json({ error: "Invalid request body" }, { status: 400 });
    13→  }
    14→
    15→  try {
    16→    const { code } = body;
    17→
    18→    if (!code || typeof code !== "string") {
    19→      return NextResponse.json({ error: "Code is required" }, { status: 400 });
    20→    }
    21→
    22→    if (code.length > MAX_CODE_SIZE) {
    23→      return NextResponse.json(
    24→        { error: `Code exceeds maximum size of ${MAX_CODE_SIZE / 1024}KB` },
    25→        { status: 400 }
    26→      );
    27→    }
    28→
    29→    // Detect and parse inline source map (errors handled internally, never throws)
    30→    const sourceMapResult = processCodeWithSourceMap(code);
    31→
    32→    // Build prompt with variable hints if available
    33→    const prompt = getBeautifyPrompt(sourceMapResult.variableHints);
    34→
    35→    const rawResult = await runClaudeCommand(prompt, code, TIMEOUT_MS);
    36→    const result = stripMarkdownFences(rawResult);
    37→
    38→    return NextResponse.json({
    39→      result,
    40→      sourceMapDetected: sourceMapResult.hasSourceMap,
    41→      originalSources: sourceMapResult.originalSources,
    42→      externalSourceMapUrl: sourceMapResult.externalSourceMapUrl,
    43→    });
    44→  } catch (error) {
    45→    console.error("Beautify error:", error);
    46→    return NextResponse.json(
    47→      { error: error instanceof Error ? error.message : "Failed to beautify code" },
    48→      { status: 500 }
    49→    );
    50→  }
    51→}
